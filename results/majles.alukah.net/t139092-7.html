    ــــ حَذْفٌ :
    الحذفُ : تركُ الحرفِ لِمُقْتَضٍ و يكونُ الحذفُ علامةً للجزمِ في موضعينِ :
    1ـ الفعلُ المضارعُ المعتلُّ الآخرِ وهو ما كانَ آخِرُهُ أَلِفًا . مثل : يَخْشَى ، أوْ وَاوًا مثل : يَدْعُو ، أوْ يَاءً مثل :يَرْمِي.
    فعندَ دُخولِ الجازِمِ يُحْذَفُ حرفُ العلّةِ فَيُقَالُ : لَمْ يَخْشَ ، لَمْ يَدْعُ ، لَمْ يَرْمِ :
    ـــ يَخْشَ : فعلٌ مضارعٌ مجزومٌ بِلَمْ و علامةُ جزمهِ حذفُ الأَلِفِ و الفتحةُ قبلها دليلٌ عليها .
    ـــ يَدعُ : فعلٌ مضارعٌ مجزومٌ بِلَمْ و علامةُ جزمهِ حذفُ الوَاوِ و الضَمَّةُ قبلها دليلٌ عليها .
    ـــ يَرْمِ : فعلٌ مضارعٌ مجزومٌ بِلَمْ و علامةُ جزمهِ حذفُ اليَاءِ و الكسرةُ قبلها دليلٌ عليها .
    2ـ الأَفْعَالُ الخَمْسَةُ : وهي : تَفْعَلاَنِ ــ يَفْعَلاَنِ ــ تَفْعَلُونَ ــ يَفْعَلُونَ و يَفْعَلْنَ .
    فكلُّ فعلٍ من هذه الأفعالُ يُجْزَمُ بِحَذْفِ النُّونِ . مثل : يَكْتُبَانِ إذَا أَدْخَلْتَ عليهِ الجَازِمَ قُلْتَ : لَمْ يَكْتُبَا :
    ــ لَمْ : حَرْفٌ جَازِمٌ .
    ـــ يَكْتُبَا : فعلٌ مضارعٌ مجزومٌ بِلَمْ و علامةُ جزمهِ حَذْفُ النُّونِ .
    ــ الأَلِفُ ( ا ) : فَاعِلٌ .

    ــــ حَرْفٌ :
    الحرفُ كلمةٌ دلّتْ على معنى في غيرها . مثل : لَمْ في قولك : لَمْ يَضْرِبْ ، فإنَّ لَمْ معناها النّفيُ و لم يظهرْ في الفعلِ بعدها . وهو ثلاثة أقسامٍ :
    1 ــ حرفٌ مشتركٌ بينَ الأسماءِ والأفعالِ . مثل : هَلْ ، تقولُ : هَلْ قَامَ أَحْمَدُ ، و هَلْ أَحْمَدُ قَائِمٌ ، فهَلْ في المثالِ الأوّلِ داخلةٌ على الفعلِ وهو قَامَ ، وفي الثاني داخلةٌ على الاسْمِ وهو أَحْمَدُ .
    2 ــ حرفٌ مختصٌّ بالأسماءِ . مثل : البَاءُ في قولكَ : مَرَرْتُ بِسَلِيمٍ .
    3 ــ حرفٌ مختصٌّ بالأفعالِ . مثل : لَمْ ، في قولكَ لَمْ يَقُمْ خَالِدٌ .

    ــــ حُرُوفُ الجَرِّ :
    عشرونَ حرفًا و سُمِّيَتْ بحُرُوفِ الجَرِّ لأنّها تَعْمَلُ الجَرَّ كما قيل : حروفُ النّصبِ و حروفُ الجزمِ . أو لأنها تَجُرُّ معاني الأفعالِ إلى الأسماءِ أيْ تُضِيفُهَا و تُوصِلُهَا إليها فَتَرْبِطُ معنى الفعلِ بالاسمِ وهي :
    البَاءُ . مِنْ . إِلَى . عَنْ . عَلَى . فِي . الكَافُ . اللاَّمُ . وَاوُ القَسَمِ . تَاءُ القَسَمِ . مُذْ . مُنْذُ . رُبَّ . حَتَّى . خَلاَ .
    عَدَا . حَاشَا . كَيْ . مَتَى .لَعَلَّ .
    ــ مُتَعَلِّقُ الجَارِ والمَجْرُورِ :
    متعلّقُ حرفِ الجرِّ الأصلِي ُّ ما كانَ مُرْتَبِطًا به منْ فعلٍ أوْ شبهِهِ . مثل : وَقَفْتُ عَلَى المِنْبَرِ ــ أَنَا كَاتِبٌ بِالقَلَمِ .
    ــ وَقَفْتُ : فعلٌ و فاعلٌ .
    ــ عَلَى المِنْبَرِ : جارٌ و مجرورٌ متعلّقٌ بِوَقَفَ .
    ــ المثال الثاني :
    ــ أَنَا : ضميرٌ منفصلٌ مبنيٌّ على السّكونِ في محلِّ رفعٍ مبتدأٌ .
    ـــ كَاتِبٌ : خَبَرُهُ مرفوعٌ .
    ــ بِالقَلَمِ : جارٌ و مجرورٌ متعلّقٌ بِكَاتِبٍ .

    حرف الخاء

    ــــ خَالَ :
    فعلٌ ماضٍ من أفعالِ الظنِّ التي تفيدُ الرجحانَ . يَنْصِبُ مفعولينِ اثنينِ أصلُهُمَا مبتدأٌ و خَبَرٌ . مثل :
    خِلْتُ البَيْتَ مَفْتُوحًا .
    ـــ خَفْضٌ :
    الخَفْضُ عبارةٌ كوفيّةٌ . وهو تَغْيِيرٌ مَخْصُوصٌ عَلاَمَتُهُ الكَسْرَةُ وَمَا نَابَ عَنْهَا ، وسُمِّيَ بذلكَ لانْخِفَاضِ الشَّفَةِ السُّفْلَى عِنْدَ النُّطْقِ بِعَلاَمَتِهِ . مثل : مَرَرْتُ بِخَالِدٍ .
    ـــ مَرَرْتُ : فعلٌ و فَاعِلٌ .
    ـــ بِخَالِدٍ : جَارٌ و مَجْرُورٌ مُتَعَلِّقٌ بِمَرَرْتُ .
    ــ عَلاَمَاتُهُ : للخفضِ ثلاثُ علاماتٍ : الكسرةُ ، الياءُ ، الفتحةُ .
    ـــ خَلاَ :
    فعلٌ ماضٍ ضُمِّنَ معنى إِلاَّ الاستثنائيّةِ: فاسْتُثْنِيَ بها كما يُسْتَثْنَى بإِلاَّ . المستثنَى بخَلاَ يجوزُ نَصْبُهُ و جَرُّهُ .
    النَّصْبُ على أنَّ خَلاَ فعلٌ و مَا بعده مفعولٌ به .
    و الجرُّ على أنَّ خَلاَ حرفُ جرٍّ شَبِيهٌ بالزَّائِدِ . مثل : جَاءَ القَوْمُ خَلاَ عَلِيًّا أوْ عَلِيٍّ .
    ـــ خِلاَفًا :
    من قولكَ : قَالَ ذَلِكَ خِلاَفًا لِقَوْلِ فُلاَنٍ .
    ــ خِلاَفًا : حَالٌ مَنْصُوبٌ .

    ـــ خِلاَلَ :
    مِنْ آيَةِ : ( فَجَاسُوا خِلاَلَ الدِّيَارِ )
    ظَرْفُ مَكَانٍ . إذْ المعنى مَا حَوَالَيْ حُدُودِهَا و مَا بَيْنَ بُيُوتِهَا .
    ـــ خَلْفَ :
    ضد قُدَّامَ . مثل : جَلَسْتُ خَلْفَ المِنْبَرِ :
    ــ خَلْفَ : ظرفُ مكانٍ منصوبٌ على الظرفيّةِ المكانيّةِ .
    ـــ المِنْبَرِ : مضافٌ إليهِ .
    ــــ خَمْسَةَ عَشَرَ :
    مُركّبٌ عدَدِيٌّ يَجِبُ فَتْحُ جُزْئَيْهِ سَوَاءٌ أَكَانَ :
    مَرْفُوعًا . مثل : جَاءَ خَمْسَةَ عَشَرَ رَجُلاً .
    مَنْصُوبًا . مثل : رَأَيْتُ خَمْسَةَ عَشَرَ رَجُلاً .
    مَجْرُورًا . مثل : أَحْسَنْتُ إِلَى خَمْسَةَ عَشَرَ رَجُلاً .
    و يكونُ حينئذٍ مَبْنِيًّا على فَتْحِ جُزْئَيْهِ مرفوعا أو منصوبا أو مجرورا مَحَلاًّ
    ـــ رَجُلاً : تمييزٌ منصوبٌ .
    ـــ خَيْلٌ :
    الخيلُ جماعةُ الأفراسِ . اسمُ جمعٍ لأنَّهُ تَضَمَّنَ معنى الجمع .
    غيرَ أنَّهُ لا واحِدَ لَهُ مِنْ لَفْظِهِ و إنَّمَا وَاحِدُهُ مِنْ معناه . فواحده فَرَسٌ جمعه خُيُولٌ .

    خَبَرٌ :
    الخَبَرُ هو الاسْمُ المرفوعُ المُسْنَدُ إلى المبتدإِ . مثل : قَائِمٌ منْ قولكَ : خَالِدٌ قَائِمٌ .
    ــ خَالِدٌ : مبتدأٌ مرفوعٌ بالإبتداءِ .
    ــ قَائِمٌ : خَبَرُ المُبْتَدَإِ مرفوعٌ بالمُبْتَدَإِ و علامةُ رفْعهِ ضَمَّةٌ ظاهرةٌ في آخرِهِ .
    ـــ أَقْسَامُهُ : يُقْسَمُ الخَبَرُ إلى قِسْمَيْنِ : مفرد و جملة .
    ـــ مفرد : الخَبَرُ المفردُ ما كانَ غَيْرَ جُمْلَةٍ ، وإنْ كانَ مُثَنَّى أوْ جَمْعًا . مثل : العَالِمُ مُحْتَرَمٌ و العَالِمَانِ مُحْتَرَمَانِ
    و العَالِمُونَ مُحْتَرَمُونَ .
    ــ العَالِمُ : مبتدأٌ مرفوعٌ بالإبْتِدَاءِ     ـــ مُحْتَرَمٌ : خَبَرٌ مرفوعٌ بالمبتدإِ .
    و العَالِمَانِ :     مبتدأٌ مرفوعٌ بالإبْتِدَاءِ . و المُحْتَرَمَانِ : خَبَرٌ مرفوعٌ بالمبتدإِ .
    و كلٌّ منهما مرفوعٌ بالألٍفِ النّائبةِ عن الضمّةِ لأنّهُ مُثَنَّى و النُّونُ فيه عِوَضٌ عنِ التَّنْوِينِ في الاسْمِ المفردِ .
    و العَالِمُونَ : مبتدأٌ مرفوعٌ بالابتداءِ . و المُحْتَرَمُونَ : خَبَرٌ مرفوعُ بالمبتدإِ .
    و كلٌّ منهما مرفوعٌ بالوَاوِ النّائبةِ عن الضمَّةِ لأنّه جمعٌ مذكّرٌ سَالِمٌ و النُّونُ عِوَضٌ عن التَّنْوِينِ في الاسْمِ المُفْرَدِ .

    الجملة : يأتي الخبر جملة فيكون الفعل مع فاعله . مثل : خَالِدٌ سَافَرَ أَبُوهُ .
    ــ خَالِدٌ : مبتدأٌ مرفوعٌ بالإبتداءِ و علامةُ رفعِهِ ضَمُّ آخِرِهِ .
    ــ سَافَرَ : فعلٌ ماضٍ مَبْنِيٌّ على الفَتْحِ .
    ـــ أَبُو : فاعلٌ مرفوعٌ بالوَاوِ النَّائِبَةِ عنِ الضَّمَةِ لأنَّهُ اسْمٌ منَ الأسْمَاءِ الخَمْسَةِ .
    ــ الهَاءُ : مُضَافٌ إلَيْهِ . وجملةُ سَافَرَ أَبُوهُ في مَحَلِّ رَفْعٍ خبر .
    و يأتِي الخَبَرُ جملَةً فيكونُ مبتدأً مع خَبَرِهِ . مثل : خَالِدٌ أَبُوهُ قَائِمٌ .
    ــ خَالِدٌ : مبتدأٌ مرفوعٌ بالإبتداءِ و علامةُ رفعِهِ ضَمُّ آخِرِهِ .
    ـــ أَبُو : مبتدأٌ ثاني مرفوعٌ بالوَاوِ النَّائِبَةِ عنِ الضَّمَةِ لأنَّهُ اسْمٌ منَ الأسْمَاءِ الخَمْسَةِ .
    ـــ الهَاءُ : مضافٌ إليهِ .
    ـــ قَائِمٌ : خَبَرُ المبتدأ الثاني مرفوعٌ . والمبتدأُ الثَّانِي و خَبَرُهُ في محلِّ رفعٍ خَبَرُ المبتدإِ الأَوَّلِ .

    ــــ رَابِطُ الجُمْلَةِ : يُشترطُ في الجملةِ أن تكونَ مشتملةً على رابطٍ يربطُهَا بالمبتدإِ . والرّابطُ إمَّا أن يكونَ ضميرًا يرجعُ إلى المبتدإ .

    حَرْفُ الذَّالِ

    ــــ ذَا :
    اسْمٌ موصولٌ للعاقِلِ وغيرِهِ بمعنى الذِي مبنيٌّ على السّكونِ يحتاجُ إلى صِلَةٍ و عَائِدٍ و مَحَلٍّ من الإعرابِ .
    و محلُّهُ من الإعرابِ حَسَبَ موقعِهِ منَ الكلامِ . ولا تكونُ ذَا إِسْمًا مَوْصُولاً إلاَّ بشرطِ أنْ تَقَعَ بعدَ مَنْ و مَا الاستفهاميّتَين ِ . مثل : مَاذَا تُرِيدُ ؟ ـــ مَنْ ذَا يَقُومُ ؟
    ـــ ذَا :
    اسْمُ إشارَةٍ للمفرَدِ المذكَّرِ .
    ــ يُسْبَقُ اسْمُ الاشارَةِ بهَاءٍ التي هي حرْفٌ للتَّنْبِيهِ . فَيُقَالُ :هَذَا وهيَ إشارَةٌ للقرِيبِ .
    ــ تَلْحَقُ ذَا الكَافُ التي هي حرْفُ خطابٍ . فيقالُ : ذَاكَ وهي إشارةٌ للبعِيدِ .
    ـــ و قدْ تلحقُهَا هذه الكَافُ مَعَ اللاَّمِ . فَيُقَالُ : ذَلِكَ وهي إشارةٌ للبَعِيدِ أَيْضًا .
    ـــ و إِذْ ذَاكَ : تُحْذَفُ الأَلِفُ مِنْ ذَا .
    ذَا :
    بمعنى صَاحِبٍ . وهي إسْمٌ مِنَ الأَسْمَاءِ الخَمْسَةِ في حَالَةِ النَّصْبِ . مثل : رَأَيْتُ ذَا مَالٍ .
    فالأَلِفُ فِي ذَا نَائِبَةٌ عَنِ الفَتْحَةِ . و مَالٍ مُضَافٌ إِلَيْهِ .

    الحُلُمِيَّةِ . مصدرُهَا الرُؤْيَا المناميّةِ . تنصبُ مفعولينِ . كقوله تعالى : ( إِنِّي أَرَانِي أَعْصِرُ خَمْرًا ) . فالمفعولُ الأوّلُ يَاءُ المتكلّمِ والمفعولُ الثّاني جُمْلَةُ أَعْصِرُ خَمْرًا .

    ـــــ رُبَّ :
    حرفُ جرٍّ ولا تكونُ إلاَّ حرْفًا . تكونُ للتّقليلِ وللتّكثيرِ . والقرينةُ هي التي تُعَيِّنُ المُرَادَ . ولا تَجُرُّ إلاَّ النّكرات .
    قدْ تُحْذَفُ رُبَّ ويَبْقَى عملُهَا بعد الواوِ . مثل : وَلَيْلٍ كَمَوْجِ البَحْرِ . ولا تَجُرُّ إلاَّ الظّاهِرَ . وشَذَّ : ورُبَّهُ عَطَبًا .
