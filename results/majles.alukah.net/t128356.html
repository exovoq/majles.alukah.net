    الشيخ عبد الوهاب قرفالي الفقيه الزاهد

    الحمد لله والصلاة والسلام على رسول الله اما بعد قال تعالى (55) وَمَا خَلَقْتُ الْجِنَّ وَالْإِنْسَ إِلَّا لِيَعْبُدُونِ (56) مَا أُرِيدُ مِنْهُمْ مِنْ رِزْقٍ وَمَا أُرِيدُ أَنْ يُطْعِمُونِ (57) إِنَّ اللَّهَ هُوَ الرَّزَّاقُ ذُو الْقُوَّةِ الْمَتِينُ (58) وقال عزوجل (280) وَاتَّقُوا يَوْمًا تُرْجَعُونَ فِيهِ إِلَى اللَّهِ ثُمَّ تُوَفَّى كُلُّ نَفْسٍ مَا كَسَبَتْ وَهُمْ لَا يُظْلَمُونَ (281) وقال سبحانه (34) كُلُّ نَفْسٍ ذَائِقَةُ الْمَوْتِ وَنَبْلُوكُمْ بِالشَّرِّ وَالْخَيْرِ فِتْنَةً وَإِلَيْنَا تُرْجَعُونَ (35) و عَنْ أَبِي هُرَيْرَةَ، قَالَ: قَالَ رَسُولُ اللهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ: «الدُّنْيَا سِجْنُ الْمُؤْمِنِ، وَجَنَّةُ الْكَافِرِ» وعن الْحُسَيْنُ بْنُ مَنْصُورٍ قَالَ: حُدِّثْتُ عَنْ فُضَيْلِ بْنِ عِيَاضٍ فِي مَعْنَى قَوْلِهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ: «الدُّنْيَا سِجْنُ الْمُؤْمِنِ وَجَنَّةُ الْكَافِرِ» قَالَ: هِيَ سِجْنُ مَنْ تَرَكَ لَذَّاتِهَا وَشَهَوَاتِهَا فَأَمَّا الَّذِي لَا يَتْرُكُ لَذَّاتِهَا وَلَا شَهَوَاتِهَا فَأَيْ سِجْنُ هِيَ عَلَيْهِ. اذا فالدنيا دار ممر و سفر وليست دار سكن و مستقر دار عمل وابتلاء وليست دار خمول وهناء و لهذا كانت سيرة النبي صلى الله عليه وسلم و اصحابه كما علمتم عَنْ جَابِرٍ، قَالَ: " لَمَّا حَفَرَ النَّبِيُّ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ وَأَصْحَابُهُ الْخَنْدَقَ، أَصَابَهُمْ جَهْدٌ شَدِيدٌ، حَتَّى رَبَطَ النَّبِيُّ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ عَلَى بَطْنِهِ حَجَرًا مِنَ الْجُوعِ . و عَنْ مُجَاهِدٍ، أَنَّ أَبَا هُرَيْرَةَ، كَانَ يَقُولُ: وَاللهِ إِنْ كُنْتُ لَأَعْتَمِدُ بِكَبِدِي عَلَى الْأَرْضِ مِنَ الْجُوعِ، وَإِنْ كُنْتُ لَأَشُدُّ الْحَجَرَ عَلَى بَطْنِي مِنَ الْجُوعِ. و أُتِيَ عَبْدُ الرَّحْمَنِ بْنُ عَوْفٍ رَضِيَ اللَّهُ عَنْهُ يَوْمًا بِطَعَامِهِ، فَقَالَ: «قُتِلَ مُصْعَبُ بْنُ عُمَيْرٍ وَكَانَ خَيْرًا مِنِّي، فَلَمْ يُوجَدْ لَهُ مَا يُكَفَّنُ فِيهِ إِلَّا بُرْدَةٌ، وَقُتِلَ حَمْزَةُ - أَوْ رَجُلٌ آخَرُ - خَيْرٌ مِنِّي، فَلَمْ يُوجَدْ لَهُ مَا يُكَفَّنُ فِيهِ إِلَّا بُرْدَةٌ، لَقَدْ خَشِيتُ أَنْ يَكُونَ قَدْ عُجِّلَتْ لَنَا طَيِّبَاتُنَا فِي حَيَاتِنَا الدُّنْيَا ثُمَّ جَعَلَ يَبْكِي» وقال طرفة بن العبد أرى الموتَ لا يُرعي على ذي قَرابةٍ ... وإنْ كان في الدّنْيا عزيزاً بمَقعَدِ
    إذا شَاءَ يوْماً قادَهُ بِزِمَامِهِ ... ومنْ يكُ في حَبْلِ المَنيّة ِ يَنْقَدِ
    ولا خيرَ في خيرٍ ترى الشَّرَّ دونَهُ ... ولا قائلٍ يأتِيكَ بعدَ التَّلَدُّدِ
    لَعَمْرُكَ! ما الأيامُ إلاّ مُعارَةٌ ... فما اسطَعْتَ مِن مَعرُوفِها فتزَوّدِ
    وهكذا والله عرفت شيخي زاهدا في الدنيا مقبلا على الاخرة شغوفا بالعلم مداوما على العمل يامر بالمعروف و ينهي عن المنكر و هو مقعد لايستطيع ان يقوم من فراشه ويقول لي يا مصطفى اخشى ان ننام ليلة فنصبح و قد نزل بنا عذاب الله مما نرى من منكر ونحن ساكتون وكنت اذا حدثته عن حال المسلمين تاثر وادخل راسه تحت غطائه وبكى بحرقة والم ثم يخرج راسه وينصحني بالعلم ويقول يا مصطفى ارايت علم ابليس عليه لعنت الله يوسوس للعلماء في مسائل لا يعرفها الا الجهابذة والنقاد المحققين فهل نفعه علمه فاقول لا فيقول فاعتبر اذا هذا هو شيخي وقريبي الذي عرفته هو الشيخ الفقيه الزاهد عبد الوهاب بن مبارك قرفالي العيبي السيحمدي الجزائري من عائلة عرفت بجهادها وكفاحها ضد الاستعمار الفرنسي ولد الشيخ في سنة 1914م في اولاد سي احمد بقرية سلامة حيث نشا وترعرع ثم التحق بزاوية عين الديسا والتي كان يزورهم فيها الشيخ البشير الابراهيمي رحمه الله فيلقي عليهم الدروس ويجيبهم عن الاسئلة فبدا بحفظ القرءان الى ان اكمله في زاوية القطاطشة التى كان من خريجيها الشيخ سعد بن قطوش المدرس المعروف في الزيتونة وكذلك حفظ فيها الشيخ جملة من المتون وشرحها كالاجرومية والرحبية ومتن ابن عاشر وكان يحدثني عن هذه الزاوية وعن توعيتها للناس في ذلك الوقت خلافا لما كانت عليه معظم الزوايا الطرقية حتى احرقت فرنسا مخطوطات وكتب الزاوية وكان يحدثني كذلك عن شيوخها وزهدهم وعبادتهم و انه فيهم من لم يدخل السوق نصف قرن و انهم لايقبلون ان يقوم الطالب باي خدمة او عمل لهم سوى التفرغ للعلم ثم سمت همة الشيخ لاكمال طلب العلم فرحل لزاوية في منطقة بني عزيز سطيف وحدثني عن اجتهاد مشائخ هذه الزاوية في التعليم فلم يكونوا ليناموا في هذه الزاوية الا قليلا وكانوا يلزمونهم بمراجعة نصف القرءان قبل صلاة الفجر
    فقرا على شيخها الشيخ الهاشمي بعض المتون في الفقه المالكي واللغة العربية وكان مما قرا عليه مختصر خليل قال لى قراته عليه في اربع سنوات وكان يحفظه باتقان قال فزارنا الشيخ عبد الحميد بن باديس ونحن نقرا فيه فقال لنا ممازحا انتم لاتعرفون الا خليل اي متعصبون ثم قال بلى توجد مسائل قيمة في خليل وقرا على كذلك في الفية ان مالك حتى بلغ الصفة المشبهة وكذلك حفظ وقرا تحفة الحكام لابن عاصم والفرائض ثم رجع الى منطقته ليعمل معلما للقرءان وهو في هذه الفترة لايدع المطالعة والتحصيل والالتقاء بالعلماء وطلبة العلم و حضور دروسهم وقد عذب من طرف الاستعمار وسجن
    وبعد الاستقلال عزم على الحج فخرج الى تونس وليبيا ومصر والتقى في طريقه بعلماء وطلبة علم الى ان وصل الى فلسطين
    فاقام بها ستة اشهر و دخل الاردن وقرا فيها الفقه الحنفي وبقى فيها مدة وعرض عليه بعضهم الامامة فرفض واصر على اداء فريضة الحج فاداها و ذهب بعدها الى المدينة ليعود الى بلاده قال لى فحضرت درسا للشيخ ابن باز فاعجبني وقررت ان ابقى لطلب العلم فلما انتهى الدرس ذهب الى الشيخ ابن باز واخبره عن رغبته فقام الشيخ بتسجيله في دار الحديث فقرا فيها عدة سنوات على عدة من اهل العلم منهم الشيخ على سنان وبعض المشايخ الافغان والشناقطة اما في المسجد النبوي فكان يحضر دروس الشيخ فلاته والشيخ ابو بكر الجزائري وكان صديقه والشيخ الامين الشنقيطي وقد اهداه نسخة من اضواء البيان مازال عندي منها تفسير سورة الحج الجزء الخامس مطبعة المدني المؤسسة السعودية لصاحبها علي صبحي المدني وقد كتب عليها الشيخ محمد غفوري تاريخ 1391ه ودرس على الشيخ محمد المختار الشنقيطي الاب والشيخ عطية سالم وغيرهم كثير جدا وهذا من حرص الشيخ على العلم والاستفادة وقال لي كنت ازور الشيخ ابن باز في بيته وقرا عليه في الصحيحين وكذلك كان من احب مشايخه الى قلبه الشيخ عبد العزيز بن صالح امام الحرم فقد كان يحدثني عنه كثيرا ثم انتقل الى مكة وعمل حارسا على المقام وواصل طلب العلم والتحصيل وكان من شيوخه في مكة الشيخ العربي التباني وكان من اعز اصدقائه تلميذ الشيخ العربي التباني محمد غفوري وبينهما رسائل كثيرة بعد رجوع الشيخ الى الجزائر مازالت عندي بعضها ثم بسبب بعض الاحداث عاد الى الجزائر وعمل بتعليم القرءان والامامة ليصبح بعدها دفين بيته فالناس عندنا لايهتمون بالعلم وكل اسئلتهم عن الطلاق والذكاة كما حدثني الشيخ فانقطع الشيخ للعبادة والمطالعة فكان يختم القرءان كل يوم واذا دخلت عليه لا تجد عنده الا مصحفه او كتاب وو الله ان همته في طلب العلم بعد التسعين كهمة شاب لارغبة له الا في العلم في العشرين و كانت نصيحته لكل من طلب النصيحة خلقك الله لعبادته فاعبده رحمه الله واسكنه فسيح جناته فقد كان ابا ومربيا وشيخا علمنا معنى الحياة ومعنى الزهد علمنا كيف يكون العلم نافعا وكيف يكون العمل صالحا ومن عجائب الشيخ انه لم يتزوج حتى ماتت امه رغبة في برها والقيام بحقها وقد كان رحمه الله رغم فقره المدقع كريما جوادا معطاء وقد اوقف علي كتبه رغبة في الاجر وقد حافظ الشيخ على صيام ستة ايام من شوال وهو ابن قرن وكان يوصيني بالدعاء له بعد موته فلا تنسوه بالدعاء يا اخوتي .
