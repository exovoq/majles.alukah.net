    ﻭَﺑَﻨَﻰ ﺑِﻬَﺎ ﺩَﺍﺭًﺍ ، ﺩَﺧَﻞَ ﺍﻟﺸَّﺎﻡَ ، ﻭَﺍﻟْﺤِﺠَﺎﺯ ، ﻭَﻣِﺼْﺮَ ، ﻭَﺍﻟْﻌِﺮَﺍﻕ ﻭَﺧُﺮَﺍﺳَﺎﻥ ، ﻭَﻛَﺘَﺐَ ﻋَﻦْ ﻋَﺎﻣَّﺔِ ﻣَﺸَﺎﻳِﺦِ ﺍﻟْﻮَﻗْﺖِ ، ﻭَﺭَﻭَﻯ ﻋَﻨْﻬُﻢْ ، ﻭَﻛَﺎﻥَ ﺛِﻘَﺔً ﺻَﺪُﻭﻗًﺎ ، ﺣَﺎﻓِﻈًﺎ ، ﻋَﺎﻟِﻤًﺎ ﺑِﺎﻟﺼَّﺤِﻴﺢ ﻭَﺍﻟﺴَّﻘِﻴﻢ ، ﺣَﺴَﻦَ ﺍﻟْﻤَﻌْﺮِﻓَ ِ ﺑِﺎﻟﺮِّﺟَﺎﻝ ﻭَﺍﻟْﻤُﺘُﻮﻥ ، ﻛَﺜِﻴﺮَ ﺍﻟﺘَّﺼَﺎﻧِﻴ ِ ، ﺟَﻴِّﺪَ ﺍﻟْﺨَﻂِّ ، ﻟَﺎﺯِﻣًﺎ ﻟِﻠْﺄَﺛَﺮِ ، ﺑَﻌِﻴﺪًﺍ ﻣِﻦَ ﺍﻟْﻔُﻀُﻮﻝِ ﻭَﺍﻟﺘَّﻌَﺼُّ ﺐِ ، ﺧَﻔِﻴﻒَ ﺍﻟﺮُّﻭﺡِ ، ﻗَﻮِﻱَّ ﺍﻟﺴَّﻴْﺮِ ﻓِﻲ ﺍﻟﺴَّﻔَﺮِ ، ﻛَﺜِﻴﺮَ ﺍﻟْﺤَﺞِّ ﻭَﺍﻟْﻌُﻤْﺮَ ِ ، ﻣَﺎﺕَ ﺑِﺒَﻐْﺪَﺍﺩَ ﻣُﻨْﺼَﺮِﻓًﺎ ﻣِﻦَ ﺍﻟْﺤَﺞِّ . ﻗَﺎﻝَ ﺍﺑْﻦُ ﺍﻟﻨَّﺠَّﺎﺭِ : ﻗَﺮَﺃْﺕُ ﺑِﺨَﻂِّ ﺷُﺠَﺎﻉٍ ﺍﻟﺬُّﻫْﻠِﻲِّ : ﺃَﺧْﺒَﺮَﻧِﻲ ﺃَﺣْﻤَﺪُ ﺑْﻦُ ﻣُﺤَﻤَّﺪِ ﺑْﻦِ ﺃَﺣْﻤَﺪَ ﺍﻟْﺒَﺰَّﺍﺯُ ، ﺣَﺪَّﺛَﻨَﺎ ﻣُﺤَﻤَّﺪُ ﺑْﻦُ ﻃَﺎﻫِﺮِ ﺑْﻦِ ﻋَﻠِﻲِّ ﺍﻟْﻤَﻘْﺪِﺳِ ُّ ، ﺃَﺧْﺒَﺮَﻧَﺎ ‏ ‏ ﻋُﺜْﻤَﺎﻥُ ﺑْﻦُ ﻣُﺤَﻤَّﺪٍ ﺍﻟْﻤَﺤْﻤِﻲُّ ﺑِﻨَﻴْﺴَﺎﺑُ ﺭَ ، ﻓَﺬَﻛَﺮَ ﺣَﺪِﻳﺜًﺎ . ﺃَﻧْﺒَﺆُﻭﻧَ ﻋَﻦْ ﺷِﻬَﺎﺏٍ ﺍﻟْﺤَﺎﺗِﻤِﻲ ِ ، ﺃَﺧْﺒَﺮَﻧَﺎ ﺃَﺑُﻮ ﺳَﻌْﺪٍ ﺍﻟﺴَّﻤْﻌَﺎﻧ ﻲُّ ، ﺳَﻤِﻌْﺖُ ﻣَﻦْ ﺃَﺛِﻖُ ﺑِﻪِ ﻳَﻘُﻮﻝُ : ﻗَﺎﻝَ ﻋَﺒْﺪُ ﺍﻟﻠَّﻪِ ﺑْﻦُ ﻣُﺤَﻤَّﺪٍ ﺍﻟْﺄَﻧْﺼَﺎﺭ ﻱُّ ﺍﻟْﻬَﺮَﻭِﻱُّ : ﻳَﻨْﺒَﻐِﻲ ﻟِﺼَﺎﺣِﺐِ
    ﺍﻟْﺤَﺪِﻳﺚِ ﺃَﻥْ ﻳَﻜُﻮﻥَ ﺳَﺮِﻳﻊَ ﺍﻟْﻘِﺮَﺍﺀَﺓ ، ﺳَﺮِﻳﻊَ ﺍﻟﻨَّﺴْﺦِ ، ﺳَﺮِﻳﻊَ ﺍﻟْﻤَﺸْﻲِ ، ﻭَﻗَﺪْ ﺟَﻤَﻊَ ﺍﻟﻠَّﻪُ ﻫَﺬِﻩِ ﺍﻟْﺨِﺼَﺎﻝَ ﻓِﻲ ﻫَﺬَﺍ ﺍﻟﺸَّﺎﺏِّ ، ﻭَﺃَﺷَﺎﺭَ ﺇِﻟَﻰ ﺍﺑْﻦِ ﻃَﺎﻫِﺮٍ ، ﻭَﻛَﺎﻥَ ﺑَﻴْﻦَ ﻳَﺪَﻳْﻪِ . ﻭَﺑِﻪِ ﻗَﺎﻝَ ﺍﻟﺴَّﻤْﻌَﺎﻧ ﻲُّ : ﻭَﺳَﻤِﻌْﺖُ ﺃَﺑَﺎ ﺟَﻌْﻔَﺮٍ ﺍﻟﺴَّﺎﻭِﻱَّ ﻳَﻘُﻮﻝُ : ﻛُﻨْﺖُ ﺑِﺎﻟْﻤَﺪِﻳﻨ ﺔِ ﻣَﻊَ ﺍﺑْﻦِ ﻃَﺎﻫِﺮٍ ، ﻓَﻘَﺎﻝَ : ﻟَﺎ ﺃَﻋْﺮِﻑُ ﺃَﺣَﺪًﺍ ﺃَﻋْﻠَﻢَ ﺑِﻨَﺴَﺐِ ﻫَﺬَﺍ ﺍﻟﺴَّﻴِّﺪِ ﺻَﻠَّﻰ ﺍﻟﻠَّﻪُ ﻋَﻠَﻴْﻪِ ﻭَﺳَﻠَّﻢَ ﻣِﻨِّﻰ ، ﻭَﺁﺛَﺎﺭِﻩِ ﻭَﺃَﺣْﻮَﺍﻟِ ِ . ﻭَﺳَﻤِﻌْﺖُ ﺑَﻌْﻀَﻬُﻢْ ﻳَﻘُﻮﻝُ : ﻛَﺎﻥَ ﺍﺑْﻦُ ﻃَﺎﻫِﺮٍ ﻳَﻤْﺸِﻲ ﻓِﻲ ﻟَﻴْﻠَﺔٍ ﻭَﺍﺣِﺪَﺓٍ ﻗَﺮِﻳﺒًﺎ ﻣِﻦْ ﺳَﺒْﻌَﺔَ ﻋَﺸْﺮَ ﻓَﺮْﺳَﺨًﺎ .

    ﻣِﻨْﻪُ ﺛَﻠَﺎﺛَﻤِﺎﺋ ﺔٍ ، ﻭَﺟَﺎﺀَ ﺑِﻬَﺎ . ﻗَﺎﻝَ : ﻭَﻛُﻨْﺖُ ﺑِﺒَﻐْﺪَﺍﺩَ ﻓِﻲ ﺳَﻨَﺔِ ﺳَﺒْﻊٍ ﻭَﺳِﺘِّﻴﻦَ ﻭَﺃَﺭْﺑَﻌِﻤِ ﺎﺋَﺔٍ ، ﻭَﺗُﻮُﻓِّﻲَ ﺍﻟْﻘَﺎﺋِﻢُ ﺑِﺄَﻣْﺮِ
