    ما صحة الخبر أنه كان صلى الله عليه وسلم ربما مشى حافيا؟ كما أشار إلى ذلك الحافظ العراقي.

    ذكر بعض أهل السير: أنه كان صلى الله عليه وسلم ربما مشى حافيا لعيادة المريض. كما أشار إلى ذلك الحافظ زين الدين العراقي رحمه الله تعالى في «ألفيته» بقوله:
    يمشي بلا نعل ولا خف إلى ... عيادة المريض حوله الملا

    فما صحة هذا الأثر؟

    رد: ما صحة الخبر أنه كان صلى الله عليه وسلم ربما مشى حافيا؟ كما أشار إلى ذلك الحافظ العراقي.

    جاء في سبل الهدى والرشاد (7/ 160)
    في مشيه صلّى الله عليه وسلم حافيا وناعلا.
    روى البزار برجال ثقات عن ابن عمر رضي الله تعالى عنهما قال: كان رسول الله صلى الله عليه وسلّم يمشي حافيا وناعلا.


    وفي أخلاق النبي لأبي الشيخ الأصبهاني (2/ 354):
    حَدَّثَنَا سَلْمُ بْنُ عِصَامٍ [الذهبي: محدث أصبهان، له غرائب، قال ابن مندة: أحد الثقات، ت 308 هـ]
    نَا الْحَسَنُ بْنُ يَحْيَى بْنِ هِشَامٍ الرُّزِّيُّ [صدوق، صاحب حديث، قال الذهبي: كان ثقة حافظا، ت (241 - 250 هـ)]
    نَا أَبُو سَلَمَةَ مُوسَى [ابن إسماعيل المِنْقَري، ثقة ثبت]
    نَا هَارُونُ بْنُ مُوسَى [الأزدي النحوي، ثقة مقرئ، إلا انه رمي بالقدر، ت (164 - 170 هـ)]
    عَنْ حُسَيْنٍ [بن ذكوان] الْمُعَلِّمِ [ثقة، ربما وهم، ت 145 هـ]
    عَنْ عَبْدِ اللَّهِ بْنِ بُرَيْدَةَ [ثقة، ت 105 هـ]
    عَنْ عِمْرَانَ بْنِ حُصَيْنٍ، أَنَّ النَّبِيَّ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ كَانَ يَمْشِي حَافِيًا وَنَاعِلًا، وَيَشْرَبُ قَائِمًا وَقَاعِدًا، وَيَنْفَتِلُ عَنْ يَمِينِهِ وَعَنْ شِمَالِهِ، وَيَصُومُ فِي السَّفَرِ وَيُفْطِرُ



    ومن طريقه سلم بن عصام رواه البغوي في الأنوار في شمائل النبي المختار (ص: 550، رقم: 826): حَدَّثَنَا الْمُطَهَّرُ بْنُ عَلِيٍّ أَنَا مُحَمَّدُ بْنُ إِبْرَاهِيمَ أَنَا عَبْدُ اللَّهِ بْنُ مُحَمَّدِ بْنِ جَعْفَرٍ نَا سَلمُ بْنُ عِصَامٍ ...

    رد: ما صحة الخبر أنه كان صلى الله عليه وسلم ربما مشى حافيا؟ كما أشار إلى ذلك الحافظ العراقي.

    والحديث الذي أراده البزار:
    رواه الإمام أحمد في مسنده ط الرسالة (11/ 200، رقم: 6627): حَدَّثَنَا مُحَمَّدُ بْنُ جَعْفَرٍ، حَدَّثَنَا سَعِيدُ بْنُ أَبِي عَرُوبَةَ،
    عَنْ حُسَيْنٍ الْمُعَلِّمِ، عَنْ عَمْرِو بْنِ شُعَيْبٍ، عَنْ أَبِيهِ، عَنْ جَدِّهِ، قَالَ: " رَأَيْتُ رَسُولَ اللهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ يُصَلِّي ينفَتِلُ عَنْ يَمِينِهِ وَعَنْ شِمَالِهِ، وَرَأَيْتُهُ يُصَلِّي حَافِيًا وَمُنْتَعِلًا، وَرَأَيْتُهُ يَشْرَبُ قَائِمًا وَقَاعِدًا ".
    ورواه البغدادي في تاريخ بغداد ت بشار (7/ 630): (أَخْبَرَنَا عَلِيُّ بْنُ طَلْحَةَ بْنِ مُحَمَّدٍ الْمُقْرِئُ، قَالَ: أَخْبَرَنَا عَبْدُ الْعَزِيزِ بْنُ جَعْفَرٍ الْخِرَقِيُّ، قَالَ: حَدَّثَنَا عَلِيُّ بْنُ إِسْحَاقَ بْنِ زَاطِيَا، قَالَ: حَدَّثَنَا بَقِيَّةُ بْنُ مِهْرَانَ الزَّنْدَرُودِي ُّ، قَالَ: حَدَّثَنَا عُثْمَانُ بْنُ عَبْدِ الرَّحْمَنِ، عَنْ عَمْرِو بْنِ شُعَيْبٍ، عَنْ أَبِيهِ، عَنْ جَدِّهِ، قَالَ: رَأَيْتُ النَّبِيَّ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ " يَشْرَبُ قَائِمًا وَقَاعِدًا، وَيَمْشِي حَافِيًا وَمُنْتَعِلا، وَيَنْصَرِفُ عَنْ يَمِينِهِ وَعَنْ شِمَالِهِ ").


    وفي الآحاد والمثاني لابن أبي عاصم (5/ 350، رقم: 2929):
    حَدَّثَنَا إِبْرَاهِيمُ بْنُ الْحَجَّاجِ
    نا حَمَّادُ بْنُ سَلَمَةَ
    عَنْ سَعِيدٍ الْجُرَيْرِيِّ
    عَنْ عَبْدِ اللَّهِ بْنِ بُرَيْدَةَ، أَنَّ رَجُلًا مِنْ أَصْحَابِ النَّبِيِّ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ كَانَ يَمْشِي حَافِيًا وَلَا يَدَّهِنُ فَقِيلَ لَهُ فِي ذَلِكَ: أَنْتَ أَمِيرُ النَّاسِ تَمْشِي حَافِيًا وَلَا تَدَّهِنُ. فَقَالَ: كَانَ رَسُولُ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ يَنْهَانَا عَنْ كَثِيرٍ مِنِ الْإِرْفَاءِ وَهُوَ الِادِّهَانُ كُلَّ يَوْمٍ وَكَانَ يَأْمُرُنَا أَنْ نَحْتَفِيَ أَحْيَانًا "

    وتابع إبراهيمَ بن الحجاج: سليمانُ بن حرب فرواه البيهقي في الآداب (ص: 230، رقم: 563): (أَنْبَأَنَا أَبُو الْحَسَنِ عَلِيُّ بْنُ مُحَمَّدِ بْنِ عَلِيٍّ الْمُقْرِئُ، أَنْبَأَنَا الْحَسَنُ بْنُ مُحَمَّدِ بْنِ إِسْحَاقَ الْإِسْفَرَايِي نِيُّ، حَدَّثَنَا يُوسُفُ بْنُ يَعْقُوبَ، حَدَّثَنَا سُلَيْمَانُ بْنُ حَرْبٍ، حَدَّثَنَا حَمَّادُ بْنُ سَلَمَةَ)

    وتابع حمادًا: ابنُ علية كما في شعب الإيمان للبيهقي (8/ 431، رقم: 6050) قال: (أَخْبَرَنَا أَبُو الْحَسَنِ عَلِيُّ بْنُ مُحَمَّدٍ الْمُقْرِئُ، أَنَا الْحَسَنُ بْنُ مُحَمَّدِ بْنِ إِسْحَاقَ، ثَنَا يُوسُفُ بْنُ يَعْقُوبَ، ثَنَا سُلَيْمَانُ بْنُ حَرْبٍ، ثَنَا حَمَّادُ بْنُ سَلَمَةَ، ...) فذكر الحديث، ثم قال: (وَرَوَاهُ فِي الِاحْتِفَاءِ: زُهَيْرُ بْنُ حَرْبٍ، عَنِ ابْنِ عُلَيَّةَ، عَنِ الْجُرَيْرِيِّ، عَنْ عَبْدِ اللهِ بْنِ بُرَيْدَةَ أَنَّ رَجُلًا سَمِعَ مِنْ رَسُولِ اللهِ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ حَدِيثًا وَقَدْ سَمِعَهُ مَعَهُ رَجُلٌ يُقَالُ لَهُ عُبَيْدٌ فَأَتَاهُ فَقَالَ: إِنَّ النَّبِيَّ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ كَانَ يَأْمُرُنَا بِالِاحْتِفَاءِ ").
    ويزيد بن هارون كما في مسند أحمد ط الرسالة (39/ 388، رقم: 23969):حَدَّثَنَا يَزِيدُ بْنُ هَارُونَ، قَالَ: أَخْبَرَنِي الْجُرَيْرِيُّ، عَنْ عَبْدِ اللهِ بْنِ بُرَيْدَةَ، أَنَّ رَجُلًا مِنْ أَصْحَابِ النَّبِيِّ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ رَحَلَ إِلَى فَضَالَةَ بْنِ عُبَيْدٍ وَهُوَ بِمِصْرَ فَقَدِمَ عَلَيْهِ وَهُوَ يَمُدُّ نَاقَةً لَهُ فَقَالَ: إِنِّي لَمْ آتِكَ زَائِرًا إِنَّمَا أَتَيْتُكَ لِحَدِيثٍ بَلَغَنِي عَنْ رَسُولِ اللهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ رَجَوْتُ أَنْ يَكُونَ عِنْدَكَ مِنْهُ عِلْمٌ فَرَآهُ شَعِثًا فَقَالَ: مَا لِي أَرَاكَ شَعِثًا وَأَنْتَ أَمِيرُ الْبَلَدِ؟ قَالَ: إِنَّ رَسُولَ اللهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ كَانَ " يَنْهَانَا عَنْ كَثِيرٍ مِنَ الْإِرْفَاهِ " وَرَآهُ حَافِيًا، فَقَالَ: مَا لِي أَرَاكَ حَافِيًا؟ قَالَ: " إِنَّ رَسُولَ اللهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ أَمَرَنَا أَنْ نَحْتَفِيَ أَحْيَانًا "
    وكذا في سنن أبي داود - تحقيق عوامة (4/ 444)، رقم: 4160): حَدَّثَنَا الْحَسَنُ بْنُ عَلِيٍّ، حَدَّثَنَا يَزِيدُ بْنُ هَارُونَ، أَخْبَرَنَا الْجُرَيْرِيُّ، ... الحديث




    وتابع سعيدًا: كهمس كما في معرفة الصحابة لأبي نعيم (6/ 3144، رقم: 7238) قال: (حَدَّثَنَا أَبُو عَمْرِو بْنُ حَمْدَانَ، ثنا الْحَسَنُ بْنُ سُفْيَانَ، ثنا إِبْرَاهِيمُ بْنُ الْحَجَّاجِ، ...) فذكر السند والحديث السابق، ثم قال: (رَوَاهُ ابْنُ الْمُبَارَكِ، عَنْ كَهْمَسٍ، عَنْ عَبْدِ اللهِ بْنِ بُرَيْدَةَ، نَحْوَهُ).

    هذه التعليقات القادمة نتيجة بحثي ، وأنا أعرضها هنا لمحض النفع.


    الأول: حديث عمران بن حصين حديث شاذ ومقلوب ومحرف.
    -(الألف) هو شاذ لأن جميع الرواة يرون هذا الحديث عن عبد الله بن عمرو رضي الله عنه مثلما هو عند أبي داود في سننه (653) والترمذي في سننه (1883) وفي الشمائل (207) وابن ماجه في سننه (931) و(1038) والبيهقي في سننه الكبير (4324) والدارقطني في سننه (2300) وأحمد في مسنده (6737) و(6771) و(6790) و(6901) و(7047) و(7142) وعبد الرزاق في مصنفه (1512) و(4490) وابن أبي شيبة في مصنفه (7943) والطحاوي في شرح معاني الآثار (2920) والطبراني في الأوسط (7892).
    كلهم يرون عن عبد الله بن عمرو ، ومنهم من يرويه كاملا ومنهم من يقطعه.
    وليس عند هارون بن موسى حديث عن حسين المعلم إلا هذا الحديث.


    -(الباء) هو حديث مقلوب لأن هارون بن موسى رواه عن حسين المعلم عن عبد الله بن بريدة عن عمران بن حصين ، فغير إسناده وذكر عبد الله بن بريدة عن عمران بن حصين ، وأصله حسين المعلم حدثنا عمرو بن شعيب عن أبيه عن جده. وقد رواه هكذا صحيحا عن حسين المعلم كل من علي بن المبارك ومحمد بن جعفر ويزيد بن زريع وهارون بن مسلم ووعبد الواحد بن واصل الحداد ويحيى بن سعيد القطان وسعيد بن أبي عروبة وعباد بن عوام وخالد بن عبد الله بن عبد الرحمن الطحان.
    علي بن المبارك هو في سنن أبي داود (653) والبيهقي في سننه الكبير (4324)
    ومحمد بن جعفر في سنن الترمذي (1883) وفي الشمائل (207)
    ويزيد بن زريع في سنن ابن ماجه (931) و(1038)
    ووهارون بن مسلم في سنن الدارقطني (2300)
    وعبد الواحد بن واصل الحداد في مسند أحمد (7047)
    ويحيى بن سعيد القطان في مسند أحمد (6790)
    سعيد بن أبي عروبة في مسند أحمد (7142)
    وعباد بن عوام في مصنف ابن أبي شيبة (7943)
    وخالد بن عبد الله بن عبد الرحمن الطحان في شرح معاني الآثار للطحاوي (2920).

    هؤلاء رواة عن حسين المعلم عن عمرو بن شعيب ، أما الذين رووه عن عمرو بن شعيب غير حسين المعلم فهم قتادة بن دعامة ومطر بن طهمان وحجاج بن أرطاة ومقاتل بن سليمان.


    -(الجيم) حديث عمران محرف ، لأن فيه "أن النبي صلى الله عليه وسلم كان يمشي حافيا وناعلا" ولفظة "يمشي" محرفة. وصحيحها "أن النبي صلى الله عليه وسلم كان يصلي حافيا وناعلا ". وانظروا المصادر المتقدمة لذلك. وقد قال الشيخ شعيب أنه تحريف (مسند أحمد 11 / 202 تــــــ شعيب الأرناؤوط).
    وقد وقع مثل هذا التحريف في كلمة يصلي إلى كلمة يمشي في حديث عائشة رضي الله عنها في مسند أحمد (41 / 115 تــــــ شعيب الأرناؤوط) ومسند الشاميين (252).
    ومثله وقع في حديث عبد الله بن عمرو رضي الله عنه في تاريخ بغداد (7 / 130).
