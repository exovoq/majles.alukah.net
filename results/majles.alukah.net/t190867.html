    من رأى صاحب بلاء، فقال: الحمد لله الذي عافاني مما ابتلاك به، وفضلني على كثير ممن خلق تفضيلا

    روى من حديث عبد الله بن عمر ، وأبي هريرة ، وعبد الله بن عباس، وأنس بن مالك ، وحذيفة بن أوس.

    حديث عبد الله بن عمر.
    رُوى من طريق عمرو بن دينار ، واختُلف عليه:

    = فرواه حماد بن زيد ، وعبد الوارث بن سعيد ، وابن جريج ، وزياد بن الربيع ، وسفيان بن عيينة ، ومحمد بن موسى السعدي ، [وَسَعِيدُ بْنُ زَيْدٍ , وَعَبَّادُ بْنُ دَاوُدَ , وَأَشْعَثُ السَّمَّانُ] ، عنه ، عَنْ سَالِمِ بْنِ عَبْدِ اللَّهِ بْنِ عُمَرَ، عَنْ ابْنِ عُمَرَ، عَنْ عُمَرَ، أَنَّ رَسُولَ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ قَالَ: " مَنْ رَأَى صَاحِبَ بَلَاءٍ، فَقَالَ: الحَمْدُ لِلَّهِ الَّذِي عَافَانِي مِمَّا ابْتَلَاكَ بِهِ، وَفَضَّلَنِي عَلَى كَثِيرٍ مِمَّنْ خَلَقَ تَفْضِيلًا، إِلَّا عُوفِيَ مِنْ ذَلِكَ البَلَاءِ كَائِنًا مَا كَانَ مَا عَاشَ"

    أخرجه الترمذي (3431) ، والطيالسي (13) ، والبزّار (124) ، وابن السني في "عمل اليوم والليلة" (308) ، والبيهقي في "شعب الإيمان" (10633) ، وفي "الدعوات الكبير" (567) ، وأبو نعيم في "حلية الأولياء" (265/6) ، والطبراني في "الدعاء" (797) ، وتمام في "فوائده" (1410) ، ومن طريقه ابن عساكر في "تاريخ دمشق" (236/5) ، والخرائطي في "فضيلة الشكر" (2) ، والرامهرمزي في "المحدث" (ص330) ، والعقيلي في "الضعفاء" (270/3) ، وأبو عبد الله الفراء في "فوائده" (44) ، والدارقطني في "الخامس من الفوائد المنتقاة" (2) ، والبغوي في "شرح السنة" (1337) ، وأبو طاهر السلفي في "الطيوريات" (456/2) ، والحنائي في "فوائده" (558/1) ، وهشام بن عمار في "حديثه" (83) ، والضياء المقدسي في "المنتقى من مسموعات مرو" (649) ، وابن عدي في "الكامل" (236/6) (436/7).

    ومتابعة كل من [سَعِيدُ بْنُ زَيْدٍ , وَعَبَّادُ بْنُ دَاوُدَ , وَأَشْعَثُ السَّمَّانُ] ، لم تثبت لأنها جاءت من طريق العباس بن الفضل الأزرق ، قال ابن معين: ((كذاب خبيث)).
    انظر "مسند الحارث" (1056) ، "ميزان الاعتدال" (386/2).

    قال الترمذي: ((هَذَا حَدِيثٌ غَرِيبٌ)).
    وقال العقيلي: ((وَفِيهِ رِوَايَةٌ مِنْ غَيْرِ هَذَا الْوَجْهِ فِيهَا لِينٌ أَيْضًا، وَهَى أَصْلَحُ مِنْ هَذِهِ الرِّوَايَةِ)).

    = ورواه حماد بن سلمة ، واختُلف عليه:
    - فرواه محمد بن الفضل السدوسي ، عنه ، عن عمرو بن دينار ، عن سالم ، عن ابن عمر ، عن عمر به.
    أخرجه عبد بن حميد في "مسنده" (38)
    - ورواه قبيصة بن عقبة ، عنه ، عن عمرو بن دينار ، عن جابر بن عبد الله به موقوفاً.
    أخرجه هناد بن السري في "الزهد" (448).

    = ورواه خارجة بن مصعب ، عنه ، عن سالم ، عن ابن عمر به مرفوعاً.
    أخرجه ابن ماجه (3892).
    وخارجة ، متروك الحديث.

    = ورواه الحكم بن سنان ، عنه ، عن نافع ، عن ابن عمر به مرفوعاً.
    أخرجه ابن عدي في "الكامل" (487/2).
    وقال ابن عدي: ((وهذا الْحَدِيث إِنَّمَا يرويه عَمْرو بْن دينار، وَهو أَبُو يَحْيى قهرمان آل الزُّبَيْر عَن سالم بْنِ عَبد اللَّهِ، عَنْ أَبِيهِ، عَن جَدِّهِ، ومَنْ قَالَ عَن عَمْرو بْنُ دِينَارٍ عَنْ نَافِعٍ، عنِ ابْن عُمَر فقد أخطأ به قاله الحكم بْن سنان وبهلول بن عُبَيد وغيرهما)).
    وقال الدارقطني في "العلل" (53/2) (399/12): ((وَوَهِمَ فِيهِ [الحكم بن سنان] عَلَيْهِ [أي عمرو بن دينار])).
    وقال أيضاً كما في "أطراف الغرائب والأفراد" (492/3): ((تفرد بِهِ حكم بن سِنَان عَن عَمْرو بن دِينَار الْبَصْرِيّ عَن نَافِع وَغَيره يرويهِ عَن عَمْرو عَن سَالم)).

    = ورواه إسماعيل بن علية ، واختُلف عليه:
    - فرواه موسى بن سهل بن كثير الوشاء ، عنه ، عن عمرو بن دينار ، عن سالم ، عن ابن عمر ، به مرفوعاً.
    أخرجه ابن الأعرابي في "معجمه" (2364) ، ومن طريقه الحنائي في "فوائده" (93).
    وقال الحنائي: ((هَذَا حَدِيثٌ غَرِيبٌ...))
    وموسى بن سهل ، قال الدارقطني: ((ضعيف، لا يحتج به)) ، وقال البرقاني: ((ضعيف جدّا)).
    - ورواه ابن أبي شيبة ، عنه ، عن عمرو بن دينار ، عن سالم ، عن ابن عمر ، به موقوفاً.
    أخرجه ابن أبي شيبة في "المصنف" (29736).

    وهذا الخلاف في الحديث حول عمرو بن دينار ، إنما هو من ضعفه وسوء حفظه ، قال أحمد: ((ضعيف ، منكر الحديث)) ، وقال ابن معين: ((ليس بشئ)) ، وقال عمرو بن على الفلاس: ((ضعيف الحديث ; روى عن سالم عن ابن عمر عن النبى صلى الله عليه وسلم أحاديث منكرة)) ، وقال أبو زرعة: ((واهى الحديث)) ، وقال أبو حاتم: ((ضعيف الحديث ; روى عن سالم بن عبد الله عن أبيه غير حديث منكر ، و عامة حديثه منكر)) ، وقال البخارى: ((فيه نظر)) ، وقال النسائي: ((ليس بثقة . روى عن سالم عن ابن عمر أحاديث منكرة)) ، وقال ابن حبان: ((لا يحل كتب حديثه إلا على جهة التعجب ، كان ينفرد بالموضوعات عن الأثبات)) ، وقال أبو داود فى حديثي عمرو بن دينار قهرمان الزبير ـ يعنى : عن سالم عن أبيه عن جده ـ ((ليسا بشىء)).

    وقال علي بن المديني: (( وقد رَوى [عمرو بن دينار] حديثًا عن سالم، عن أبيه، عن عمرَ، عن النبيِّ صلى الله عليه وسلم أنه قال: «مَن رأى مُبتلًى ...»، وهذا ممَّا أَنكروه)).
    انظر: "مسند الفاروق" لابن كثير (41/3).
    ورجّح الدارقطني - في هذا الخلاف حول عمرو بن دينار - في "العلل" (2827) رواية الجماعة: حماد بن زيد ، ومن وافقه.

    ورُوي من طريق أيوب السختياني ، واختُلف عليه:
    = فرواه المغيرة بن مسلم ، عنه ، عن نافع ، عن ابن عمر به مرفوعاً.
    أخرجه الطبراني في "الأوسط" (5324) ، والبزار (5838) عن زكريا بن يحيي عن المغيرة بن مسلم به.
    قال الطبراني: ((لَمْ يَرْوِ هَذَا الْحَدِيثَ عَنْ أَيُّوبَ إِلَّا الْمُغِيرَةُ بْنُ مُسْلِمٍ، وَلَا عَنِ الْمُغِيرَةِ إِلَّا شَبَابَةُ، تَفَرَّدَ بِهِ: زَكَرِيَّا بْنُ يَحْيَى)).
    وقال البزار: ((وَهَذَا الْحَدِيثُ لا نعلَمُ أحَدًا رَوَاهُ عَن أَيُّوبَ، عَن نافعٍ، عَن ابْنِ عُمَر إلاَّ الْمُغِيرَةُ بْنُ مُسْلِمٍ وَالْمُغِيرَةُ لَيْسَ بِهِ بَأْسٌ بَصْرِيٌّ مَشْهُورٌ وَالْحَدِيثُ غريب)).
    وقال ابن القطان في "إحكام النظر" (ص512): ((أبو المغيرة بن مسلم مشهور، ليس به بأس، فهو إسناد حسن)).
    وجزم الدارقطني في "العلل" (344/12) أن المغيرة بن مسلم ، وهم فيه.

    = ورواه سفيان الثوري ، عنه ، عن عمرو بن دينار ، عن سالم ، عن ابن عمر به مرفوعاً.
    أخرجه أبو نعيم في "أخبار أصبهان" (259/1) من طريق مهران بن أبي عمر ، عن الثوري به.
    ومهران قال ابن حجر: ((صدوق له أوهام سىء الحفظ)) ، وقال ابن معين: ((كان عنده غلط كثير فى حديث سفيان)) ، وقال العقيلي: ((روى عن الثورى أحاديث لا يتابع عليها)) ، وقال البخاري: ((فى حديثه اضطراب)).
    قلت: فمثله لا يقبل تفرده عن الثوري - دون سائر أصحابه - بحال من الأحوال.
    وقال الدارقطني في "العلل" (2771): ((رواه الثوري، عن أيوب، عن رجل، عَنْ سَالِمٍ، عَنِ ابْنِ عُمَرَ، عَنِ النَّبِيِّ صَلَّى اللَّهُ عليه وسلم وهذا الرجل هُوَ عَمْرُو بْنُ دِينَارٍ، قَهْرَمَانُ آلُ الزُّبَيْرِ، وهو أصح من حديث المغيرة)).

    = ورواه معمر بن راشد ، واختُلف عليه:
    - فرواه عبد الرزاق ، عنه ، عن أيوب ، عن سالم قال: كَانَ يُقَالُ....فذكره.
    أخرجه عبد الرزاق كما في "المصنف" (20713) ، و"جامع معمر" (19655) ، والبيهقي في "شعب الإيمان" (10632).
    وقال الحنائي في "فوائده" (558/1): ((وَهَذَا أَقْرَبُ إِلَى الصَّوَابِ إِنْ شَاءَ اللَّهُ)).
    وكذا رجح الدارقطني وقفه على سالم ، فقال في "العلل" (53/2): ((وَالصَّوَابُ عَنْ سَالِمٍ)).

    - ورواه يحيي بن العلاء ، عنه ، عن أيوب ، عن سالم ، عن ابن عمر به مرفوعاً.
    أخرجه ابن منده في "مجالس من أماليه" (52).
    وقال ابن منده: ((غريب من حديث أيوب لا يُعرف عنه إلا من هذا الوجه)).
    قلت: يحيي بن العلاء البجلي ، متروك رُمي بالوضع.

    وله طريق اخر عن ابن عمر يرويه مَرْوَانُ بْنُ مُحَمَّدٍ الطَّاطَرِيُّ، ثنا الْوَلِيدُ بْنُ عُتْبَةَ، ثنا مُحَمَّدُ بْنُ سُوقَةَ، عَنْ نَافِعٍ، عَنِ ابْنِ عُمَرَ، عَنْ عُمَرَ بن الخطاب به.
    أخرجه الطبراني في "الدعاء" (798) ، وأبو نعيم في "الحلية" (13/5) ، وفي "تاريخ أصبهان" (323/1) ، ومن طريقه ابن عساكر في "تاريخ دمشق" (330/53).
    وقال أبو نعيم: ((غَرِيبٌ مِنْ حَدِيثِ مُحَمَّدٍ، تَفَرَّدَ بِهِ مَرْوَانُ، عَنِ الْوَلِيدِ)).

    والوليد بن عتبة ، قال المزي في ترجمة الوليد بن عتبة الدمشقي في "تهذيب الكمال" (50/31): ((روى مَرَوَان بْن مُحَمَّد الدمشقي الطاطري عَنِ الْوَلِيد بْن عتبة، عَنْ مُحَمَّد بْن سوقة، فلا أدري هُوَ [الدمشقي] أو غيره)) ، وقال الحافظ ابن حجر في "تهذيب التهذيب" (142/11): ((الظاهر أنه هو هذا [أي الدمشقي])).
    والوليد بن عتبة الدمشقي ، قال البخاري: ((معروف الحديث)) ، بينما قال أبو حاتم: ((مجهول)) ، وقال الذهبي: ((لا يُدرى من هو)) ، وقال أيضاً: ((مجهول)) ، وقال ابن حجر في "التقريب": ((مستور)) ، بينما قال في "اللسان": ((مجهول)).
    انظر: "التاريخ الكبير" (150/8) ، "الجرح والتعديل" (13/9) ، "الميزان" (341/4) ، "المغنى في الضعفاء" (723/2).

    حديث أبي هريرة
    أخرج الترمذي (3432) عن أَبي جَعْفَرٍ السِّمْنَانِيُّ ، وَغَيْرُ وَاحِدٍ ، والطبراني في "الدعاء" (799) عن أَبي زُرْعَةَ عَبْدُ الرَّحْمَنِ بْنُ عَمْرٍو الدِّمَشْقِيُّ، وَعَبْدُ الرَّحْمَنِ بْنُ مَعْدَانَ بْنِ جُمُعَةَ اللَّاذِقِيُّ ، جميعهم عن مُطَرِّفُ بْنُ عَبْدِ اللَّهِ المَدِينِيُّ قَالَ: حَدَّثَنَا عَبْدُ اللَّهِ بْنُ عُمَرَ العُمَرِيُّ، عَنْ سُهَيْلِ بْنِ أَبِي صَالِحٍ، عَنْ أَبِيهِ، عَنْ أَبِي هُرَيْرَةَ، قَالَ: قَالَ رَسُولُ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ: " مَنْ رَأَى مُبْتَلًى، فَقَالَ: الحَمْدُ لِلَّهِ الَّذِي عَافَانِي مِمَّا ابْتَلَاكَ بِهِ، وَفَضَّلَنِي عَلَى كَثِيرٍ مِمَّنْ خَلَقَ تَفْضِيلًا، لَمْ يُصِبْهُ ذَلِكَ البَلَاءُ "

    وأخرج البيهقي في "شعب الإيمان" (4129) عن محمود بن هشام ، ويعقوب بن سفيان ، والطبراني في "المعجم الأوسط" (4742) ، وفي "المعجم الصغير" (675) عن عَبْدُ الرَّحْمَنِ بْنُ مَعْدَانَ بْنِ جُمُعَةَ اللَّاذِقِيُّ، وَأَبي زُرْعَةَ ، والبزار (6217) عن عَبد الله بن شَبِيب ، والخرائطي في "فضيلة الشكر" (3) عن عباس بن محمد الدوري ، وأبو الفضل الأزهري في "حديثه" (421) عن هَارُونُ بْنُ سُفْيَانَ الْمُسْتَمْلِيّ ُ ، وأبو الطيب الحوراني في "حديثه" (16) ، ومن طريقه ابن الشيخة في"الثاني من شعار الأبرار" (2166) عن يزيد بن محمد بن عبد الصمد الدمشقي ، وأبو بكر الأنصاري في "مشيخته" (646) عن يعقوب بن سفيان ، وابن عدي في "الكامل" (235/5) عن أَحْمَدُ بْنُ دَاوُدَ بْنِ أَبِي صالح الحراني ، جميعهم عن مُطَرِّفُ بْنُ عَبْدِ اللَّهِ المَدِينِيُّ قَالَ: حَدَّثَنَا عَبْدُ اللَّهِ بْنُ عُمَرَ العُمَرِيُّ، عَنْ سُهَيْلِ بْنِ أَبِي صَالِحٍ، عَنْ أَبِيهِ، عَنْ أَبِي هُرَيْرَةَ، قَالَ: قَالَ رَسُولُ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ: "إِذَا رَأَى أَحَدُكُمْ مُبْتَلًى فَلْيَقُلْ: الْحَمْدُ لِلَّهِ الَّذِي فَضَّلَنِي عَلَيْهِ وَعَلَى كَثِيرٍ مِنْ عِبَادِهِ تَفْضِيلًا، فَإِذَا قَالَ ذَلِكَ فَقَدْ شَكَرَ تِلْكَ النِّعْمَةِ" وهذا لفظ الطبراني.

    قال الترمذي: ((هَذَا حَدِيثٌ حَسَنٌ غَرِيبٌ مِنْ هَذَا الوَجْهِ)).
    وقال ابن الشيخة: ((هَذَا حَدِيثٌ حَسَنٌ)).
    وقال المنذري في "الترغيب" (139/4): ((إسناده حسن)).
    وقال الطبراني: ((لَمْ يَرْوِ هَذَا الْحَدِيثَ عَنْ سُهَيْلِ بْنِ أَبِي صَالِحٍ إِلَّا عَبْدُ اللَّهِ بْنُ عُمَرَ، تَفَرَّدَ بِهِ: مُطَرِّفُ بْنُ عَبْدِ اللَّهِ)).
    وقال البزار: ((لا نَعْلَمُهُ يُرْوَى عَنْ أَبِي هُرَيْرَةَ إِلا بِهَذَا الإِسْنَادِ، وَعَبْدُ اللَّهِ بْنُ عُمَرَ قَدِ احْتَمَلَ أَهْلُ الْعِلْمِ حَدِيثَهُ)).
    وقال ابن عدي: ((وَهَذَا لا أَعْلَمُ يَرْوِيهِ عَنْ عَبد اللَّهِ بْنِ عُمَر غير أبي مصعب مطرف هذا)).

    قلت: والراجح لفظ "فقد شكر تلك النعمة" ، حيث تابع مطرف عليه بهذا اللفظ:
    - محمد بن سنان الباهلي.
    أخرجه الواحدي في "الوسيط" (118/3) ، وابن أبي الدنيا في "الشكر" (187) ، والبيهقي في "شعب الإيمان" (10634).
    - أبو بكر العوفي.
    أخرجه الخرائطي في "فضيلة الشكر" (3).

    وعبد الله بن عمر العمري ، ضعيف الحديث ، وتابعه عبد الله بن جعفر بن نجيح المدني.
    أخرجه الطبراني في "الدعاء" (800).
    وعبد الله بن جعفر ، قال ابن معين: (( ليس بشىء)) ، وقال النسائي: ((متروك الحديث)) ، وقال أبو حاتم: ((منكر الحديث جدا، ضعيف الحديث، يحدث عن الثقات بالمناكير، يكتب حديثه ولا يحتج به)) ، وقال الذهبي: ((مُجْمَعٌ عَلَى ضَعْفِهِ)).

    وأخرج الطبراني في "الدعاء" (801) من طريق عَبْدُ اللَّهِ بْنُ صَالِحٍ، حَدَّثَنِي اللَّيْثُ، عَنْ عِيسَى بْنِ مُوسَى بْنِ إِيَاسِ بْنِ الْبُكَيْرِ، عَنْ صَفْوَانَ بْنِ سُلَيْمٍ، عَنْ رَجُلٍ، عَنْ أَبِي هُرَيْرَةَ به.
    وعيسى بن موسى ، قال أبو حاتم: ((ضعيف)) ، والإسناد فيه جهالة.

    حديث أنس بن مالك
    أخرج البزار (6218) عن عَبد اللَّهِ بْنُ شَبِيب، حَدَّثنا مطرف بن عَبد الله، وَإسحاق الفروي، حَدَّثنا عَبد اللَّهِ بْنُ عُمَر، عَن عَبد رَبِّهِ بْنِ سَعِيد، عَن أَنَسٍ، عَن النَّبِيّ صَلَّى اللَّهُ عَلَيه وَسَلَّم به.
    وعبد الله بن شبيب متروك.

    حديث عبد الله بن عباس
    أخرج تمام في "فوائده" (1459) عن جَعْفَرُ بْنُ مُحَمَّدِ بْنِ جَعْفَرِ بْنِ رُشَيْدٍ الْكُوفِيُّ، ثنا سُلَيْمَانُ بْنُ عَبْدِ الرَّحْمَنِ، ثنا نَاشِبُ بْنُ عَمْرٍو، ثنا مُقَاتِلُ بْنُ حَيَّانَ، عَنْ عَبْدِ اللهِ بْنِ أَبِي مُلَيْكَةَ، عَنِ ابْنِ عَبَّاسٍ، قَالَ: قَالَ رَسُولُ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ: " مَا مِنْ عَبْدٍ مُؤْمِنٍ يَنْظُرُ إِلَى صَاحِبِ بَلَاءٍ مَا كَانَ مِنْ بَلَائِهِ، فَيَحْمَدُ اللَّهَ عَلَى عَافِيَتِهِ، ثُمَّ يَقُولُ: اللَّهُمَّ عَافِنِي مِمَّا ابْتَلَيْتَهُ، وَتَمِّمْ عَلَيَّ نِعْمَتَكَ إِلَّا عَافَاهُ اللَّهُ عَزَّ وَجَلَّ مِنْ ذَلِكَ الْبَلَاءِ فَلَنْ يُبْتَلَى بِهِ أَبَدًا "
    وجعفر بن محمد ، مجهول.
    وناشب بن عمرو ، قال الدارقطني: ((ضعيف)) ، وقال البخاري: ((منكر الحديث)).

    حديث حذيفة بن أوس.
    أخرج ابن الأثير في "أسد الغابة" (704/1) من طريق عَبْدُ اللَّهِ بْنُ مُحَمَّدِ بْنِ يُوسُفَ الْعَبْدِيُّ، أخبرنا عَبْدُ اللَّهِ بْنُ أَبَانِ بْنِ عُثْمَانَ بْنِ حُذَيْفَةَ بْنِ أَوْسٍ، قَالَ: حَدَّثَنِي أَبَانُ بْنُ عُثْمَانَ، عن أَبِيهِ عُثْمَانَ بْنِ حُذَيْفَةَ، عن جَدِّهِ حُذَيْفَةَ بْنِ أَوْسٍ، قَالَ: قَالَ رَسُولُ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ: مَنْ رَأَى مُبْتَلًى فَقَالَ: الْحَمْدُ للَّهِ الَّذِي عَافَانِي مِمَّا ابْتَلاكَ بِهِ، وَفَضَّلْنِي عَلَى كَثِيرٍ مِنْ خَلْقِهِ تَفْضِيلًا، إِلا عَافَاهُ اللَّهُ مِنْ ذَلِكَ الْبَلاءِ، كَائِنًا مَا كَانَ.

    وعبد الله بن محمد بن يوسف ، قال ابن عدي: ((مجهول)) ، وقال ابن يونس: ((لم يكن بذاك تعرف وتنكر)).
    وعبد الله بن أبان ، قال ابن عدي: ((ليس بالمعروف ، حدث عن الثقات بالمناكير)).
    وأبان بن عثمان ، وأبوه ، مجهولان.

    والله أعلم.
    طويلب علم
