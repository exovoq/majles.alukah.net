    منظومة أسماء الله الحسني لسيدي أحمد الدردير

    السلام عليكم ورحمة الله وبركاته إخواني أهلَ مجلس الألوكة، فإنّي أحمد اللهَ إليكم
    وبعد فإنّي كنت أبحث عن أحد المنظومات في العقيدة لأحفظها، فعثرت قدرا على هذه المنظومة المباركة، فاستحسنتها على - ضحالة علمي - وأردت حفظها، إلاّ أنني أردت مشورتكم ونُصحكم،فأنتم شيوخي وأساتذتي وأهلي، بارك الله في مجلسنا الموقر وحباه بألطافه عز وجل، فإن أنتم أذنتم بحفظها فالمرجو من جنابكم ضبط وشكل أبياتها حتَّى يسهُل حفظها على المبتدئين أمثالي،جزاكم الله عني وعن المسلمين خيرا، وإن رأيتم أنّ هناك منظومة أيسر في حفظها، وأنفع في مضمونها وأغزرَ في علمها و أبلغَ في بيانها فالمرجو الإشارة إليها وإرشادنا وتوجيهنا لما هو خير لنا،بارك الله فيكم وسدَّد خُطاكم وجزاكم الله خيرا
    أخوكم أبو سعد معاذ بن حسن المساوي المراكشي المغربي
    ودونكم المنظومة واسمها منظومة أسماء الله الحسني لسيدي أحمد الدردير
    يقول الناظم رحمه الله
    تـبـاركــت يــــا الله ربــــي لــــك الـثـنــا ~ فـحــمــدًا لـمــولانــا وشــكـــرًالـرب ــنــا
    بأسمـائـك الحـسـنـى وأسـرارهــا الـتــي ~ أقمـت بهـا الأكـوان مــن حـضـرة الفـنـا
    فـنـدعـوك يـــا الله يـــا مـبــدع الـــورى ~ يقـيـنـًا يقـيـنـا الـهــم والـكــرب والـعـنـا
    ويـــا رب يـــا رحـمــن هـبـنـا مـعـارفـًا ~ ولـطـفــًا وإحـسـانــًا ونـــــورًا يـعـمـنــا
    وســـر يـــا رحـيــم العالـمـيـن بجمـعـنـا ~ إلـى حـضـرة الـقـرب المـقـدس واهـدنـا
    ويـــا مـالــك مــلــك جـمـيــع عـوالـمــي ~ لروحـي وخـلـص مــن ســواك عقولـنـا
    وقـدس أيـا قـدوس نفـسـي مــن الـهـوى~ وسلـم جميـعـي يــا ســلام مــن الضـنـى
    ويــا مـؤمـن هــب لــي أمـانـًا وبـهـجـةً ~ وجـمّــل جـنـانـي يـــا مهـيـمـن بالـمـنـى
    وجـــد لـــي بـعــزٍ يـــا عـزيــز وقـــوةٍ ~ وبـالـجـبـر يــــا جــبــار بــــدد عــدونــا
    وكـبــر شــؤونــي فــيــك يــــا مـتـكـبـر ~ ويــا خـالـق الأكـــوان بالـفـيـض عـمـنـا
    ويـا بـارئ احفظنـي مــن الخـلـق كلـهـم ~ بفضـلـك واكـشـف يــا مـصـور كـربـنـا
    وبالـغـفـر يـــا غـفــار مـحــص ذنـوبـنـا ~ وبالـقـهـر يــــا قــهــار اقــهــر عــدونــا
    وهــب لــي أيــا وهــاب علـمـًا وحكـمـةً ~ ولـلـرزق يـــا رزاق وســـع وجـــد لـنــا
    وبالـفـتـح يـــا فــتــاح عــجــل تـكـرمــًا ~ وبـالـعـلـم نــــور يــــا عـلــيــم قـلـوبـنــا
    ويـا قابـض اقبضـنـا عـلـى خـيـر حـالـةٍ ~ ويـــا بـاســط الأرزاق بـسـطـاً لـرزقـنـا
    ويـا خافـض اخفـض لـي القلـوب تحببـا ~ ويـا رافـع ارفــع ذكـرنـا واعــل قـدرنـا
    وبـالـزهــد والـتـقــوى مــعـــزٌ أعــزنـــا ~ وذلــــل بـصـفــوٍ يــــا مــــذل نـفـوسـنـا
    ونــفــذ بــحــق ٍ يــــا سـمـيــع مـقـالـتــي ~ وبـصــر فـــؤادي يـــا بـصـيـر بعـيـبـنـا
    ويـــا حـكــم يـــا عـــدل حـكــم قـلـوبـنـا ~ بـعـدلـك فـــي الأشـيــا وبـالـرشـد قـونــا
    وحـــف بـلـطـف يـــا لـطـيـف أحـبـتـي ~ وتوجـهـموا بالـنـور كــي يـدركـوا الـمـنـى
    وكـــن يـــا خـبـيــراً كـاشـفــاً لـكـروبـنـا ~ وبـالـحـلــم خــلـــق يـاحـلـيــم نـفـوسـنــا
    وبالـعـلـم عـظــم يـــا عـظـيـم شـؤونـنــا ~ وفـــي مـقـعـد الـصــدق الأجـــل أحـلـنـا
    غـفــورٌ شـكــورٌ لـــم تـــزل متـفـضـلاً ~ فبالشـكـر والـغـفـران مـــولاي خُـصـنـا
    عـلـي ٌ كبـيـرٌ جــل عــن وهـــمِ واهـــمٍ ~ فسبحانـك اللهم عـن وصـف مـن جـنـى
    وكـن لـي حفيظـًا يــا حفـيـظ مــن الـبـلا ~ مـقـيـت ٌ أقـتـنـا خـيــر قــــوت ٍ وهـنـنــا
    وأنـت غياثـي يــا حسـيـب مــن الــردى ~ وأنـــت مـــلاذي يـــا جـلـيـل ُ وحسـبـنـا
    وجـد يـا كريمـاً بالعطـا مـنـك والـرضـا ~ وتـزكـيـة الأخـــلاق والـجــود والـغـنـى
    رقـيــبٌ علـيـنـا فـاعــفُ عـنــا وعـافـنـا ~ ويـســر علـيـنـا يــــا مـجـيــب أمــورنــا
    وياواسـعـاً وســـع لـنــا الـعـلـم والـعـطـا ~ حكيـمـًا أنـلـنـا حـكـمـةً مـنــك تـهـدنـا
    ودودٌ فــجــد بــالـــود مــنـــك تـكــرمــًا ~ علـيـنـا وشـــرف يـــا مـجـيـد ُ شـؤونـنـا
    ويــا بـاعـث ابعثـنـا عـلـى خـيـر حـالــةٍ ~ شـهـيــد ٌ فـأشـهـدنـا عــــلاك بـجـمـعـنـا
    ويــــا حــــق حـقـقـنـا بــســـر مــقـــدس ~ وكـيــل تـوكـلـنـا عـلـيــك بــــك اكـفـنــا
    قـــويٌ مـتـيـن ٌ قـــو عـزمــي وهـمـتــي ~ ولــــي ٌ حـمـيــدٌ لــيــس إلا لــــك الـثـنــا
    ويـا محصـي الأشيـاء يـا مبـدي الـورى ~ تـعـطــف عـلـيـنــا بـالـمـســرة والـهــنــا
    أعـدنــا بــنــور ٍ يــــا مـعـيــد ُ وأحـيـنــا ~علـى الديـن يـا محـيي الأنــام مــن الفـنـا
    مـمـيــتٌ أمـتـنــي مـسـلـمــا ً ومــوحـــدًا ~ وشــرف بــذا قــدري كـمـا أنــت ربـنــا
    ويـــا حـــي يـــا قـيــوم قــــوم أمــورنــا ~ ويــــا واجــــدُ أنــــت الـغـنــي فـأغـنـنـا
    ويـــا مـاجــد شـــرف بـمـجـدك قـدرنــا ~ ويـــا واحـــدُ فـــرج كـروبــي وغـمـنــا
    ويــا صـمـد فـوضـت أمـــري إلـيــك ~ لاتكـلـنـي لـنـفـسـي واهــدنــا رب سـبـلـنـا
    ويـا قـادر اقـدرنـا عـلـى صـدمـة الـعـدا ~ ومـقـتـدرٌ خـلــص مـــن الـغـيـر سـرنــا
    وقــــدم أمــــوري يــــا مــقـــدم هـيــبــةً ~وأخــــر عــدانــا يــــا مــؤخــر بـالـعـنـا
    ويـــا أول مــــن غــيــر بــــدءٍ وآخــــر~ بغـيـر انتـهـاءٍ أنــت فــي الـكـل حسـبـنـا
    ويــا ظـاهـراً فــي كــل شــيءٍ شـؤونــه ~ ويــا باطـنـًا بالـغـيـب لازلـــت محـسـنـا
    ويــــا والــيــًا لـسـنــا لـغـيــرك نـنـتـمـي~ فبالـنـصـر يـــا متعـالـيـا ً كـــن مـعـزنــا
    ويــا بــر يــا تـــواب جـــد لـــي بـتـوبـة ~ نـصـوح ٍ بـهـا تمـحـو عـظـائـم جـرمـنـا
    ومنـتـقـم هــــاك انـتـقــم مــــن عــدونــا ~ عـفــوٌ رؤوف ٌ عـافـنــا وارأفــــن بــنــا
    ويـــا مـالــك الـمـلـك الـعـظـيـم بـقـهــره ~ ويـا ذا الجـلال الطـف بنـا فــي أمـورنـا
    ويـــــا مـقــســط بـالاسـتـقـامـة قـــونـــا ~ ويـــا جـامــع فـاجـمـع عـلـيــك قـلـوبـنـا
    غـنــيٌ ومـغــن أغـنـنـا بــــك ســيــدي~ ومــانــعٌ امــنــع كــــل كــــربٍ يـهـمـنـا
    ويــا ضــار ضــر المعـتـديـن بظلـمـهـم ~ ويــــا نــافــع انـفـعـنـا بــأنــوار ديـنــنــا
    ويــا نــور نــور ظـاهــري وسـرائــري ~ بـحـبـك يـــا هــــادي وقــــوم طـريـقـنـا
    بــديـــعٌ فـأتـحـفـنــا بـــدائـــع حــكــمــةٍ ~ويـــا بـاقـيـًا بـــك أبـقـنــا فــيــك أفـنـنــا
    ويـــا وارثـــًا ورثـنــي عـلـمـاً وحـكـمـةً ~ رشـيــدٌ فـأرشـدنـا إلـــى طــــرق الـثـنــا
    وأفـرغ علينـا الصبـر بالشكـر والرضـا ~ وحـســن يـقـيـنٍ يــــا صــبــور ووفــنــا
    بأسمـائـك الحـسـنـى دعـونــاك سـيــدي~ تـقـبـل دعـانــا ربــنــا واسـتـجــب لــنــا
    بأسـرارهـا عـمــر فـــؤادي وظـاهــري ~ وحـقـق بـهـا روحــي لأظـفــر بالـمـنـى
    ونــور بـهـا سمـعـي وشـمـي ونـاظـري~ وقـــو بـهــا ذوقـــي ولـمـسـي وعـقـلـنـا
    ويـســربـهــا أمـــري وقـــو عـزائـمــي ~ وزكي بــهــا نـفـســي وفــــرج كـروبـنــا
    ووســع بـهـا علـمـي ورزقــي وهـمـتـي ~ وحسـن بـهـا خَلـقـي وخُلـقـي مــع الهـنـا
    وهــب لــي بـهـا حـبــًا جـلـيـلاً مـجـمـلاً ~ وزدنـــي بـفــرط الـحــب فـيــك تـفـنـنـا
