    إِنَّهُ لَيْسَ مِنْ فَرَسٍ عَرَبِيٍّ إِلَّا يُؤْذَنُ لَهُ مَعَ كُلِّ فَجْرٍ يَدْعُو بِدَعْوَتَيْنِ ..

    قال أحمد في مسنده 21497 : حَدَّثَنَا يَحْيَى بْنُ سَعِيدٍ، عَنْ عَبْدِ الْحَمِيدِ بْنِ جَعْفَرٍ، حَدَّثَنِي يَزِيدُ بْنُ أَبِي حَبِيبٍ، عَنْ سُوَيْدِ بْنِ قَيْسٍ، عَنْ مُعَاوِيَةَ بْنِ حُدَيْجٍ، عَنْ أَبِي ذَرٍّ، قَالَ: قَالَ رَسُولُ اللهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ:
    إِنَّهُ لَيْسَ مِنْ فَرَسٍ عَرَبِيٍّ إِلَّا يُؤْذَنُ لَهُ مَعَ كُلِّ فَجْرٍ يَدْعُو بِدَعْوَتَيْنِ يَقُولُ: اللهُمَّ إِنَّكَ خَوَّلْتَنِي مَنْ خَوَّلْتَنِي مِنْ بَنِي آدَمَ، فَاجْعَلْنِي مِنْ أَحَبِّ أَهْلِهِ وَمَالِهِ إِلَيْهِ، أَوْ أَحَبَّ أَهْلِهِ وَمَالِهِ إِلَيْهِ .
    قَالَ أَبُو عَبْدِ الرَّحْمَنِ: قَالَ أَبِي: خَالَفَهُ عَمْرُو بْنُ الْحَارِثِ، فَقَالَ: عَنْ يَزِيدَ، عَنْ عَبْدِ الرَّحْمَنِ بْنِ شِمَاسَةَ، وقَالَ لَيْثٌ: عَنْ ابْنِ شِمَاسَةَ، أَيْضًا
    وقال أحمد في مسنده 21442 : حَدَّثَنَا حَجَّاجٌ، وَهَاشِمٌ، قَالَا: حَدَّثَنَا لَيْثٌ، حَدَّثَنِي يَزِيدُ بْنُ أَبِي حَبِيبٍ، عَنْ ابْنِ شِمَاسَةَ: أَنَّ مُعَاوِيَةَ بْنَ حُدَيْجٍ مَرَّ عَلَى أَبِي ذَرٍّ، وَهُوَ قَائِمٌ عِنْدَ فَرَسٍ لَهُ فَسَأَلَهُ: مَا تُعَالِجُ مِنْ فَرَسِكَ هَذَا؟ فَقَالَ: "إِنِّي أَظُنُّ أَنَّ هَذَا الْفَرَسَ قَدْ اسْتُجِيبَ لَهُ دَعْوَتُهُ ".
    قَالَ: وَمَا دُعَاءُ الْبَهِيمَةِ مِنَ الْبَهَائِمِ؟
    قَالَ: "وَالَّذِي نَفْسِي بِيَدِهِ، مَا مِنْ فَرَسٍ إِلَّا وَهُوَ يَدْعُو كُلَّ سَحَرٍ فَيَقُولُ: اللهُمَّ أَنْتَ خَوَّلْتَنِي عَبْدًا مِنْ عِبَادِكَ، وَجَعَلْتَ رِزْقِي بِيَدِهِ، فَاجْعَلْنِي أَحَبَّ إِلَيْهِ مِنْ أَهْلِهِ وَمَالِهِ وَوَلَدِهِ "
    قَالَ عَبْدُ اللهِ بْنُ أَحْمَدَ قَالَ أَبِي: "وَوَافَقَهُ عَمْرُو بْنُ الْحَارِثِ عَنْ ابْنِ شِمَاسَةَ "

    رد: إِنَّهُ لَيْسَ مِنْ فَرَسٍ عَرَبِيٍّ إِلَّا يُؤْذَنُ لَهُ مَعَ كُلِّ فَجْرٍ يَدْعُو بِدَعْوَتَيْنِ ..

    المشاركة الأصلية كتبت بواسطة احمد ابو انس
    قال أحمد في مسنده 21497 : حَدَّثَنَا يَحْيَى بْنُ سَعِيدٍ، عَنْ عَبْدِ الْحَمِيدِ بْنِ جَعْفَرٍ، حَدَّثَنِي يَزِيدُ بْنُ أَبِي حَبِيبٍ، عَنْ سُوَيْدِ بْنِ قَيْسٍ، عَنْ مُعَاوِيَةَ بْنِ حُدَيْجٍ، عَنْ أَبِي ذَرٍّ، قَالَ: قَالَ رَسُولُ اللهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ:
    إِنَّهُ لَيْسَ مِنْ فَرَسٍ عَرَبِيٍّ إِلَّا يُؤْذَنُ لَهُ مَعَ كُلِّ فَجْرٍ يَدْعُو بِدَعْوَتَيْنِ يَقُولُ: اللهُمَّ إِنَّكَ خَوَّلْتَنِي مَنْ خَوَّلْتَنِي مِنْ بَنِي آدَمَ، فَاجْعَلْنِي مِنْ أَحَبِّ أَهْلِهِ وَمَالِهِ إِلَيْهِ، أَوْ أَحَبَّ أَهْلِهِ وَمَالِهِ إِلَيْهِ .
    قَالَ أَبُو عَبْدِ الرَّحْمَنِ: قَالَ أَبِي: خَالَفَهُ عَمْرُو بْنُ الْحَارِثِ، فَقَالَ: عَنْ يَزِيدَ، عَنْ عَبْدِ الرَّحْمَنِ بْنِ شِمَاسَةَ، وقَالَ لَيْثٌ: عَنْ ابْنِ شِمَاسَةَ، أَيْضًا
    هذا الإسناد معل بالشذوذ، قال الحاكم في المستدرك (1/177) : "فَإِنَّ أَهْلَ مِصْرَ أَعْرَفُ بِحَدِيثِهِمْ مِنْ أَهْلِ الْبَصْرَةِ". اهـ.
    وكما أشار الإمام أحمد، قال الدارقطني كما في العلل [1123] له:
     يَرْوِيهِ يَزِيدُ بْنُ أَبِي حَبِيبٍ، وَاخْتُلِفَ عَنْهُ؛
    فَرَوَاهُ عَبْدُ الْحَمِيدِ بْنُ جَعْفَرٍ، عَنْ يَزِيدَ، عَنْ سُوَيْدِ بْنِ قَيْسٍ، عَنْ مُعَاوِيَةَ بْنِ حُدَيْجٍ، عَنْ أَبِي ذَرٍّ، عَنِ النَّبِيِّ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ، قَالَ ذَلِكَ يَحْيَى الْقَطَّانُ، عَنْ عَبْدِ الْحَمِيدِ.
    وَوَقَفَهُ غَيْرُ يَحْيَى، عَنْ عَبْدِ الْحَمِيدِ.
    وَكَذَلِكَ رَوَاهُ اللَّيْثُ، عَنْ يَزِيدَ بْنِ أَبِي حَبِيبٍ مَوْقُوفًا أَيْضًا، وَهُوَ الْمَحْفُوظُ". اهـ.
    ولعله يقصد بالذي غير يحيى هو عمرو بن الحارث، ذكره الإمام أحمد.
    المشاركة الأصلية كتبت بواسطة احمد ابو انس
    وقال أحمد في مسنده 21442 :
    حَدَّثَنَا حَجَّاجٌ، وَهَاشِمٌ، قَالَا: حَدَّثَنَا لَيْثٌ، حَدَّثَنِي يَزِيدُ بْنُ أَبِي حَبِيبٍ، عَنْ ابْنِ شِمَاسَةَ: أَنَّ مُعَاوِيَةَ بْنَ حُدَيْجٍ مَرَّ عَلَى أَبِي ذَرٍّ، وَهُوَ قَائِمٌ عِنْدَ فَرَسٍ لَهُ فَسَأَلَهُ: مَا تُعَالِجُ مِنْ فَرَسِكَ هَذَا؟ فَقَالَ: "إِنِّي أَظُنُّ أَنَّ هَذَا الْفَرَسَ قَدْ اسْتُجِيبَ لَهُ دَعْوَتُهُ ".
    قَالَ: وَمَا دُعَاءُ الْبَهِيمَةِ مِنَ الْبَهَائِمِ؟
    قَالَ: "وَالَّذِي نَفْسِي بِيَدِهِ، مَا مِنْ فَرَسٍ إِلَّا وَهُوَ يَدْعُو كُلَّ سَحَرٍ فَيَقُولُ: اللهُمَّ أَنْتَ خَوَّلْتَنِي عَبْدًا مِنْ عِبَادِكَ، وَجَعَلْتَ رِزْقِي بِيَدِهِ، فَاجْعَلْنِي أَحَبَّ إِلَيْهِ مِنْ أَهْلِهِ وَمَالِهِ وَوَلَدِهِ "
    قَالَ عَبْدُ اللهِ بْنُ أَحْمَدَ قَالَ أَبِي: "وَوَافَقَهُ عَمْرُو بْنُ الْحَارِثِ عَنْ ابْنِ شِمَاسَةَ "
    أورده ابن كثير في الجامع [12374]، فقال: "تفرد به"،
    قلتُ: لم يتفرد به الإمام أحمد، بل توبع بالروايتين، ولكن يبدو أنه وقع سهو في الرواية وهو قوله عن ابن شماسة مرسلًا، والصواب عن ابن شماسة عن معاوية بن حديج متصل.
    وذلك فيما أخرجه ابن الحكم في فتوح مصر (1/169) فقال:
    حدثنا أبى عبد الله بن عبد الحكم، وشعيب بن الليث، قالا: حدثنا الليث بن سعد، عن يزيد بن أبى حبيب، عن ابن شماسة، أن معاوية بن حديج حدثه، فذكره.
    وقال: حدثنا أحمد بن عمرو، حدثنا ابن وهب، عن عمرو بن الحارث، عن يزيد بن أبى حبيب، عن عبد الرحمن بن شماسة المهرى، عن معاوية بن حديج، فذكره.
    وأخرجه سعيد بن منصور في سننه [2444]، ومن طريقه السرقسطي في الدلائل [456]، فقال:
    نا عَبْدُ اللَّهِ بْنُ وَهْبٍ، قَالَ: أَخْبَرَنِي عَمْرُو بْنُ الْحَارِثِ، عَنْ يَزِيدَ بْنِ أَبِي حَبِيبٍ، عَنْ عَبْدِ الرَّحْمَنِ بْنِ شِمَاسَةَ، عَنْ مُعَاوِيَةَ بْنِ حُدَيْجٍ، فذكره.
    وأخرجه أبو الشيخ الأصبهاني في العظمة [1282]، فقال: أَخْبَرَنَا أَبُو يَعْلَى، حَدَّثَنَا هَارُونُ بْنُ مَعْرُوفٍ، حَدَّثَنَا ابْنُ وَهْبٍ، أَخْبَرَنِي عَمْرُو بْنُ الْحَارِثِ،به.
    وهذا إسناد صحيح موقوف، رجاله رجال مسلم، وله حكم الرفع، فإن مثله لا يقال بالرأي.
    وقد توبع ابن شماسة، وذلك فيما أخرجه أبو عبيدة بن المثنى في الخيل (ص:2)، فقال:
    قال عمر بن عمران السدوسي قال حدثنا عبد الرحمن بن زياد بن انعم عن يزيد بن أبي حبيب البصري عمن حدثه عن معاوية بن حديج:
    أنه لما افتتح مصر كان لكل قوم مراوغة يمرغون فيها خيولهم، فمر معاوية بأبي ذر وهو يمرغ فرسه، فسلم عليه ووقف ثم قال: يا أبي ذر ما هذا الفرس؟
    قال: " فرس لي لا أراه إلا مستجاباً"، قال: " وهل تدعوا الخيل وتجاب"، قال: نعم ليس من ليلة إلا والفرس يدعو فيها ربه فيقول:
    "رب إنك سخرتني الابن آدم وجعلت رزقي في يده الهم فاجعلني احب إليه من أهله وولده فمنها المستجاب ولا أرى فرسي هذا إلا مستجابا". اهـ.
    وقد ذكره ابن جزي الكلبي في الخيل (42) بسياق حسن، والدمياطي في الخيل (ص:66)،والدميري في حياة الحيوان الكبرى، وغيرهم.
    وقد وقع عند الدمياطي خطأ وهو قوله: " فمر حديج بن صومي بأبي ذر رضي الله عنه"، مما جعل من بعده ينقل عنه هكذا مثل شهاب الدين النويري وابن هذيل الفزاري الأندلسي.
    وهذا إسناد ضعيف، فيه عبد الرحمن بن زياد بن أنعم الإفريقي، "كان رجلا صالحا ضعيف في حفظه"، كذا قال الحافظ ابن حجر في التقريب.
    وقد ورد تسمية المبهم، وذلك فيما أخرجه سعيد بن منصور في سننه [2443]، فقال:
    نا فَرَجُ بْنُ فَضَالَةَ، قَالَ: نا عَبْدُ الرَّحْمَنِ بْنُ زِيَادِ بْنِ أَنْعُمٍ، عَنْ عَلِيِّ بْنِ رَبَاحٍ، عَنْ مُعَاوِيَةَ بْنِ حُدَيْجٍ، قَالَ:
    مَرَرْتُ بِأَبِي ذَرٍّ وَهُوَ يُمَرِّغُ فَرَسًا لَهُ، ثُمَّ أَخَذَ يَمْسَحُ بِثَوْبِهِ، فَقُلْتُ: وَاللَّهِ، إِنَّكَ لَتُحِبُّ فَرَسَكَ هَذَا ! قَالَ: " نَعَمْ، وَاللَّهِ إِنِّي لأَرَى هَذَا قَدِ اسْتُجِيبَ لَهُ ! " قُلْتُ: وَهَلْ يَدْعُو الْخَيْلُ؟ !
    قَالَ: " نَعَمْ، مَا مِنْ فَرَسٍ إِلا وَلَهُ دَعْوَةٌ يَدْعُو بِهَا فَمِنْهَا مَا يُسْتَجَابُ لَهُ، وَمِنْهَا مَا لا يُسْتَجَابُ لَهُ، يَقُولُ: اللَّهُمَّ، مَلَّكْتَنِي ابْنَ آدَمَ، وَجَعَلْتَ رِزْقِي بِيَدِهِ، فَاجْعَلْنِي أَحَبَّ إِلَيْهِ مِنْ أَهْلِهِ وَمَالِهِ، وَمَا أَرَى فَرَسِي هَذَا إِلا قَدِ اسْتُجِيبَ لَهُ ". اهـ.
    وهذا إسناد ضعيف، فالراوي عن الإفريقي هو فرج بن فضالة وهو "ضعيف"، كذا قال الحافظ ابن حجر في التقريب.
    ولكن ليس ضعفهما بالشديد، ويتقوى برواية ابن شماسة، فهو حسن لغيره.
    والله أعلم.
