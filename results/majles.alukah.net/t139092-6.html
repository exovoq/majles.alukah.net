    ــــ تَنْبِيهٌ :
    التَّنْبِيهُ هو طلبُ المتكلِّمِ من المخاطبِ إلقاءُ سمعِهِ لِمَا سيُلْقَى عليه منَ الكلامِ .
    ــ أَحْرُفُهُ : أَلاَ ، أَمَا ، هَا ، يَا .
    أَلاَ و أَمَا : يُسْتَفْتَحُ بها الكلامُ وتُفيدانِ تَنْبِيهَ السَّامِعِ إلى ما يُلْقَى إليه من الكلامِ . مثل : أَلاَ إِنَّ العِلْمَ مُفِيدٌ .
    أَمَا إِنَّ خَالِدًا مُسَافِرٌ .
    هَا : حرفٌ موضوعٌ لتنبيهِ المخاطبِ كالهاءِ الدّاخلةِ على اسمِ الاشارةِ . مثل : هَذَا كِتَابٌ جَيِّدٌ .
    يَا : أصلُهَا حرفُ نداءٍ . فإنْ لمْ يكنْ بعدها مُنَادَى كانتْ حرفًا يُقْصَدُ به تنبيهُ السّامعِ إلى ما بعدها . مثل :
    قوله تعالى ( يَا لَيْتَ قَوْمِي يَعْلَمُونَ ) .
    ــــ تَنْدِيمٌ :
    التَّنْدِيمُ : جعلُ الفاعلِ يَنْدَمُ على فواتِ الأمرِ وعلى التَّهَاوُنِ به .
    ــ حُرُوفُهُ :
    ــ هَلاَّ : مثل : هَلاَّ اجْتَهَدْتَ . تُقَرِّعُ السَّامِعَ على إِهْمَالِهِ و تُوَبِّخُهُ على عدمِ الاجتهادِ فَتَجْعَلُهُ يَنْدَمُ على ما فَرَّطَ
    و ضَيَّعَ ، وهكذا يُقَالُ في بقيَّةِ الأمثلَةِ .
    ــ أَلاَّ : مثل : أَلاَّ زُرْتَنَا .
    ــ لَوْمَا : مثل : لَوْمَا اجْتَهَدِتَ .
    ــ أَلاَ : مثل : أَلاَ تُبْتَ .
    فهذه الحروفُ هي حروفُ التَّحْضِيضِ إذا دخلَتْ على الفعلِ المضارعِ كما قَدَّمْنَا .
    وإذا دخلَتْ على الماضي كانتْ حُرُوفَ تَنْدِيمٍ كما مَثَّلْنَا .

    ــــ تَنْوِينٌ :
    التنوينُ نونٌ ساكنةٌ تَلْحَقُ آخِرَ الاسْمِ لَفْظًا و تُفَارِقُهُ خَطًّا . وهو أربعَةُ أقسامٍ :
    1ـــ تَنْوينُ التَّمْكِينِ : وهو اللاّحِقُ للأسماءِ المُعْرَبَةِ المُنْصَرِفَةِ . وفائدتُهُ الدّلالةُ على خِفَّةِ الاسْمِ وتَمَكُّنِهِ في بابِ الاسميَّةِ لكونِهِ لم يُشبهِ الحرفَ فَيُبْنَى ولا الفعلَ فيُمْنَعُ منَ الصَّرْفِ . مثل : رَجُلٌ ، مُحَمَّدٌ .
    2 ــ تَنْوِينُ التَّنْكِيرِ :هو اللاّحقُ للاسماءِ المبنيَّةِ فَرْقًا بينَ مَعْرِفَتِهَا ونَكِرَتِهَا ، ما نُوِّنٌ منها كانَ نكرَةً . مثل : سِيبَوَيْهٍ.
    وما لمْ يُنَوَّنْ كانَ مَعْرِفَةً . مثل : سِيبَوَيْهٍ بتركِ التَّنْوِينِ .
    3 ــ تَنْوِينُ العِوَضِ :هو على ثلاثةِ أقسامٍ :
    1ـ عِوَضٌ عَنْ جُمْلَةٍ : وهو اللاّحقُ لِإذٍ منْ حِينَئِذٍ و يَوْمَئِذٍ . قال تعالى ( وَ أَنْتُمْ حِينَئِذٍ تَنْظُرُونَ ) .
    والأصْلُ : وأنتمْ حينَ إذا بلغَتِ الرُّوحُ الحلقومَ تنظرونَ ، فحُذِفَتْ جُملةُ بَلَغَتِ الرّوحُ الحُلْقُومَ وأُتِيَ بتنوينِ إذٍ عِوَضًا عنها .فصارَ حينئذٍ تنظرونَ .
    2ـ عِوَضٌ عنْ كَلِمَةٍ : وهو تنوينُ كُلٍّ . قال تعالى ( قُلْ كُلٌّ يَعْمَلُ عَلَى شَاكِلَتِهِ ) أيْ كلُّ إنسانٍ .وكذا تنوينُ بَعْضٍ.
    3 ـ عِوَضٌ عَنْ حَرْفٍ : وهو اللاّحقُ للجُمُوعِ المعتلّةِ الآتيةِ على وَزْنِ فَوَاعِلَ. مثل : جَوَارٍ أصلُهَا جَوَارِيُ .أُسْتُثْقِلَتْ الضمَّةُ على الياءِ وعُوِّضَ عنها التّنوينُ .
    3 ـ تَنْوِينُ المُقَابَلَةِ : وهو اللاّحقُ لكُلِّ ما جُمِعَ بِأَلِفٍ و تَاءٍ مَزِيدَتَيْنِ و سُمِّيَ بذلك لأنّهمْ جعلوهُ في مُقَابَلَةِ النُّونِ في جمعِ المذكّرِ السّالِمِ . مثل : مُسَلَّمَاتٌ .

    ـــ تِـــــهْ :
    اسمُ إشارةٍ للمُفردةِ المؤنّثةِ ساكنةٌ ولكَ أنْ تكسِرَهَا فتُبْنَى على الكسرِ والسّكونِ ، يسبقها هاءُ التّنبيهِ فيقال :هَاتِهِ
    ـــ تَوَابِعٌ :
    جمعٌ مفردهُ تَابِعٌ ، وهو المشاركُ لما قبلَهُ في إعرابِهِ مُطْلَقًا .
    والتّوابع أرْبَعَةٌ : النَّعْتُ ، العَطْفُ ، التَّوكِيدُ ، البَدَلْ .
    ـــ تَوَانِيًا :
    التَّوَانِي هو التَّقْصِيرُ في الأمْرِ وعَدَمُ الاهْتِمَامِ به . مثل : أَتَوَانِيًا وَ قَدْ جَدَّ قُرَنَاؤُكَ . وإعرابها :
    ــ أَ : الهمزةُ للاستفهامِ .
    ــ تَوَانِيًا : مفعولٌ مطلقٌ لفعلٍ محذوفٍ وُجُوبًا .
    ــ وَ : الواوُ حاليّةٌ .
    ــ قَدْ : حرفُ تحقيقٍ .
    ـــ جَدَّ قُرَنَاؤُكَ : فعلٌ وفاعلٌ و مضافٌ إليه .
    ـــ تِي :
    اسمُ إشارةٍ للمفرَدَةِ المؤنَّثَةِ مبنيٌّ على السّكونِ . تُسْبَقُ بهاءٍ وهي حرفُ تنبيهٍ ، فيقالُ : هَاتِي . وهذه إشارةٌ للقريبِ
    وتَلْحَقُ تِي الكَافُ التي هي حرفُ خطابٍ فيقالُ : تِيكَ وهي إشارةٌ للبعيدِ ، وقدْ تَلْحَقُهَا هذه الكافُ مع اللاّمِ فيقالُ : تِلْكَ وهي إشارةٌ للبعيدِ أَيْضًا . مثل : تِلْكَ مُعَلِّمَةٌ .

    حرف الثّاء

    ــــ ثُمَّ :
    حرفُ عطفٍ يدلُّ على التَّرْتِيبِ والتَّرَاخِي . مثل : جَاءَ عَلِيٌّ ثُمَّ سَعِيدٌ .
    ـــ ثَمَّ :
    بمعنى هناكَ يُشَارُ بها إلى المكانِ البعيدِ . مبنيٌّ على الفَتْحِ في محلِّ نَصْبٍ على الظَّرْفِيَّةِ . وقدْ تُجَرُّ بِمِنْ وإلى .
    ـــ ثَلاَثَةَ عَشَرَ :
    مركّبٌ عدديٌّ يَجِبُ فَتْحُ جُزْءَيْهِ سواءٌ أكانَ :
    مرفوعًا . مثل : جَاءَ ثَلاَثَةَ عَشَرَ رَجُلاً .
    منصوبًا : مثل : رَأَيْتُ ثَلاَثَةَ عَشَرَ رَجُلاً .
    مجرورً : مثل : مَرَرْتُ بِثَلاَثَةَ عَشَرَ رَجُلاً .
    فهو حينئذٍ مبنيٌّ على فتْحِ جُزْءَيْهِ .
    محلّهُ من الإعرابِ :
    الرّفعُ في المثالِ الأوّلِ لأنّه فاعلٌ .
    النّصبُ في المثالِ الثّانِي لأنّهُ مفعولٌ به .
    والجرُّ في المثال الثّالثِ لأنّهُ في محلِّ جَرٍّ بالياءِ .

    حرف الجيم

    ــــ جَرٌّ :
    الجرُّ أحَدُ ألْقَابِ الإعْرَابِ ويُرَادِفُهُ الخَفْضُ . فالجَرُّ عبارةُ البصريّينَ والخفضُ عبارةُ الكوفيّينَ ، وسُمِّيَ جَرًّا لانْجِرَارِ أيْ انْخِفَاضِ الشَّفَةِ السُّفْلَى عِنْدَ النُّطْقِ بِهِ .
    ـــ جَرُّ الاسْمِ :
    الجرُّ خاصٌّ بالاسْماءِ فلا يكونُ في الحُرُوفِ ولا في الأفعالِ ، يُجَرُّ الاسْمُ إذا كانَ مسْبُوقًا بحرفٍ منْ حُروفِ الجَرِّ أوْ كانَ مُضَافًا إليهِ . مثل : مَرَرْتُ بِمَدْرَسَةِ خَالِدٍ . فمدرسةِ مجرورَةٌ لأنَّهَا مسْبُوقَةٌ بحرفِ جَرٍّ وهو البَاءُ . وخَالِدٍ اسْمٌ مجرورٌ لأنَّهُ مُضَافٌ إليهِ .

    ــــ جَزْمٌ :
    الجزمُ هو قطعُ الحركةِ منَ الفعلِ المضارعِ الصّحيحِ الآخِرِ . مثل : لَمْ يَخْرُجْ خَالِدٌ . والأصل : يَخْرُجُ .
    ــ لَمْ :حرفُ نفيٍ وجزمٍ و قَلْبٍ .
    ـــ يَخْرُجْ : فعلٌ مضارعٌ مجزومٌ بلَمْ و علامةُ جزمِهِ السّكونُ .
    ـــ خَالِدٌ : فاعلٌ .
    و قطعُ الحرفِ من الفعلِ المضارعِ المعتلِّ الآخرِ لأجلِ الجازِمِ . مثل : لَمْ يَرْمِ الصَيَّادُ الطَّائِرَ . والأصل يَرْمِي .
    ـــ لَمْ : حَرْفُ نَفْيٍ و جَزْمٍ و قَلْبٍ .
    ــــ يَرْمِ : فعلٌ مضارعٌ مجزومٌ بلَمْ وعلامةُ جَزْمِهِ حَذْفُ حَرْفِ العِلَّةِ مِنْ آخِرِهِ وهو اليَاءُ . والكَسْرَةُ قَبْلَهَا دَلِيلٌ عَلَيْهَا
    ـــ الصَيَّادُ : فاعلٌ مرفوعٌ .
    ـــ الطَّائِرَ : مفعولٌ به منصوبٌ .
    لِلْجَزْمِ عَلاَمَتَانِ : السُّكُونُ و حَذْفُ حَرْفِ العِلَّةِ في آخرِ الفعلِ .

    ــــ جَزْمُ الفِعْلِ :
    يُجزمُ من الأفعالِ الفعلُ المضارعُ و جزمُهُ إمَّا لَفْظِيٌّ و إمَّا مَحَلِيٌّ :
    ــ لَفْظِيٌّ : يكونُ جزمُ المضارعِ لفظيًّا إنْ كانَ مُعْرَبًا . مثل : لَمْ يَكْتُبْ .
    ــ لَمْ : حرفٌ جَازمٌ .
    ــ يَكْتُبْ : فعلٌ مضارعٌ مجزومٌ بلَمْ و علامةُ جزمِهِ سُكُونُ آخِرِهِ .
    ـــ مَحَلِيٌّ : يكونُ جزمُ المضارعِ مَحَلِيًّا إنْ كَانَ مَبْنيًّا . مثل : لاَ تَلْعَبَنَّ .
    ــ لاَ : النَّاهِيَةِ .
    ــ تَلْعَبَنَّ : فعلٌ مضارعٌ مبنيٌّ على الفَتْحِ في مَحَلِّ جَزْمٍ بلاَ النَّاهِيَةِ .
    ـــ جَزْمُهُ :

    حرف الحاء

    ــــ حَاشَا :
    فعلٌ ماضٍ ضُمِّنَ معنى إلاَّ الاستثنائيّةِ فاسْتُثْنِيَ به كما اسْتُثْنِيَ بإلاَّ . فالمستثنى بحَاشَا يجوزُ نَصْبُهُ و جَرُّهُ . فالنّصْبُ على أنّه فعلٌ ماضٍ و ما بعده مفعولٌ به . والفاعِلُ ضميرٌ مستترٌ يَعُودُ على المستثنى منه . والجرُّ على أنّهُ حرفُ جرٍّ شبيهٌ بالزَّائِدِ . مثل : جَاءَ القَوْمُ حَاشَا عَلِيًّا و حَاشَا عَلِيٍّ . وكلاهما صحيح .
    و قد تكونُ حَاشَا للتّنْزِيهِ دونَ الاستثناء فَيُجَرُّ ما بعدها إمّا باللاّمِ . مثل : حَاشَ لِلّهِ . وإمَّا باللإضافةِ إليها .
    مثل : حَاشَ اللهِ .
    ـــ حَبَّذَا :
    فعلٌ لإنْشَاْءِ المَدْحِ . مُرَكَّبٌ منْ حَبَّ و ذَا الإشاريّةِ . مثل : حَبَّذَا رَجُلاً عَدْنَانُ .
    ـــ حَبَّ : فعلٌ ماضٍ .
    ـــ ذَا : اسمُ إشَارَةٍ فَاعِلُهُ .
    ـــ رَجُلاً : تمييزٌ .
    ـــ عَدْنَانُ : مبتدأٌ مؤخَّرٌ . خَبَرُهُ جملةُ حَبَّذَا مُقَدَّمَةٌ عليهِ .
    ــــ حَبْسًا :
    منْ قولكَ حَبْسًا اللِصَّ . فَحَبْسًا مفعولٌ مطلقٌ بفعلٍ محذوفٍ تقديرُهُ احْبِسْ حَبْسًا . واللِصَّ : مفعولٌ به لِحَبْسًا .

    ـــــ حقًّا :
    مصدرٌ منصوبٌ بفعلٍ محذوفٍ تقديرهُ أَحَقُّ .
    ـــــ حَمُ :
    اسمٌ من الأسماءِ الخمسةِ . يُرْفَعُ بالواوِ و يُنْصَبُ بالأَلِفِ و يُجَرُّ باليَاءِ . مثل : مَرَرْتُ بِحَمِيكَ .
    ــــ حَمْدًا :
    مصدرٌ منصوبُ بفعلٍ محذوفٍ تقديرُهُ : أَحْمَدُ اللهَ حَمْدًا .
    ـــ حَيَّ :
    اسْمُ فعلِ أمرٍ بمعنى أَقْبِلْ . وفاعلهُ ضميرٌ مستترٌ وجوبًا تقديرهُ أَنْتَ . مثل : حَيَّ عَلَى الصَّلاَةِ : أيْ أقبلْ إليها وأَسْرِعْ .

    ـــــ حَيْثُ :
    ظرفٌ للمكانِ مبنيٌّ على الضّمِ في محلِّ نصبٍ على الظرفيّةِ . تُلاَزِمْ حَيْثُ الإضافةَ إلى الجملةِ . والأكثرُ إضافتها إلى الجملةِ الفعليّةِ . مثل : اجْلِسْ حَيْثُ يَجْلِسُ أَهْلُ الفَضْلِ . و تُضَافُ إلى الجملةِ الاسميّةِ . مثل : اجْلِسْ حَيْثُ خَالِدٌ جَالِسٌ . ولا تُضَافُ إلى المفرَدِ . فإنْ جَاءَ بعدها مفرَدٌ رُفِعَ على أنّهُ مبتدأٌ خَبَرُهُ مَحْذُوفٌ . مثل :
    اجْلِسْ حَيْثُ خَالِدٌ جَالِسٌ . و قَدْ تُجَرُّ بِمَنْ أَوْ إلَى . مثل : ارْجَعْ مِنْ حَيْثُ أَتَيْتَ إلَى حَيْثُ كُنْتَ .
    ـــ حَيْثُمَا :
    تَلْحَقُ مَا الكافةُ حَيْثُ فتكونُ اسْمَ شرْطٍ يَجْزِمُ فعلينِ الأوّلُ فعلُ الشّرْطِ والثّانِي جوابُهُ وجَزَاؤُهُ . مثل :
    حَيْثُمَا تَقْعُدْ أُقْعُدْ :
    ـــ حَيْثُمَا : اسْمُ شَرْطٍ جَازِمٌ مَبْنِيٌّ عَلَى السُّكُونِ في مَحَلِّ نَصْبٍ على الظَّرْفِيَّةِ بِفِعْلِ الشَّرْطِ .
    ــــ تَقْعُدْ : فِعْلُ الشَّرْطِ مَجْزُومٌ بِحَيْثُمَا . والفاعلُ مُسْتَتِرٌ تَقْدِيرُهُ أَنْتَ .
    ـــ أُقْعُدْ : جَوَابُ الشَّرْطِ مَجْزُومٌ بِسُكُونِ آخِرِهِ و الفَاعِلُ مُسْتَتِرٌ تَقْدِيرُهُ أَنَا .

    ـــــ حَيْصَ بَيْصَ :
    تركيبٌ مزجيٌّ مبنيٌّ على فتحِ جُزْءَيْهِ . مثل : وَقَعُوا فِي حَيْصَ بَيْصَ . أيْ في حَيْرَةٍ واخْتِلاَطٍ و شِدَّةٍ لا مَحِيصَ لَهُمْ عَنْهَا و لا مَفَرَّ . والحَيْصُ في الأَصْلِ العُدُولُ و الإنْحِرَافُ . والبَيْصُ في الأصْلِ : الشِدَّةُ و الضِّيقُ .
    ــــ حِينَ :
    الحِينُ هو الزَّمَانُ المُبْهَمُ . مثل : قَرَأْتُ حِينًا .
    ـــ قَرَأْتُ : فِعْلٌ و فَاعِلٌ .
    ـــ حِينًا : منصوبٌ على الظَّرْفِيَّةِ الزَّمَانِيَّةِ .
    وتُبْنَى حِينَ إذَا أُضِيفَتْ إلى جُمْلَةٍ صَدْرُهَا مَبْنِيٌّ كقولِ الشَّاعِرِ : عَلَى حِينِ عَاتَبْتُ المَشِيبَ عَلَى الصِّبَا .
    ـــــ حِينَئِذٍ :
    ( وَ أَنْتُمْ حِينَئِذٍ تَنْظُرُونَ )
    ـــ الوَاوُ : وَاوُ الحَالِ .
    ـــ أَنْ : ضميرٌ منفصِلٌ مبنيٌّ على السّكونِ في محلِّ رفعٍ مبتدأٌ .
    ـــ التَّاءُ : تَاءُ الخِطَابِ .
    ـــ المِيمُ : عَلاَمَةُ الجَمْعِ .
    ـــ حِينَ : ظرفُ زَمَانٍ منصوبٌ على الظرفيّةِ .
    ـــ إِذٍ : مضافٌ إليهِ مجرورٌ بكسرَةٍ ظَاهِرَةٍ .أَوْ أَنَّهَا مَبْنِيَّةٌ على السُّكُونِ و كُسِرَتْ للتَّخَلُّصِ مِنِ الْتِقَاءِ السَّاكِنَيْنِ .
    ـــ تَنْظُرُ : فعلٌ مضارَعٌ مرفوعٌ بُثُبُوتِ النُّونِ .
    ــــ الوَاوُ : فَاعِلٌ .
    ـــ حَيَّهَلَ :
    اسْمُ فِعْلِ أَمْرٍ مُتَعَدٍّ بِمَعْنَى ائْتِ . مثل : حَيَّهَلَ الأَمْرَ . أَيْ ائْتِهِ .
