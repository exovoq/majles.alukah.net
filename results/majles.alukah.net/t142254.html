    أما حديث عبد الله بن مسعود
    أخرجه أبو حنيفة (ت150) في [مسنده/ رواية الحصكفي/12-الآداب] .
    وأخرجه البلاذري (ت279) في [أنساب الأشراف/10/56-الفكر] فقال: وَحَدَّثَنِي الْحُسَيْنُ بْنُ عَلِيِّ بْنُ الأَسْوَدِ، حَدَّثَنِي يَحْيَى بْنُ آدَمَ عَنْ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ .
    وأخرجه الترمذي (ت279) في [سننة/3805-الصديق] فقال: حَدَّثَنَا إِبْرَاهِيمُ بْنُ إِسْمَاعِيلَ بْنِ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ قَالَ: حَدَّثَنِي أَبِي، عَنْ أَبِيهِ.
    وأخرجه عبد الله بن أحمد (ت290) في [الفضائل/1/238/294-الرسالة] فقال: حَدَّثَنِي أَبُو الْفَضْلِ الْخُرَاسَانِيّ ُ قثنا سَهْلُ بْنُ عُثْمَانَ الْعَسْكَرِيُّ قثنا يَحْيَى بْنُ أَبِي زَائِدَةَ، عَنْ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ. ومن طريقه أأخرجه الحاكم (ت405) في [المستدرك/3/80/4456-الكتب العلمية].
    وأخرجه الطبراني (ت360) في [الأوسط/7/168/7177-الحرمين] فقال: حَدَّثَنَا مُحَمَّدُ بْنُ أَحْمَدَ الرَّقَّامُ، نَا إِبْرَاهِيمُ بْنُ سَلْمِ بْنِ رُشَيْدٍ الْهُجَيْمِيُّ، ثَنَا عَمْرُو بْنُ زِيَادٍ الْبَاهِلِيُّ، ثَنَا عَبْدُ اللَّهِ بْنُ الْمُبَارَكِ، عَنْ سُفْيَانَ.
    وأخرجه الطبراني (ت360) في [الكبير/9/72/8426-ابن تيمية] فقال: حَدَّثَنَا الْحَسَنُ بْنُ الْعَبَّاسِ الرَّازِيُّ، ثنا سَهْلُ بْنُ عُثْمَانَ، ثنا يَحْيَى بْنُ زَكَرِيَّا بْنِ أَبِي زَائِدَةَ، عَنْ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ، عَنْ أَبِيهِ.
    ح وَحَدَّثَنَا سَلَمَةُ بْنُ إِبْرَاهِيمَ بْنِ إِسْمَاعِيلَ بْنِ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ، حَدَّثَنِي أَبِي، عَنْ أَبِيهِ، عَنْ جَدِّهِ.
    وأخرجه تمام (ت414) في [فوائده/2/276/1732-الرشد] فقال: أَخْبَرَنَا أَبُو يَعْقُوبَ إِسْحَاقُ بْنُ إِبْرَاهِيمَ الْأَذْرَعِيُّ، ثنا عَبْدُ اللَّهِ بْنُ جَعْفَرِ بْنِ أَحْمَدَ أَبُو مُحَمَّدٍ الْعَسْكَرِيُّ، بِالرَّافِقَةِ، ثنا سَهْلُ بْنُ مُحَمَّدٍ الْعَسْكَرِيُّ، ثنا يَحْيَى بْنُ زَكَرِيَّا بْنِ أَبِي زَائِدَةَ، عَنْ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ.
    وأخرجه أبو نعيم (ت430) في [الحلية/1/128-السعادة] فقال: حَدَّثَنَا حَبِيبُ بْنُ الْحَسَنِ، ثَنَا إِبْرَاهِيمُ بْنُ شَرِيكٍ، ثَنَا إِبْرَاهِيمُ بْنُ إِسْمَاعِيلَ، حَدَّثَنِي أَبِي، عَنْ أَبِيهِ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ.
    وأخرجه البغوي (ت516) في [شرح السنة/14/102/3896-المكتب الإسلامي] فقال: أَخْبَرَنَا أَبُو مَنْصُورٍ عَبْدُ الْمَلِكِ، وَأَبُو الْفَتْحِ نَصْرٌ، ابْنَا عَلِيِّ بْنِ أَحْمَدَ بْنِ مَنْصُورٍ بْنِ مُحَمَّدُ بْنُ الْحُسَيْنِ بْنِ شَاذَوَيْهِ الطُّوسِيُّ، بِهَا قَالا: نَا أَبُو الْحَسَنِ مُحَمَّدُ بْنُ أَحْمَدَ بْنِ كَيْسَانَ النَّحْوِيُّ، نَا أَبُو إِسْحَاقَ إِبْرَاهِيمُ بْنُ شَرِيكٍ الأَسَدِيُّ، نَا إِبْرَاهِيمُ بْنُ إِسْمَاعِيلَ هُوَ ابْنُ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ، نَا أَبِي، عَنْ أَبِيهِ.
    قال ثلاثتهم (أبو حنيفة، ويحيى، وسفيان):
    عَنْ سَلَمَةَ بْنِ كُهَيْلٍ، عَنْ أَبِي الزَّعْرَاءِ، عَنْ عَبْدِ اللَّهِ بْنِ مَسْعُودٍ، قَالَ: قَالَ رَسُولُ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ: «اقْتَدُوا بِاللَّذَيْنِ مِنْ بَعْدِي أَبِي بَكْرٍ وَعُمَرَ، وَاهْتَدُوا بِهَدْي عَمَّارٍ، وَتَمَسَّكُوا بِعَهْدِ ابْنِ أُمِّ عَبْدٍ».اهـ
    قال الترمذي: هَذَا حَدِيثٌ غَرِيبٌ مِنْ هَذَا الوَجْهِ مِنْ حَدِيثِ ابْنِ مَسْعُودٍ، لاَ نَعْرِفُهُ إِلاَّ مِنْ حَدِيثِ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ، وَيَحْيَى بْنُ سَلَمَةَ يُضَعَّفُ فِي الحَدِيثِ.اهـ
    قال الطبراني: لَمْ يَرْوَ هَذَا الْحَدِيثَ عَنْ سُفْيَانَ إِلَّا ابْنُ الْمُبَارَكِ، وَلَا عَنِ ابْنِ الْمُبَارَكِ إِلَّا عَمْرُو بْنُ زِيَادٍ الْبَاهِلِيُّ، تَفَرَّدَ بِهِ: إِبْرَاهِيمُ بْنُ سَلْمِ بْنِ رُشَيْدٍ .اهـ
    قال ابن القيسراني: وَيحيى بن سَلمَة مَتْرُوك الحَدِيث، لَيْسَ بِشَيْء.([1]) اهـ
    قال البغوي: هَذَا حَدِيثٌ غَرِيبٌ لَا يُعْرَفُ إِلا مِنْ حَدِيثِ يَحْيَى بْنِ سَلَمَةَ بْنِ كُهَيْلٍ. اهـ
    قال الذهبي: سنده واه.([2]) اهـ

    ([1])(ذخيرة الحفاظ/1/437/594-السلف)

    ([2]) (التلخيص/ 4456).

    أما حديث عبد الله بن عمر

    أخرجه العقيلي (ت322) في [الضعفاء/4/94-المكتبة العلمية] فقال: حَدَّثَنَاهُ أَحْمَدُ بْنُ الْخَلِيلِ الْخُرَيْبِيُّ، حَدَّثَنَا إِبْرَاهِيمُ بْنُ مُحَمَّدٍ الْحَلَبِيُّ، حَدَّثَنِي مُحَمَّدُ بْنُ عَبْدِ اللَّهِ بْنِ عُمَرَ بْنِ الْقَاسِمِ بْنِ عَبْدِ اللَّهِ بْنِ عُبَيْدِ اللَّهِ بْنِ إِبْرَاهِيمَ بْنِ عُمَرَ بْنِ الْخَطَّابِ.
    وأخرجه ابن عساكر (ت571) في [تاريخه/30/228/6350-الفكر] فقال: أخبرنا أبو نصر أحمد بن محمد بن أحمد الإسكندراني وأبو الفتح محمد بن الموفق بن محمد الجرجاني وأبو الفتح محمد بن علي بن نصر الحمادي الأدرقاني وأبو النضر عبد الرحمن بن عبد الجبار بن عثمان العامي وأبو جعفر محمد بن علي بن محمد الطبري وأبو المظفر عبد الفاطر بن عبد الرحيم بن عبد الله السقطي المقرئان بهراة قالوا أنا أبو سهل نجيب بن ميمون بن سهل أنا أبو علي منصور بن عبد الله بن خالد الذهلي الخالدي نا أبو إسحاق إبراهيم بن محمد بن عمروية المروزي الملقب بعبد دليل وأبو سعيد الحسن بن أحمد بن محمد بن المبارك التستري قالا أنا أحمد بن صبيح بن وضاح نا محمد بن قطن نا ذا النون.
    وأخرجه ابن عساكر (ت571) في [تاريخه/30/228/6351-الفكر] أخبرنا أبو بكر وجيه بن طاهر وأبو الفتح محمد بن الموفق بن مبارك بن أبي مطيع الوكيل وعبد الجبار بن أبي سعيد بن أبي القاسم الطبيب وأبو العلاء صاعد بن أبي الفضل بن أبي عثمان الشعيبي الماليني قالوا أنبأ أم الفضل بنت عبد الصمد بن علي بن محمد الهريمية قالت أنبأ أبو محمد بن أبي شريح نا أبو القاسم عبد الرحمن بن الحسن الأسدي زاد وجيه بهمذان نا أبو عبد الرحمن عبد الله بن محمد بن العباس الضبي نا أحمد بن خلاد القطان نا محمد بن عبد الله العمري المدني.
    قَالَ محمد العمري: أَخْبَرَنَا . وقال ذو النون: حدثنا.
    مَالِكٌ، عَنْ نَافِعٍ، عَنِ ابْنِ عُمَرَ قَالَ: قَالَ رَسُولُ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ: «اقْتَدُوا بِاللَّذَيْنِ مِنْ بَعْدِي، أَبِي بَكْرٍ وَعُمَرَ، رَضِيَ اللَّهُ عَنْهُمَا» .
    قال الذهبي في [الميزان/1/105/411-المعرفة] : أحمد بن صليح، عن ذي النون المصري، عن مالك، عن نافع، عن ابن عمر بحديث: اقتدوا باللذين من بعدى. وهذا غلط، وأحمد لا يعتمد عليه.اهـ
    وقال الذهبي في [الميزان/1/142/557-المعرفة] : أحمد بن محمد بن غالب الباهلي، غلام الخليل، وكان من كبار الزهاد ببغداد، قال ابن عدي: سمعت أبا عبد الله النهاوندي يقول: قلت لغلام خليل، ما هذه الرقائق التي تحدث بها؟ قال: وضعناها لنرقق بها قلوب العامة. وقال أبو داود: أخشى أن يكون دجال بغداد. ومن مصائبه، قال: حدثنا محمد بن عببدالله العمري، حدثنا مالك، عن نافع، عن ابن عمر، قال: قال رسول الله صلى الله عليه وسلم: اقتدوا باللذين من بعدى، أبي بكر وعمر.فهذا ملصق بمالك.اهـ
    وقال الذهبي في [الميزان/3/610/7812-المعرفة] : محمد بن عبد الله بن عمر بن القاسم بن عبد الله بن عبيد الله بن عاصم ابن عمر بن الخطاب العدوي العمري، ذكره العقيلي، وقال: لا يصح حديثه، ولا يعرف بنقل الحديث، حدثناه أحمد بن الخليل، حدثنا إبراهيم بن محمد الحلبي، حدثني محمد بن عبد الله ابن عمر بن القاسم، أخبرنا مالك، عن نافع، عن ابن عمر - مرفوعاً: اقتدوا باللذين من بعدى: أبو بكر، وعمر. فهذا لا أصل له من حديث مالك، بل هو معروف من حديث حذيفة بن اليمان. وقال الدارقطني: العمري هذا يحدث عن مالك بأباطيل. وقال ابن مندة: له مناكير.اهـ
    العلم النافع ، وذكر الله الحقيقي ، يُهذب الطبع ، ويحسن الأخلاق (البحر المديد /5/317)

    أما حديث أنس بن مالك

    أخرجه ابن حبان (ت354) في [الثقات/2/190]، وفي [أخبار الخلفا/2/453] فقال: حَدَّثَنَا مُحَمَّدُ بْنُ الْقَاسِمِ الدَّقَّاقُ بِالْمِصِّيصَةِ ثَنَا يُوسُفُ بْنُ سَعِيدِ بْنِ مُسْلِمٍ.
    وأخرجه ابن عساكر (ت571) في [تاريخه/44/233/9771-الفكر] فقال: أخبرنا أبو القاسم زاهر أيضا أنا أبو عثمان البحيري قراءة عليه وأن حاضر أنا أبو بكر أحمد بن الحسن بن أحمد القاضي أنا عبد الله بن محمد بن عدي أنا إسحاق بن إبراهيم بن يونس.
    قالا (يوسف وإسحاق):
    ثَنَا هَارُونُ بْنُ زِيَادٍ الْحِنَّائِيُّ ثَنَا الْحَارِثُ بْنُ عُمَيْرِ عَنْ حُمَيْدٍ عَنْ أَنَسٍ قَالَ: قَالَ النَّبِيُّ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ: «اقْتَدُوا بِاللَّذَيْنِ مِنْ بعدِي أَبى بكر وَعمر».
    الحارث بن عمير أبو عمير     البصري نزيل مكة من الثامنة وثقه الجمهور وفي أحاديثه مناكير ضعفه بسببها الأزدي وابن حبان وغيرهما فلعله تغير حفظه في الآخر [التقريب/1041].
    يرويه عنه هارون بن زياد بن بشير: ذكره ابن حبان في [الثقات/16221].
    ويوسف ثقة حافظ إلا أن السند إليه ضعيف؛ يرويه عنه محمد بن القاسم، وهو مجهول الحال، ولكن تابعه الثقة الحافظ إسحاق بن إبراهيم المنجنيقي، والسند إليه صحيح.
    وأخرجه ابن عدي (ت365) في [الكامل/3/29-المكتبة العلمية] : فقال: حَدَّثَنَا عَلِيُّ بْنُ الْحَسَنِ بْنِ سُلَيْمَانَ، حَدَّثَنا أَحْمَدُ بْنُ مُحَمد بْنِ الْمُعَلَّى الآدَمِيُّ.
    حَدَّثَنَا مُحَمد بْنُ عَبد الْحَمِيدِ الفرغاني، حَدَّثَنا صالح بن حكيم البصري.
    حَدَّثَنَا مُحَمد بْنُ سَعِيد الْحَرَّانِيُّ، حَدَّثَنا جَعْفَرُ بْنُ مُحَمد بْنِ الصباح.
    وأخرجه ابن عساكر (ت571) في [تاريخه/33/119-الفكر] فقال: أخبرناه أبو القاسم بن السمرقندي أنا أبو القاسم بن مسعدة أنا حمزة بن يوسف أنا أبو أحمد بن عدي نا علي بن الحسن بن سليمان نا أحمد بن محمد بن المعلى الأدمي.
    قال ثلاثتهم (أحمد الآدمي، وصالح البصري، وجعفر بن محمد):
    حَدَّثَنا مُسْلِمُ بْنُ صَالِحٍ أَبُو رَجَاءٍ، حَدَّثَنا حَمَّادُ بْنُ دَلِيلٍ، عَنْ عُمَر بْنِ نَافِعٍ، عَنْ عَمْرو بْنِ هَرِمٍ قَالَ: دَخَلْتُ أَنَا وَجَابِرُ بْنُ زَيْدٍ عَلَى أَنَسِ بْنِ مَالِكٍ فَقَالَ قَال رَسُول اللهِ صَلَّى اللَّهُ عَلَيهِ وَسلَّمَ: «اقْتَدُوا بِاللَّذِينَ مِنْ بَعْدِي أَبُو بَكْرٍ وعمر وتمسكوا بعهد بن أُمِّ عَبد وَاهْتَدُوا بِهَدْيِ عَمَّارٍ».اهـ
    قال ابنُ عَدِي: وحماد بن دليل هذا قليل الرواية. اهـ إشارة إلى أنه لم يكن بصاحب حديث كما قال عنه الإمام أحمد، كما في [تاريخ بغداد/8/152]. وقد ضعفه الأزدي كما في [الضعفاء لابن الجوزي/991].
    يرويه عنه مسلم بن صالح: وهو مجهول الحال.

    أما حديث حذيفة بن اليمان

    ففيه خمسة أوجه
    الوجه الأول: وهو المحفوظ، عن الثوري، عن عَبْدِ الْمَلِكِ، عَنْ مَوْلًى لِرِبْعِيِّ، عَنْ رِبْعِيِّ، عَنْ حُذَيْفَةَ، وهو الوجه المحفوظ عن عبد الملك، وهذا الوجه المحفوظ عن عبد الملك ضعيف؛ لجهالة هلال مولى ربعي بن حراش ([1])، وعبد الملك ثقة حافظ من رجال الصحيحين ([2]).


    أخرجه ابن أبي شيبة (ت235) في [المصنف/6/350/31942-رشد] فقال: حَدَّثَنَا وَكِيعٌ.
    وأخرجه ابن أبي شيبة (ت235) في [المصنف/7/433/37049-رشد] فقال: حَدَّثَنَا وَكِيعٌ.
    وأخرجه أحمد (ت241) في [مسنده/38/309/23276-الأرناؤوط] فقال: حَدَّثَنَا وَكِيعٌ.
    وأخرجه أحمد (ت241) في [مسنده/38/418/23419-الأرناؤوط] فقال: حَدَّثَنَا وَكِيعٌ.
    وأخرجه أحمد (ت241) في [الفضائل/1/332/478-الرسالة] فقال: قثنا وَكِيعٌ.
    وأخرجه ابن ماجة (ت273) في [سننة/97-الصديق] فقال: حَدَّثَنَا عَلِيُّ بْنُ مُحَمَّدٍ قَالَ: حَدَّثَنَا وَكِيعٌ، ح وَحَدَّثَنَا مُحَمَّدُ بْنُ بَشَّارٍ قَالَ: حَدَّثَنَا مُؤَمَّلٌ.
    وأخرجه يعقوب الفسوي (ت277) في [المعرفة/1/480-الرسالة] فقال: حَدَّثَنَا أَبُو عَاصِمٍ الضَّحَّاكُ بْنُ مَخْلَدٍ وقبيصة.
    وأخرجه الترمذي (ت279) في [سننة/3799-الصديق] فقال: حَدَّثَنَا مَحْمُودُ بْنُ غَيْلَانَ قَالَ: حَدَّثَنَا وَكِيعٌ.
    وأخرجه ابن أبي خيثمة (ت279) في [تاريخه/3/170/4317-الفاروق] حَدَّثَنا مُصْعَب بْنُ عَبْد الله وابن كاسب، قالا: حدثنا إِبْرَاهِيمُ بْنُ سَعْد.
    وأخرجه البلاذري (ت279) في [أنساب الأشراف/10/57-الفكر] فقال: وَحَدَّثَنِي عَبْدُ اللَّهِ بْنُ صَالِحٍ الْمُقْرِئُ، حَدَّثَنِي إِبْرَاهِيمُ بْنُ سَعْدٍ الزُّهْرِيُّ.
    وأخرجه ابن ابي عاصم (ت287) في [السنة/2/545/1148-الألباني] فقال: حَدَّثَنَا أَبُو بَكْرٍ، ثنا وَكِيعٌ.
    وأخرجه ابن ابي عاصم (ت287) في [السنة/2/545/1149-الألباني] فقال: ثنا يَعْقُوبُ بْنُ حُمَيْدٍ، ثنا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه ابن ابي عاصم (ت287) في [السنة/2/617/1422-الألباني] فقال: ثنا أَبُو بَكْرٍ، ثنا وَكِيعٌ.
    وأخرجه ابن ابي عاصم (ت287) في [السنة/2/617/1423-الألباني] فقال: ثنا يَعْقُوبُ، ثنا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه عبد الله بن أحمد (ت290) في [السنة/2/579/1367-ابن القيم] فقال: حَدَّثَنَا مُصْعَبُ بْنُ عَبْدِ اللَّهِ الزُّبَيْرِيُّ، وَحَدَّثَنَا يَعْقُوبُ بْنُ حُمَيْدِ بْنِ كَاسِبٍ، بِمَكَّةَ قَالَا: حَدَّثَنَا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه عبد الله بن أحمد (ت290) في [السنة/2/580/1369-ابن القيم]فقال: حَدَّثَنِي زُهَيْرُ بْنُ حَرْبٍ، نا قَبِيصَةُ.
    وأخرجه البزار (ت292) في [مسنده/7/250/2828-العلوم والحكم] فقال: أَخْبَرَنَا بِهِ أَحْمَدُ بْنُ الْوَلِيدِ الْكَرْخِيُّ، قَالَ: أَخْبَرَنَا عَبْدُ الْعَزِيزِ بْنُ عَبْدِ اللَّهِ الْأُوَيْسِيُّ، قَالَ: أَخْبَرَنَا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه البزار (ت292) في [مسنده/7/250/2819-العلوم والحكم] فقال: أَخْبَرَنَا بِهِ عَمْرُو بْنُ عَبْدِ اللهِ الْأَوْدِيُّ، قَالَ: أَخْبَرَنَا وَكِيعٌ.
    وأخرجه الخلال (311) في [السنة/1/275/336-الراية] فقال: أَخْبَرَنَا مُحَمَّدٌ، قَالَ: ثَنَا وَكِيعٌ.
    وأخرجه الطحاوي (ت321) في [بيان مشكل الآثار/3/256/1224-الرسالة] فقال: حَدَّثَنَا عَبْدُ الْمَلِكِ بْنُ مَرْوَانَ الرَّقِّيُّ، قَالَ: حَدَّثَنَا الْفِرْيَابِيُّ .
    وأخرجه الطحاوي (ت321) في [بيان مشكل الآثار/3/257/1230-الرسالة] فقال: حَدَّثَنَا عَلِيُّ بْنُ عَبْدِ الرَّحْمَنِ، قَالَ: حَدَّثَنَا مُصْعَبُ بْنُ عَبْدِ اللهِ الزُّبَيْرِيُّ، قَالَ: حَدَّثَنَا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه الآجري (ت360) في [الشريعة/4/1871/1341-الوطن] فقال: حَدَّثَنَا أَبُو بَكْرٍ قَاسِمُ بْنُ زَكَرِيَّا الْمُطَرِّزُ قَالَ: حَدَّثَنَا بُنْدَارٌ مُحَمَّدُ بْنُ بَشَّارٍ قَالَ: حَدَّثَنَا مُؤَمَّلُ بْنُ إِسْمَاعِيلَ , قَالَ الْمُطَرِّزُ: وَحَدَّثَنَا عَمْرُو بْنُ عَلِيٍّ قَالَ: حَدَّثَنَا أَبُو عَامِرٍ.

    وأخرجه القطيعي (ت368) في [جزء الألف دينار/253/168-النفائس] فقال: حَدَّثَنَا إِدْرِيسُ قَالَ: حَدَّثَنَا مُصْعَبُ بْنُ عَبْدِ اللَّهِ الزُّبَيْرِيُّ قَالَ: حَدَّثَنَا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه ابن شاهين (ت385) في [شرح مذاهب أهل السنة/219/147-قرطبة] فقال: حَدَّثَنَا عَبْدُ اللَّهِ بْنُ مُحَمَّدٍ، ثنا مُصْعَبُ بْنُ عَبْدِ اللَّهِ الزُّبَيْرِيُّ، ثنا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه الحاكم (ت405) في [المستدرك/3/79/4454-الكتب العلمية] فقال: فَحَدَّثَنَا أَبُو بَكْرِ بْنُ إِسْحَاقَ، وَعَلِيُّ بْنُ حَمْشَاذٍ، قَالَا: ثنا بِشْرُ بْنُ مُوسَى، ثنا الْحُمَيْدِيُّ.
    وأخرجه اللالكائي (ت418) في [شرح اصول الاعتقاد/7/1395/2498-طيبة] فقال: أنا عَبْدُ اللَّهِ بْنُ مُحَمَّدِ بْنِ عَلِيِّ بْنِ زِيَادٍ النَّيْسَابُورِ يُّ، قَالَ: أنا مَكِّيُّ بْنُ عَبْدَانَ، قَالَ: نا عَبْدُ اللَّهِ بْنُ هَاشِمِ بْنِ حَيَّانَ الْعَبْدِيُّ الطُّوسِيُّ، قَالَ: نا أَبُو دَاوُدَ الْحَفْرِيُّ.
    وأخرجه أبو نعيم (ت430) في [الإمامة/1/253/48-العلوم والحكم] فقال: حَدَّثَنَا عَبْدُ اللَّهِ بْنُ الْحُسَيْنِ بْنِ بُنْدَارٍ، ثَنَا مُحَمَّدُ بْنُ إِسْمَاعِيلَ، ثَنَا قَبِيصَةُ بْنُ عُقْبَةَ.
    وأخرجه البيهقي (ت458) في [المدخل/122/61-الخلفا] فقال: أَخْبَرَنَا أَبُو عَبْدِ اللَّهِ الْحَافِظُ وَأَبُو الْقَاسِمِ الْحَسَنُ بْنُ مُحَمَّدِ بْنِ حُبَيْبٍ الْمُفَسِّرُ، مِنْ أَصْلِهِ وَأَبُو نَصْرٍ أَحْمَدُ بْنُ عَلِيٍّ الفَّامِيُّ قَالُوا: ثنا أَبُو الْعَبَّاسِ مُحَمَّدُ بْنُ يَعْقُوبَ , ثنا الْحَسَنُ بْنُ مُكْرَمٍ، ثنا أَبُو عَاصِمٍ النَّبِيلُ.
    وأخرجه البيهقي (ت458) في [المدخل/122/61-الخلفا] أَخْبَرَنَا أَبُو عَبْدِ اللَّهِ الْحَافِظُ، ثنا أَبُو عَبْدِ اللَّهِ مُحَمَّدُ بْنُ عَبْدِ اللَّهِ الصَّفَّارُ , ثنا أَبُو إِسْمَاعِيلَ التِّرْمِذِيُّ، ثنا عَبْدُ الْعَزِيزِ بْنُ عَبْدِ اللَّهِ الْأُوَيْسِيُّ، ثنا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه البيهقي (ت458) في [الكبرى/16590-الحديث] فقال: أَخْبَرَنَا أَبُو الْحُسَيْنِ بْنُ بِشْرَانَ، بِبَغْدَادَ , أَنْبَأَ أَبُو جَعْفَرٍ مُحَمَّدُ بْن عَمْرٍو الرُّزَازُ , ثَنَا يَحْيَى بْنُ جَعْفَرٍ، أَنْبَأَ الضَّحَّاكُ.
    وأخرجه البيهقي (ت458) في [الكبرى/16591-الحديث] فقال: وَأَخْبَرَنَا أَبُو الْحُسَيْنِ بْنُ الْفَضْلِ، أَنْبَأَ عَبْدُ اللهِ بْنُ جَعْفَرٍ، ثَنَا يَعْقُوبُ بْنُ سُفْيَانَ، حَدَّثَنِي عَبْدُ الْعَزِيزِ بْنُ عَبْدِ اللهِ، ثَنَا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه البيهقي (ت458) في [الاعتقاد/340- الآفاق] فقال: أَخْبَرَنَا أَبُو الْحُسَيْنِ بْنُ الْفَضْلِ الْقَطَّانُ، أنا عَبْدُ اللَّهِ بْنُ جَعْفَرٍ، حَدَّثَنَا يَعْقُوبُ بْنُ سُفْيَانِ، ثنا أَبُو عَاصِمٍ الضَّحَّاكِ بْنُ مَخْلَدٍ وَقَبِيصَةُ.
    وأخرجه البيهقي (ت458) في [الاعتقاد/340- الآفاق] فقال: أَخْبَرَنَا أَبُو الْحُسَيْنِ بْنُ الْفَضْلِ الْقَطَّانُ، أنا عَبْدُ اللَّهِ بْنُ جَعْفَرٍ، حَدَّثَنَا يَعْقُوبُ بْنُ سُفْيَانِ، ثنا أَبُو عَاصِمٍ الضَّحَّاكِ بْنُ مَخْلَدٍ وَقَبِيصَةُ.
    وأخرجه ابن عبد البر (ت463) في [الجامع/2/1166/2308-ابن الجوزي] فقال: وَحَدَّثَنَا سَعِيدُ بْنُ نَصْرٍ، ثنا قَاسِمٌ، ثنا مُحَمَّدُ بْنُ إِسْمَاعِيلَ، ثنا الْحُمَيْدِيُّ.
    وأخرجه ابن عبد البر (ت463) في [الجامع/2/1167/2309-ابن الجوزي] فقال: أَخْبَرَنَا خَلَفُ بْنُ الْقَاسِمِ، ثنا أَبُو طَالِبٍ مُحَمَّدُ بْنُ زَكَرِيَّا بِبَيْتِ الْمَقْدِسِ ثنا أَبُو عِمْرَانَ مُوسَى بْنُ نَصْرٍ الْبَغْدَادِيُّ ثنا مُصْعَبُ بْنُ عَبْدِ اللَّهِ الزُّبَيْرِيُّ ثنا إِبْرَاهِيمُ بْنُ سَعْدٍ.
    وأخرجه الخطيب (ت463) في [الفقيه والمتفقه/2/290-ابن الجوزي] فقال: أنا الْحَسَنُ بْنُ عَلِيٍّ التَّمِيمِيُّ , أنا أَحْمَدُ بْنُ جَعْفَرِ بْنِ حَمْدَانَ , نا عَبْدُ اللَّهِ بْنُ أَحْمَدَ , حَدَّثَنِي أَبِي , نا وَكِيعٌ.
    وأخرجه ابن عساكر (ت571) في [تاريخه/5/14-الفكر] فقال: أخبرنا أبو القاسم بن السوسي أنا أبو القاسم بن أبي العلاء أنا أبو نصر المري نا أبو سعيد أحمد بن عثمان بن البقال البغداذي الفقيه من حفظه نا عبد الله بن محمد البغوي ببغداد نا مصعب بن عبد الله الزبيري نا إبراهيم بن سعد.
    قال ثمانيتهم (وكيع، ومؤمل، والحميدي، وأبو داود، وأبو عاصم الضحاك، وقبيصة، وأبو عامر عبد الملك بن عمرو القيسي، وإبراهيم):
    حدثنا سُفْيَانَ , عَنْ عَبْدِ الْمَلِكِ بْنِ عُمَيْرٍ (    [3]), عَنْ هِلَالٍ مَوْلًى لِرِبْعِيٍّ , عَنْ رِبْعِيٍّ , عَنْ حُذَيْفَةَ , قَالَ: كُنَّا عِنْدَ النَّبِيِّ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ , فَقَالَ: «إِنِّي لَا أَدْرِي قَدْرَ بَقَائِي فِيكُمْ , فَاقْتُدُوا بِاللَّذَيْنِ مِنْ بَعْدِي وَأَشَارَ إِلَى أَبِي بَكْرٍ وَعُمَرَ وَتَمَسَّكُوا بِعَهْدِ عَمَّارٍ , وَمَا حَدَّثَكُمُ ابْنُ مَسْعُودٍ فَصَدِّقُوهُ».ا ـ
    قال الطبراني:     لَمْ يَقُلْ فِي هَذَا الْحَدِيثِ: عَنْ سُفْيَانَ الثَّوْرِيِّ، عَنْ عَبْدِ الْمَلِكِ بْنِ عُمَيْرٍ: «عَنْ هِلَالٍ مَوْلَى رِبْعِيِّ، إِلَّا إِبْرَاهِيمُ بْنُ سَعْدٍ».اهـ
    قلت: يعني: أنه كان يسمي المولى هلالاً، وقد تابعه عليه الحميدي، والسند إليه صحيح.
    قال الترمذي (ت279) في [العلل/371/689-عالم الكتب]: وَقَالَ الثَّوْرِيُّ: عَنْ عَبْدِ الْمَلِكِ , عَنْ مَوْلًى لِرِبْعِيٍّ , عَنْ رِبْعِيٍّ , عَنْ حُذَيْفَةَ قَالَ: قَالَ النَّبِيُّ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ وَهُوَ الصَّحِيحُ. اهـ
    وقال ابن أبي حاتم (ت327) في [العلل/6/444/2655-الحميضي] : وسألتُ أَبِي عَنْ حديثٍ :
    رَوَاهُ إِبْرَاهِيمُ بْنُ سعد، عن الثَّوري، عن عبد الملك بْنِ عُمَير، عَنْ هلالٍ مَوْلَى رِبْعِيٍّ، عن رِبْعِيٍّ، عَنْ حُذَيفة، عَنِ النبيِّ –صلى الله عليه وسلم- قَالَ: اقْتَدُوا بِاللَّذَيْنِ مِنْ بَعْدِي.
    ورواه زائدة وغيرُه، عن عبد الملك، عَنْ رِبْعِيٍّ، عَنْ حُذَيفة، عَنِ النبيِّ –صلى الله عليه وسلم. قلتُ: فأيُّهما أصَحُّ؟ قَالَ أَبِي: حدَّثنا ابْنُ كَثِيرٍ، عن الثَّوري، عن عبد الملك بْنِ عُمَير، عَنْ مَوْلًى لرِبْعِيٍّ، عَنْ رِبْعِيٍّ، عَنْ حُذَيفة. قلتُ: فأيُّهما أصَحُّ؟
    قَالَ:
    مَا قَالَ الثَّوري؛ زَادَ رَجُلا وجَوَّد الحديثَ، فأمَّا إبراهيمُ بنُ سَعْدٍ فسَمَّى الرجلَ، وأمَّا ابنُ كَثِيرٍ فَلَمْ يُسَمِّ الْمَوْلَى.اهـ
    قال الخليلي (ت446) في [الإرشاد/1/378] : حديث ابن عيينة، عن عبد الملك بْنِ عُمَيْرٍ، عَنْ رِبْعِيٍّ، عَنْ حذيفة، عن النبيِّ -صلى الله عليه وسلم-: " اقتدوا باللَّذَين من بعدي": رواه عنه الأئمة: الشافعي وغيره، يقال: سمعه من زائدة، عن عبد الملك، والحديث صحيحٌ معلول؛ لأن في بعض الروايات: عن عبد الملك، عَنْ مَوْلًى لِرِبْعِيٍّ، عَنْ رِبْعِيٍّ. اهـ.
    قال ابن عبد البر (ت463) في [الجامع/2/1166/2308-ابن الجوزي]: رَوَاهُ جَمَاعَةٌ عَنِ ابْنِ عُيَيْنَةَ عَنْ عَبْدِ الْمَلِكِ بْنِ عُمَيْرٍ عَنْ رِبْعِيٍّ عَنْ حُذَيْفَةَ هَكَذَا لَمْ يَذْكُرُوا مَوْلَى رِبْعِيٍّ، وَالصَّحِيحُ مَا ذَكَرْنَا مِنْ رِوَايَةِ الْحُمَيْدِيِّ عَنْهُ وَكَذَلِكَ رِوَايَةُ الثَّوْرِيُّ وَهُوَ أَحْفَظُ وَأَتْقَنُ عِنْدَهُمْ.اهـ
    قال الذهبي (ت748) في [السير/1/478-الرسالة] : الثَّوْرِيُّ: عَنْ عَبْدِ المَلِكِ بنِ عُمَيْرٍ، عَنْ مَوْلَىً لِرِبْعِيٍّ، عَنْ رِبْعِيٍّ، عَنْ حُذَيْفَةَ... فذكره، ثم قال: رَوَاهُ جَمَاعَةٌ هَكَذَا عَنْهُ.
    وَرَوَاهُ: أَسْبَاطٌ، عَنِ الثَّوْرِيِّ، فَأَسْقَطَ مِنْهُ مَوْلَى رِبْعِيٍّ.
    وَرَوَاهُ: مِسْعَرٌ، عَنْ عَبْدِ المَلِكِ بنِ عُمَيْرٍ، عَنْ رِبْعِيٍّ.
    وَرَوَاهُ: سَالِمٌ المُرَادِيُّ، عَنْ عَمْرِو بنِ هَرِمٍ، عَنْ رِبْعِيٍّ، عَنْ حُذَيْفَةَ.
    وَقَالَ: وَكِيْعٌ، عَنْ سَالِمٍ المُرَادِيِّ، فَقَالَ: عَنْ عَمْرِو بنِ مُرَّةَ.
    وَالأَوَّلُ أَشْبَهُ.اهـ

    مَا قَالَ الثَّوري؛ زَادَ رَجُلا وجَوَّد الحديثَ، فأمَّا إبراهيمُ بنُ سَعْدٍ فسَمَّى الرجلَ، وأمَّا ابنُ كَثِيرٍ فَلَمْ يُسَمِّ الْمَوْلَى.اهـ
    قال ابن أبي حاتم (ت327) في [العلل/6/435/2648-الحميضي] قَالَ أَبِي: كَانَ يحدِّث بِهِ أيامَ الموسم عن عبد الملك بْنِ عُمَير، وَلَمْ يَذْكُرْ زَائِدَةَ، ثُمَّ قَالَ: لَمْ آخذُه مِنْ عبد الملك؛ إنما حدَّثَناه زائدة، عن عبد الملك. وَقَالَ سُفْيَانُ: إِذَا ذكرتُ لَهُمْ زائدة لم يسألوني عَنْهُ، وَهَذَا حديثٌ فِيهِ فَضِيلَةٌ للشَّيخين.اهـ
    قال اللالكائي (ت418) في [شرح اصول الاعتقاد/7/1395/2500-طيبة] قَالَ أَبُو حَامِدٍ: وثنا بِهِ عَبْدُ الرَّحْمَنِ مَرَّةً أُخْرَى، عَنْ سُفْيَانَ، عَنْ عَبْدِ الْمَلِكِ، وَلَمْ يَذْكُرْ زَائِدَةَ فِي حَدِيثِ الثَّوْرِيِّ.اه
    قال الخليلي (ت446) في [الإرشاد/1/378] : حديث ابن عيينة، عن عبد الملك بْنِ عُمَيْرٍ، عَنْ رِبْعِيٍّ، عَنْ حذيفة، عن النبيِّ -صلى الله عليه وسلم-: " اقتدوا باللَّذَين من بعدي": رواه عنه الأئمة: الشافعي وغيره، يقال: سمعه من زائدة، عن عبد الملك، والحديث صحيحٌ معلول؛ لأن في بعض الروايات: عن عبد الملك، عَنْ مَوْلًى لِرِبْعِيٍّ، عَنْ رِبْعِيٍّ. اهـ.
    قال ابن عبد البر (ت463) في [الجامع/2/1166/2308-ابن الجوزي]: رَوَاهُ جَمَاعَةٌ عَنِ ابْنِ عُيَيْنَةَ عَنْ عَبْدِ الْمَلِكِ بْنِ عُمَيْرٍ عَنْ رِبْعِيٍّ عَنْ حُذَيْفَةَ هَكَذَا لَمْ يَذْكُرُوا مَوْلَى رِبْعِيٍّ، وَالصَّحِيحُ مَا ذَكَرْنَا مِنْ رِوَايَةِ الْحُمَيْدِيِّ عَنْهُ وَكَذَلِكَ رِوَايَةُ الثَّوْرِيُّ وَهُوَ أَحْفَظُ وَأَتْقَنُ عِنْدَهُمْ .اهـ
    قال الذهبي (ت748) في [السير/راشدون/79-الرسالة] : وقال ابن عيينة، عن زائدة، عن عبد الملك بن عميرة، عن ربعي، عن حذيفة قال: قال رسول الله صلى الله عليه وسلم: «اقتدوا باللذين من بعدي أبي بكر وعمر». وكذا رواه سفيان بن حسين الواسطي عن عبد الملك. وكان سفيان ربما دلسه وأسقط منه زائدة. ورواه سفيان الثوري، عن عبد الملك، عن هلال مولى ربعي، عن ربعي.اهـ
    قال الذهبي (ت748) في [السير/1/414-الرسالة] : رَوَاهُ طَائِفَةٌ، عَنِ الثَّوْرِيِّ، بِإِسْقَاطِ مَوْلَى رِبْعِيٍّ. وَكَذَا رَوَاهُ: زَائِدَةُ، وَغَيْرُهُ، عَنْ عَبْدِ المَلِكِ. وَرُوِيَ عَنْ: عَمْرِو بنِ هَرِمٍ، عَنْ رِبْعِيٍّ، عَنْ حُذَيْفَةَ.اهـ
    قال الذهبي (ت748) في [السير/1/478-الرسالة] : الثَّوْرِيُّ: عَنْ عَبْدِ المَلِكِ بنِ عُمَيْرٍ، عَنْ مَوْلَىً لِرِبْعِيٍّ، عَنْ رِبْعِيٍّ، عَنْ حُذَيْفَةَ...فذك ه ثم قال: رَوَاهُ جَمَاعَةٌ هَكَذَا عَنْهُ.
    وَرَوَاهُ: أَسْبَاطٌ، عَنِ الثَّوْرِيِّ، فَأَسْقَطَ مِنْهُ مَوْلَى رِبْعِيٍّ.
    وَرَوَاهُ: مِسْعَرٌ، عَنْ عَبْدِ المَلِكِ بنِ عُمَيْرٍ، عَنْ رِبْعِيٍّ.
    وَرَوَاهُ: سَالِمٌ المُرَادِيُّ، عَنْ عَمْرِو بنِ هَرِمٍ، عَنْ رِبْعِيٍّ، عَنْ حُذَيْفَةَ.
    وَقَالَ وَكِيْعٌ، عَنْ سَالِمٍ المُرَادِيِّ، فَقَالَ: عَنْ عَمْرِو بنِ مُرَّةَ.
    وَالأَوَّلُ أَشْبَهُ.
    قال العلائي (ت761) في [جامع التحصيل/250-عالم الكتب]: سفيان بن عيينة الإمام المشهور مكثر من التدليس لكن عن الثقات كما تقدم .. ومن تدليسه ما رواه عن عبد الملك بن عمير عن ربعي عن حذيفة رضي الله عنه حديث اقتدوا باللذين من بعدي وإنما سمعه من زائدة عن عبد الملك كما جاء عنه في رواية.اهـ

    الوجه الثالث: عن سالم المرادي، عَنْ عَمْرِو بْنِ هَرِمٍ , عَنْ رِبْعِيِّ وَأَبِي عَبْدِ اللَّهِ رَجُلٍ مِنْ أَصْحَابِ حُذَيْفَةَ , عَنْ حُذَيْفَةَ، وهو وجه منكر غير محفوظ عن ربعي، والسند إليه ضعيف.

    أخرجه ابن أبي شيبة (ت235) في [المصنف/7/433/37050-رشد] فقال: حَدَّثَنَا وَكِيعٌ.
    وأخرجه أحمد (ت241) في [مسنده/38/399/23386-الأرناؤوط] فقال: حَدَّثَنَا مُحَمَّدُ بْنُ عُبَيْدٍ.
    وأخرجه أحمد (ت241) في [الفضائل/1/332/479-الرسالة] فقال: قثنا مُحَمَّدُ بْنُ عُبَيْدٍ هُوَ الطَّنَافِسِيُّ .
    وأخرجه الترمذي (ت279) في [سننة/3663-الصديق] فقال: حَدَّثَنَا سَعِيدُ بْنُ يَحْيَى بْنُ سَعِيدٍ الأُمَوِيُّ قَالَ: حَدَّثَنَا وَكِيعٌ.
    أخرجه عبد الله بن أحمد (ت290) في [الفضائل/1/186/198-الرسالة] فقال: قثنا محمد بن الصباح البزاز قثنا إسماعيل بن زكريا، وهو الخلقاني.
    وأخرجه الخلال (311) في [السنة/1/274/335-الراية] فقال: أَخْبَرَنَا مُحَمَّدٌ، قَالَ: ثَنَا وَكِيعٌ.
    وأخرجه الطحاوي (ت321) في [بيان مشكل الآثار/3/259/1233-الرسالة] فقال: وَحَدَّثَنَا يُونُسُ بْنُ عَبْدِ الْأَعْلَى، وَالرَّبِيعُ بْنُ سُلَيْمَانَ، جَمِيعًا قَالَا: حَدَّثَنَا يَحْيَى بْنُ حَسَّانَ، قَالَ: حَدَّثَنَا إِسْمَاعِيلُ بْنُ زَكَرِيَّا.
    وأخرجه العقيلي (ت322) في [الضعفاء/2/150/651-المكتبة العلمية] حَدَّثَنَاهُ مُحَمَّدُ بْنُ إِسْمَاعِيلَ قَالَ: حَدَّثَنَا مُحَمَّدُ بْنُ فُضَيْلٍ قَالَ: حَدَّثَنَا وَكِيعٌ.
    وأخرجه ابن حبان (ت354) في [صحيحه/15/327/6902-الرسالة] فقال: أَخْبَرَنَا أَحْمَدُ بْنُ عَلِيِّ بْنِ الْمُثَنَّى، قَالَ: حَدَّثَنَا أَبُو بَكْرِ بْنُ أَبِي شَيْبَةَ، حَدَّثَنَا وَكِيعٌ.
    قال ثلاثتهم (وكيع، ومحمد بن عبيد الطنافسي، وإسماعيل بن زكريا الخلقاني):
    حَدَّثَنَا سَالِمٌ الْمُرَادِيُّ، عَنْ عَمْرِو بْنِ هَرِمٍ الْأَزْدِيِّ، عَنْ أَبِي عَبْدِ اللَّهِ، ورِبْعِيِّ بْنِ حِرَاشٍ، عَنْ حُذَيْفَةَ قَالَ: بَيْنَا نَحْنُ عِنْدَ رَسُولِ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ قَالَ: «إِنِّي لَسْتُ أَدْرِي مَا قَدْرُ بَقَائِي فِيكُمْ، فَاقْتَدُوا بِاللَّذَيْنِ مِنْ بَعْدِي ـ يُشِيرُ إِلَى أَبِي بَكْرٍ وَعُمَرَ ـ وَاهْدُوا هَدْيَ عَمَّارٍ وَعَهْدَ ابْنِ أُمِّ عَبْدٍ».اهـ
    قَالَ أَبُو جَعْفَرٍ الطحاوي:     سَالِمٌ أَبُو الْعَلَاءِ هَذَا هُوَ رَجُلٌ مِنْ أَهْلِ الْكُوفَةِ يُقَالُ لَهُ: الْأَنْعُمِيُّ، وَهُوَ ثِقَةٌ مَقْبُولُ الرِّوَايَةِ، وَقَدْ رَوَى عَنْهُ أَبُو نُعَيْمٍ، وَقَالَ: هُوَ سَالِمُ بْنُ الْعَلَاءِ أَبُو الْعَلَاءِ.اهـ
    قال العقيلي (ت322) في [الضعفاء/2/150/651-المكتبة العلمية] سَالِمٌ أَبُو الْعَلَاءِ الْمُرَادِيُّ، كُوفِيٌّ حَدَّثَنَا مُحَمَّدُ بْنُ عِيسَى قَالَ: حَدَّثَنَا عَبَّاسٌ قَالَ: سَمِعْتُ يَحْيَى بْنَ مَعِينٍ قَالَ: سَالِمٌ أَبُو الْعَلَاءِ ضَعِيفٌ.اهـ
    قال الذهبي (ت748) في [التاريخ/2/138-بشار] : وَرَوَاهُ سَالِمٌ أَبُو الْعَلاءِ - وَهُوَ ضَعِيفٌ - عَنْ عَمْرِو بْنِ هَرِمٍ، عَنْ رِبْعِيٍّ.اهـ
    قال الذهبي (ت748) في [السير/1/478-الرسالة] الثَّوْرِيُّ: عَنْ عَبْدِ المَلِكِ بنِ عُمَيْرٍ، عَنْ مَوْلَىً لِرِبْعِيٍّ، عَنْ رِبْعِيٍّ، عَنْ حُذَيْفَةَ..فذكر ه ثم قال: رَوَاهُ جَمَاعَةٌ هَكَذَا عَنْهُ...ثم ذكر اختلاف الأوجه كما تقدَّم في الوجهين الأولين، ثم قال: وَرَوَاهُ: سَالِمٌ المُرَادِيُّ، عَنْ عَمْرِو بنِ هَرِمٍ، عَنْ رِبْعِيٍّ، عَنْ حُذَيْفَةَ. وَقَالَ: وَكِيْعٌ، عَنْ سَالِمٍ المُرَادِيِّ، فَقَالَ: عَنْ عَمْرِو بنِ مُرَّةَ، وَالأَوَّلُ أَشْبَهُ.اهـ


    الوجه الرابع: عن عَبْدِ الْمَلِكِ بْنُ عُمَيْرٍ، عَنْ مُنْذِرٍ، عَنْ رِبْعِيٍّ، عَنْ حُذَيْفَةَ، ولا شك أن المنذر هنا خطأ ، لا أدري ممن يكون، فقد تقدَّم المحفوظ عن السفيانين في هذا الحديث.


    أخرجه أحمد (ت241) في [الفضائل/1/359/526-الرسالة] فقال: حَدَّثَنَا عَبْدُ اللَّهِ بْنُ الصَّقْرِ السُّكَّرِيُّ قَالَ نا مُحَمَّدُ بْنُ مُصَفَّى قَالَ نا مُؤَمَّلُ بْنُ إِسْمَاعِيلَ قثنا سُفْيَانُ قَالَ نا عَبْدُ الْمَلِكِ بْنُ عُمَيْرٍ، عَنْ مُنْذِرٍ، عَنْ رِبْعِيٍّ، عَنْ حُذَيْفَةَ قَالَ: قَالَ رَسُولُ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ: «اقْتَدُوا بِاللَّذَيْنِ مِنْ بَعْدِي أَبِي بَكْرٍ، وَعُمَرَ» .
    الوجه الخامس: عَنْ هِلَالٍ مَوْلَى الرَّبْعِيِّ، عَنْ حُذَيْفَةَ، هكذا بإسقاط الربعي بينهما، وهو بدون شك: إما أن يكون اختصاراً للوجه الأول، كما عند البزار، وأبي نعيم، أو الثاني كما عند أبي نعيم، والبيهقي، أو سقط كما أشار ابن عساكر.


    قال البزار: وَرَوَاهُ الثَّوْرِيُّ، عَنْ عَبْدِ الْمَلِكِ، عَنْ مَوْلًى لِرِبْعِيٍّ، عَنْ حُذَيْفَةَ، وَسَمَّى مَوْلَى رِبْعِيٍّ: إِبْرَاهِيمَ بْنَ سَعْدٍ، عَنِ الثَّوْرِيِّ.([5]) اهـ
    أخرجه أبو نعيم (ت430) في [الحلية/9/109-السعادة] فقال: حَدَّثَنَا مُحَمَّدُ بْنُ عَبْدِ الرَّحْمَنِ بْنِ سَهْلٍ قَالَ: حَدَّثَنِي أَبُو عَلِيٍّ حَسَّانُ بْنُ أَبَانَ بْنِ عُثْمَانَ الْقَاضِي بِمِصْرَ , حَدَّثَنِي أَبُو أَحْمَدَ جَامِعُ بْنُ الْقَاسِمِ , ثنا أَبُو بَكْرٍ الْمُسْتَمْلِي مُحَمَّدُ بْنُ يَزِيدَ بْنِ حَكِيمٍ قَالَ: رَأَيْتُ مُحَمَّدَ بْنَ إِدْرِيسَ الشَّافِعِيَّ فِي الْمَسْجِدِ الْحَرَامِ، وَقَدْ جُعِلَتْ لَهُ طَنَافِسُ يَجْلِسُ عَلَيْهَا، فَآتَاهُ رَجُلٌ مِنْ أَهْلِ خُرَاسَانَ فَقَالَ: يَا أَبَا عَبْدِ اللَّهِ مَا تَقُولُ فِي أَكْلِ فَرْخِ الزُّنْبُورِ؟ قَالَ: حَرَامٌ. فَقَالَ الْخُرَاسَانِيّ ُ: حَرَامٌ؟ فَقَالَ: نَعَمْ , مِنْ كِتَابِ اللَّهِ وَسُنَّةِ رَسُولِ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ، وَالْمَعْقُولِ. أَعُوذُ بِاللَّهِ مِنِ الشَّيْطَانِ الرَّجِيمِ. بِسْمِ اللَّهِ الرَّحْمَنِ الرَّحِيمِ: {وَمَا آتَاكُمُ الرَّسُولُ فَخُذُوهُ وَمَا نَهَاكُمْ عَنْهُ فَانْتَهُوا} [الحشر: 7] هَذَا مِنْ كِتَابِ اللهِ وَحَدَّثَنَا سُفْيَانٌ، عَنْ زَائِدَةَ، عَنْ عَبْدِ الْمَلِكِ بْنِ عُمَيْرٍ، عَنْ مَوْلَى الرَّبْعِيِّ، عَنْ حُذَيْفَةَ، أَنَّ رَسُولَ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ قَالَ: «اقْتَدُوا بِالَّذِينَ مِنْ بَعْدِي أَبِي بَكْرٍ وَعُمَرَ». هَذِهِ سُنَّةُ رَسُولِ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ.
    وأخرجه أبو نعيم (ت430) في [الإمامة/1/253/48-العلوم والحكم] فقال: حَدَّثَنَا عَبْدُ اللَّهِ بْنُ الْحُسَيْنِ بْنِ بُنْدَارٍ، ثَنَا مُحَمَّدُ بْنُ إِسْمَاعِيلَ، ثَنَا قَبِيصَةُ بْنُ عُقْبَةَ، ثَنَا سُفْيَانُ، عَنْ زَائِدَةَ، عَنْ عَبْدِ الْمَلِكِ بْنِ عُمَيْرٍ، عَنْ مَوْلَى الرَّبْعِيِّ، عَنِ الرَّبْعِيِّ، عَنْ حُذَيْفَةَ، قَالَ: قَالَ رَسُولُ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ: «اقْتَدُوا بِالَّذَيْنِ مِنْ بَعْدِي» . يُشِيرُ إِلَى أَبِي بَكْرٍ وَعُمَرَ رَضِيَ اللَّهُ عَنْهماُ وَأَرْضَاهُمَا.
    وَرَوَاهُ إِبْرَاهِيمُ بْنُ سَعِيدٍ، عَنْ سُفْيَانَ، عَنْ عَبْدِ الْمَلِكِ، عَنْ هِلَالٍ مَوْلَى الرَّبْعِيِّ، عَنْ حُذَيْفَةَ، عَنِ النَّبِيِّ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ نَحْوَهُ، ثَنَا عَبْدُ اللَّهِ، ثَنَا مُحَمَّدُ بْنُ إِسْمَاعِيلَ، حَدَّثَنَا ابْنُ كَاسِبٍ، حَدَّثَنَا إِبْرَاهِيمُ بِهِ.
    وقال البيهقي (ت458) في [الاعتقاد/340- الآفاق]: وَرَوَاهُ إِبْرَاهِيمُ بْنُ سَعْدٍ، عَنْ سُفْيَانَ، عَنْ عَبْدِ الْمَلِكِ، عَنْ هِلَالٍ مَوْلَى رِبْعِيٍّ، عَنْ حُذَيْفَةَ .اهـ
    وأخرجه ابن عساكر (ت571) في [تاريخه/5/14-الفكر] فقال: أخبرنا أبو القاسم بن السوسي أنا أبو القاسم بن أبي العلاء أنا أبو نصر المري نا أبو سعيد أحمد بن عثمان بن البقال البغداذي الفقيه من حفظه نا عبد الله بن محمد البغوي ببغداد نا مصعب بن عبد الله الزبيري نا إبراهيم بن سعد نا سفيان الثوري عن عبد الملك بن عمير عن هلال مولى ربعي عن حذيفة قال قال رسول الله -صلى الله عليه وسلم- اقتدوا باللذين من بعدي أبي بكر وعمر رضي الله عنهما .
    سقط منه عن ربعي ولا بد منه .
    أخبرناه عاليا على الصواب أبو القاسم الشحامي أنا أبو بكر محمد بن عبد الله العمري نا عبد الرحمن بن أحمد بن أبي شريح أنا أبو القاسم البغوي نا مصعب نا إبراهيم بن سعد عن سفيان بن سعيد عن عبد الملك بن عمير عن هلال مولى ربعي عن ربعي عن حذيفة عن النبي (صلى الله عليه وسلم) قال اقتدوا باللذين من بعدي أبي بكر وعمر.اهـ

    أخرجه ابن بالويه في [مجلس له/مخطوط] فقال: أَخْبَرَنَا أَبُو ظَهِيرٍ عَبْدُ اللَّهِ بْنُ فَارِسٍ الْعُمَرِيُّ الْبَلْخِيُّ، ثَنَا مَحَمَّدُ بْنُ مَنْصُورٍ الْفَقِيهُ، ثَنَا الْقَعْنَبِيُّ، عَنْ مَالِكِ بْنِ أَنَسٍ، عَنْ سَعِيدِ بْنِ أَبِي سَعِيدٍ الْمَقْبُرِيِّ، عَنْ أَبِي شُرَيْحٍ الْكَعْبِيِّ رَضِيَ اللَّهُ عَنْهُ، أَنَّ رَسُولَ اللَّهِ –صلى الله عليه وسلم- قَالَ: «اقْتَدُوا بِاللَّذِينَ مِنْ بَعْدِي أَبُو بَكْرٍ وَعُمَرُ، إِنَّ لِي وَزِيرَيْنِ فِي السَّمَاءِ وَوَزِيرَيْنِ فِي الأَرْضِ، أَمَّا فِي السَّمَاءِ جِبْرِيلُ وَمِيكَائِيلُ، وَأَمَّا فِي الأَرْضِ أَبُو بَكْرٍ وَعُمَرُ، هُمَا عِنْدِي بِمَنْزِلَةِ الرَّأْسِ مِنَ الْجَسَدِ، وَمَثَلُهُمَا فِي الأَنْبِيَاءِ بِالرَّأْفَةِ فَمَثَلُ أَبِي بَكْرٍ رَضِيَ اللَّهُ عَنْهُ كَمَثَلِ إِبْرَاهِيمَ وَعِيسَى عَلَيْهِمَا السَّلامُ إِنْ تُعَذِّبْهُمْ فَإِنَّهُمْ عِبَادُكَ وَإِنْ تَغْفِرْ لَهُمْ فَإِنَّكَ أَنْتَ الْعَزِيزُ الْحَكِيمُق، وَمَثَلُ عُمَرَ كَمَثَلِ مُوسَى وَنُوحٍف رَبِّ لا تَذَرْ عَلَى الأَرْضِ مِنَ الْكَافِرِينَ دَيَّارًا».اهـ
    عبد الله بن فارس البلخي: مجهول الحال ([1]).

    أما حديث أبي الدرداء عويمر بن مالك الأنصاري

    أخرجه الطبراني (ت360) في [مسند الشاميين/2/57/913-الرسالة] فقال: حَدَّثَنَا عَبْدُ الرَّحْمَنِ بْنُ مُعَاوِيَةَ الْعُتْبِيُّ، ثَنَا مُحَمَّدُ بْنُ نَصْرٍ الْقَارِئُ، ثَنَا أَبُو الْيَمَانِ الْحَكَمُ بْنُ نَافِعٍ، ثَنَا إِسْمَاعِيلُ بْنُ عَيَّاشٍ، عَنِ الْمُطْعِمِ بْنِ الْمِقْدَامِ الصَّنْعَانِيِّ ، عَنْ عَبْدِ اللَّهِ بْنِ عَنْبَسَةَ الْكَلَاعِيِّ، عَنْ أَبِي إِدْرِيسَ الْخَوْلَانِيِّ ، عَنْ أَبِي الدَّرْدَاءِ، قَالَ: قَالَ رَسُولُ اللَّهِ صَلَّى اللهُ عَلَيْهِ وَسَلَّمَ: «اقْتَدُوا بِاللَّذَيْنِ مِنْ بَعْدِي: أَبِي بَكْرٍ وَعُمَرَ , فَإِنَّهُمَا حَبْلُ اللَّهِ الْمَمْدُودُ , فَمَنْ تَمَسَّكَ بِهِمَا فَقَدْ تَمَسَّكَ بِعُرْوَةِ اللَّهِ الْوُثْقَى الَّتِي لَا انْفِصَامَ لَهَا».اهـ

    المشاركة الأصلية كتبت بواسطة محمد عبد الأعلى
    بارك الله فيكم
    يتبادر سؤال: هل عرض الأسانيد بهذه الطريقة مع بيان ما فيها =يكفي للقول بتضعيف الحديث؟
    بصورة أخرى: من أقسام المقبول الحديث الحسن لغيره الذي يرتقى للحسن بمجموع طرقه =فهل نفيتم عن هذه الطرق كونها مقبولة بمجموعها؟
    وما ردكم على من قام بتصحيحه أو تحسينه؟

    قال ابن حزم في [الإحكام/6/80]:     وأما الرواية اقتدوا باللذين من بعدي فحديث لا يصح لأنه مروي عن مولى لربعي مجهول وعن المفضل الضبي وليس بحجة ...ولو صح لكان عليهم لا لهم لأنهم نعني أصحاب مالك وأبي حنيفة والشافعي أترك الناس لأبي بكر وعمر وقد بينا أن أصحاب مالك خالفوا أبا بكر مما رووا في الموطأ خاصة في خمسة مواضع وخالفوا عمر في نحو ثلاثين قضية مما رووا في الموطأ خاصة وقد ذكرنا أيضا أن عمر وأبا بكر اختلفا وأن اتباعهما فيما اختلفا فيه متعذر ممتنع لا يقدر عليه أحد وإنما الصحيح في هذا الباب ما ناولنيه بعض أصحابنا وحدثنيه أيضا..عن ابن عباس قال قال النبي صلى الله عليه وسلم اعقلوا أيها الناس قولي فقد بلغت وقد تركت فيكم أيها الناس ما إن اعتصمتم به فلن تضلوا كتاب الله وسنة نبيه ..اهـ

    وقال الغزالي في [المستصفى/1/168]: وَقَدْ ذَهَبَ قَوْمٌ إلَى أَنَّ مَذْهَبَ الصَّحَابِيِّ حُجَّةٌ مُطْلَقًا، وَقَوْمٌ إلَى أَنَّهُ حُجَّةٌ إنْ خَالَفَ الْقِيَاسَ، وَقَوْمٌ إلَى أَنَّ الْحُجَّةَ فِي قَوْلِ أَبِي بَكْرٍ وَعُمَرَ خَاصَّةَ لِقَوْلِهِ - صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ -: «اقْتَدُوا بِاَللَّذَيْنِ مِنْ بَعْدِي» وَقَوْمٌ إلَى أَنَّ الْحُجَّةَ فِي قَوْلِ الْخُلَفَاءِ الرَّاشِدِينَ إذَا اتَّفَقُوا.
    وَالْكُلُّ بَاطِلٌ عِنْدَنَا فَإِنَّ مَنْ يَجُوزُ عَلَيْهِ الْغَلَطُ وَالسَّهْوُ وَلَمْ تَثْبُتْ عِصْمَتُهُ عَنْهُ فَلَا حُجَّةَ فِي قَوْلِهِ، فَكَيْفَ يُحْتَجُّ بِقَوْلِهِمْ مَعَ جَوَازِ الْخَطَأِ؟ وَكَيْفَ تُدَّعَى عِصْمَتُهُمْ مِنْ غَيْرِ حُجَّةٍ مُتَوَاتِرَةٍ؟ وَكَيْفَ يُتَصَوَّرُ عِصْمَةُ قَوْمٍ يَجُوزُ عَلَيْهِمْ الِاخْتِلَافُ؟ وَكَيْفَ يَخْتَلِفُ الْمَعْصُومَانِ ؟ كَيْفَ وَقَدْ اتَّفَقَتْ الصَّحَابَةُ عَلَى جَوَازِ مُخَالَفَةِ الصَّحَابَةِ فَلَمْ يُنْكِرْ أَبُو بَكْرٍ وَعُمَرُ عَلَى مَنْ خَالَفَهُمَا بِالِاجْتِهَادِ ، بَلْ أَوْجَبُوا فِي مَسَائِلِ الِاجْتِهَادِ عَلَى كُلِّ مُجْتَهِدٍ أَنْ يَتَّبِعَ اجْتِهَادَ نَفْسِهِ؟ فَانْتِفَاءُ الدَّلِيلِ عَلَى الْعِصْمَةِ وَوُقُوعُ الِاخْتِلَافِ بَيْنَهُمْ وَتَصْرِيحُهُمْ بِجَوَازِ مُخَالَفَتِهِمْ فِيهِ ثَلَاثَةُ أَدِلَّةٍ قَاطِعَةٍ وَلِلْمُخَالِفِ خَمْسُ شُبَهٍ:...الشُّبْ هَةُ الثَّالِثَةُ: قَوْلُهُمْ: إنَّهُ إنْ لَمْ يَجِبْ اتِّبَاعُ الْخُلَفَاءِ فَيَجِبْ اتِّبَاعُ أَبِي بَكْرٍ وَعُمَرَ بِقَوْلِهِ - صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ -: «اقْتَدُوا بِاَللَّذَيْنِ مِنْ بَعْدِي أَبِي بَكْرٍ وَعُمَرَ» . قُلْنَا: تُعَارِضُهُ الْأَخْبَارُ السَّابِقَةُ، فَيَتَطَرَّقُ إلَيْهِ الِاحْتِمَالَات ُ الثَّلَاثَةُ، ثُمَّ نَقُولُ بِمُوجِبِهِ فَيَجِبُ الِاقْتِدَاءُ بِهِمَا فِي تَجْوِيزِهِمَا لِغَيْرِهِمَا مُخَالَفَتَهُمَ ا بِمُوجِبِ الِاجْتِهَادِ. ثُمَّ لَيْتَ شِعْرِي لَوْ اخْتَلَفَا كَمَا اخْتَلَفَا فِي التَّسْوِيَةِ فِي الْعَطَاءِ فَأَيُّهُمَا يُتَّبَعُ؟اهـ

    وقال الشيخ مقبل في [أحاديث معلة ظاهرها الصحة/118/120]:     قال الترمذي رحمه الله (ج10ص147) : حَدَّثَنَا الْحَسَنُ بْنُ الصَّبَّاحِ الْبَزَّارُ حَدَّثَنَا سُفْيَانُ بْنُ عُيَيْنَةَ عَنْ زَائِدَةَ عَنْ عَبْدِ الْمَلِكِ بْنِ عُمَيْرٍ عَنْ رِبْعِيٍّ وَهُوَ ابْنُ حِرَاشٍ عَنْ حُذَيْفَةَ قَالَ قَالَ رَسُولُ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وعلى آله وَسَلَّمَ اقْتَدُوا بِاللَّذَيْنِ مِنْ بَعْدِي أَبِي بَكْرٍ وَعُمَرَ. هذا حديث حسن. ثم عقبه الترمذي بأنه منقطع وأن عبد الملك بن عمير لم يسمعه من ربعي، وإنما سمعه من مولى ربعي هلال. قال أبو عبد الرحمن: وهلال مولى ربعي مجهول، لم يرو عنه إلا عبد الملك بن عمير ولم يوثقه معتبر، وزاد المناوي في " فيض القدير" أن ابن حجر يقول: إن أبا حاتم أعله بأن ربعي بن حراش لم يسمعه من حذيفة. اهـ
    ومما ذكر أن حديث ابن مسعود وحديث أنس يشهدان له لا يصلح، لأنه منقطع وهما شديدا الضعف. والله أعلم.اهـ
