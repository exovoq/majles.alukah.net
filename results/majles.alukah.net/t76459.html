    الاختلاف على الأعمش في حديث ابن مسعود: "إني لأوعك كما يوعك رجلان منكم"

    الاختلاف على الأعمش
    في حديث ابن مسعود:
    "إني لأوعـك كما يوعـك رجلان منكم"

    علل الدارقطني 5/153-154 س(785)
    وَسُئِلَ عَنْ حَدِيثِ عَلْقَمَةَ ، عَنْ عَبْدِ اللَّهِ " دَخَلْتُ عَلَى رَسُولِ اللَّهِ وَإِنَّهُ لَيُوعَكُ ، فَقُلْتُ : مَا أَشَدَّ حُمَّاكَ ، فَقَالَ : إِنِّي لَأُوعَكُ كَمَا يُوعَكُ رَجُلَانِ مِنْكُمْ أَمَا أَنَّهُ لَيْسَ مِنْ عَبْدٍ وَلَا أَمَةٍ يَمْرَضُ مَرَضًا إِلَّا حَطَّ اللَّهُ بِهَا خَطَايَاهُ كَمَا تَتَحَافَتْ عَنِ الشَّجَرَةِ وَرَقُهَا " .
    فَقَالَ : يَرْوِيهِ الْأَعْمَشُ وَاخْتُلِفَ عَنْهُ :
    - فَرَوَاهُ النَّضْرُ بْنُ إِسْمَاعِيلَ ، عَنِ الْأَعْمَشِ ، عَنْ إِبْرَاهِيمَ ، عَنْ عَلْقَمَةَ ، عَنْ عَبْدِ اللَّهِ (1) .
    - وَرَوَاهُ شَرِيكٌ ، عَنِ الْأَعْمَشِ ، عَنْ عُمَارَةَ بْنِ عُمَيْرٍ ، عَنِ الْأَسْوَدِ ، وَهَمَّامٌ ، عَنْ عَبْدِ اللَّهِ ، قَالَ ذَلِكَ يَزِيدُ بْنُ هَارُونَ ، عَنْ شَرِيكٍ (2) .
    - وَرَوَاهُ عَمْرُو بْنُ عَبْدِ الْغَفَّارِ ، عَنِ الْأَعْمَشِ ، عَنْ خَيْثَمَةَ ، عَنْ عَبْدِ اللَّهِ (3) .
    - وَرَوَاهُ أَبُو مُعَاوِيَةَ (4) ، وَجَرِيرٌ (5) ، وَعُبَيْدَةُ بْنُ حُمَيْدٍ (6) ، وَابْنُ فُضَيْلٍ(7) ، وَعِيسَى بْنُ يُونُسَ(8) ، وَالثَّوْرِيُّ(9) ، وَابْنُ نُمَيْرٍ (10)، وَيَعْلَى بْنُ عُبَيْدٍ (11) ، عَنِ الْأَعَمْشِ ، عَنْ إِبْرَاهِيمَ التَّيْمِيِّ ، عَنِ الْحَارِثِ بْنِ سُوَيْدٍ ، عَنْ عَبْدِ اللَّهِ ، وَهُوَ الصَّحِيحُ .انتهى

    قلتُ:
    ورواه مثلهم: يَحْيَى بْنُ عَبْدِ الْمَلِكِ بن أبي غنية (12) ، و عَلِيُّ بْنُ مُسْهِرٍ (13) ، و عَبْدُ الْعَزِيزِ بْنُ مُسْلِمٍ القسمليّ (14)، وشُعبة بن الحجّاج (15) وأبو حمزة السّكريّ (16) و محمّد بن عبيد (17) وأبو يحي الحمّاني (18) ، والوليد بن القاسم (19).
    فهؤلاء ثمانية زائدة على الثمانية الذين ذكرهم الدّارقطني.

    قال أبو نعيم الأصبهاني- بعد أن خرّجه في "الحلية" 4/128 من طريق: ابن مسهر، وعيسى بن يونس، وجرير، و يحي بن عبد الملك-:" ورواه الثوري، وشعبة، وأبو معاوية، وأبو حمزة، ويعلى بن عبيد، في آخرين. والحديث متّفق على صحّته" انتهى
    __________

    (1) النّضْر بن إسماعيل بن حازم البجلي،أبو المغيرة الكوفيّ، القاص،ليس بالقوي./ت س

    - ابنُ سعد في "الطّبقات" 2/185: أخبرنا النَّضْر بن إسماعيل أبو المغيرة عن الأعمش عن إبراهيم عن علقمة قال:
    دخل عبد الله بن مسعود على النبي .فذكره

    - ابنُ أبي الدُّنيا في "المرض والكفّارات" رقم(68): حدّثنا أحمد بن منيع.
    شيخ ابن أبي الدنيا: أحمد بن منيع بن عبد الرحمن أبو جعفر البغوي الأصم ثقة حافظ مُصَنِّفُ "المسند" من شيوخ الشيخين

    - البزّارُ في "المسند" 4/328 رقم(1519): حدّثنا محمّد بن الوليد الفحّام.
    شيخ البزّار: محمّد بن الوليد الفحّام البغدادي؛ صدوقٌ، من شيوخ النّسائيّ.

    ثلاثُتهم:(محمّد بن سعد، وأحمد بن منيع، و محمّد بن الوليد الفحّام) عن النّضر بن إسماعيل البجلي به.

    قال البزّارُ: "ولا نعلمُ روى هذا الحديثَ عن الأعمشِ بهذا الإسنادِ إلاَّ النّضْر بن إسماعيل."

    رد: الاختلاف على الأعمش في حديث ابن مسعود: "إني لأوعك كما يوعك رجلان منكم"

    أقر الله عينك أخي الكريم بكل ما يسرك في الدنيا والآخرة
    قلت : رواه الأعمش واختلف عنه فرواه جرير بن عبد الحميد عن الأعمش عن إبراهيم التيمي عن الحارث بن سويد قال قال عبد الله بن مسعود : دخلت على رسول الله صلى الله عليه وسلم وهو يوعك وعكا شديدا فمسسته بيدي فقلت يا رسول الله إنك لتوعك وعكا شديدا فقال رسول الله صلى الله عليه وسلم أجل إني أوعك كما يوعك رجلان منكم فقلت ذلك أن لك أجرين فقال رسول الله صلى الله عليه وسلم أجل ثم قال رسول الله صلى الله عليه وسلم ما من مسلم يصيبه أذى مرض فما سواه إلا حط الله له سيئاته كما تحط الشجرة ورقها
    أخرجه البخاري في الصحيح (5660) ثنا قتيبة
    وأخرجه مسلم في الصحيح (6651) ثنا عثمان بن أبي شيبة وإسحاق بن إبراهيم
    وأخرجه مسلم في الصحيح (6651) وأبو يعلى في المسند (5164) وابن أبي الدنيا في المرض (2) قالوا ثنا أبو خيثمة أربعتهم قالوا ثنا جرير عن الأعمش عن إبراهيم التيمي به

    وتابع جريرا جماعة
    1/ سفيان الثوري
    أخرجه البخاري في الصحيح (5647) ثنا محمد بن يوسف
    وأخرجه البخاري في الصحيح أيضا (5661) ثنا قبيصة
    وأخرجه مسلم في الصحيح (6652) ثني محمد بن رافع ثنا عبد الرزاق
    وأخرجه النسائي في الكبرى (7441) نا يعقوب بن إبراهيم ثنا يحيى أربعتهم قالوا ثنا سفيان ثني سليمان عن إبراهيم به
    ومن طريق قبيصة أخرجه الطحاوي في شرح مشكل الآثار (2208) ومن طريق البخاري عن قبيصة أخرجه البغوي في شرح السنة (1433)

    2/ شعبة بن الحجاج
    أخرجه أحمد في المسند (4205)
    وأخرجه النسائي في الكبرى (7463) نا بشر بن خالد قالا ثنا محمد بن جعفر ثنا شعبة عن سليمان قال سمعت إبراهيم التيمي به

    3/ عيسى بن يونس 4/ ويحيى بن عبد الملك
    أخرجه مسلم في (6652) ثنا إسحاق بن إبراهيم نا عيسى بن يونس ويحيى بن عبد الملك بن أبي غنية قالا عن الأعمش عن إبراهيم به
    ومن طريق إسحاق أخرجه أبو نعيم في الحلية (4/ 128)

    5/ يعلى بن عبيد
    أخرجه أحمد في المسند (3619) والدارمي في المسند (2813) وابن سعد في الطبقات (2035)
    وأخرجه الشاشي في المسند (834) ثنا عيسى بن أحمد العسقلاني
    وأخرجه البيهقي في الكبرى[1] (6532) نا أبو محمد ثنا أبو عثمان البصري ثنا محمد بن عبد الوهاب
    وأخرجه البغوي في شرح السنة (1432) من طريق السمعاني نا أبو جعفر الرياني نا حميد بن زنجويه خمستهم قالوا ثنا يعلى بن عبيد نا الأعمش عن إبراهيم التيمي به


    6/ محمد بن عبيد
    أخرجه أحمد في المسند (4346) ثنا محمد بن عبيد ثنا الأعمش عن إبراهيم التيمي به
    ومن طريق محمد بن عبيد أيضا أخرجه البيهقي في الكبرى (13416) وفي الشعب (9315)

    7/ يحيى بن عبد الحميد
    أخرجه الشاشي في المسند (833) ثنا عباس بن محمد الدوري نا الحماني عن الأعمش عن إبراهيم التيمي به

    8/ علي بن مسهر
    أخرجه أبو نعيم في الحلية (4/ 128) ثنا أبو إسحاق إبراهيم بن محمد بن حمزة بن عمارة ثنا أبو يعلى ثنا عبد الغفار بن عبد الله ثنا علي بن مسهر عن الأعمش عن إبراهيم التيمي به

    9/ عبد العزيز بن مسلم
    أخرجه البخاري في الصحيح (5667) ثنا موسى
    وأخرجه الطحاوي في شرح مشكل الآثار (2209) ثنا أحمد بن داود ثنا عبيد الله بن محمد التيمي قالا ثنا عبد العزيز بن مسلم ثنا سليمان عن إبراهيم التيمي به
    ومن طريق موسى أبي سلمة التبوذكي أخرجه ابن السماك في الثاني من الفوائد المنتقاة (67)


    10/ محمد بن ميمون السكري
    أخرجه البخاري في الصحيح (5648) ثنا عبدان عن أبي حمزة عن الأعمش عن إبراهيم التيمي به

    11/ عبيدة بن حميد
    أخرجه ابن أبي الدنيا في المرض والكفارات (69) ثنا أحمد بن منيع ثنا عبيدة بن حميد عن الأعمش عن إبراهيم التيمي به
    ومن طريق أحمد بن منيع أخرجه المقدسي في الأمراض (8)

    12/ الوليد بن القاسم
    أخرجه أبو عوانه إتحاف المهرة (12504) عن محمد بن أحمد بن الجنيد ثنا الوليد بن القاسم يعني عن الأعمش به

    13/ محمد بن فضيل 14/ وعبد الله بن نمير
    علقه عنهما الدارقطني في العلل (5/ 154) ولم أقف عليه موصولا

    فهؤلاء خمسة عشر جرير بن عبد الحميد وسفيان الثوري وشعبة وعيسى بن يونس ويحيى بن عبد الملك وابنا عبيد ويحيى بن عبد الحميد وعلي بن مسهر وعبد العزيز بن مسلم وأبو حمزة السكري وعبيدة بن حميد والوليد بن القاسم ومحمد بن فضيل وعبد الله بن نمير رووه عن الأعمش عن إبراهيم التيمي عن الحارث بن سويد عن عبد الله بن مسعود عن رسول الله صلى الله عليه وآله وسلم

    [1]وفي الشعب (9315)

    رد: الاختلاف على الأعمش في حديث ابن مسعود: "إني لأوعك كما يوعك رجلان منكم"

    ورواه عن الأعمش أبو معاوية واختلف عنه فرواه أحمد في المسند (3618) ثنا أبو معاوية ثنا الأعمش عن إبراهيم التيمي عن الحارث بن سويد عن عبد الله عن النبي صلى الله عليه وآله وسلم
    ومن طريق أحمد أخرجه ابن الجوزي في الثبات عند الممات (ص 44)

    وتابع الإمام أحمد جمع من الحفاظ
    1/ أبو بكر بن أبي شيبة
    أخرجه ابن أبي شيبة في المصنف (10905) وفي المسند (263) وعنه مسلم في الصحيح (6652) ثنا أبو معاوية عن الأعمش عن إبراهيم التيمي به

    2/ محمد بن العلاء
    أخرجه مسلم في الصحيح (6652) والنسائي في الكبرى (7461) قالا نا محمد بن العلاء ثنا أبو معاوية ثنا الأعمش عن إبراهيم التيمي به

    3/ الطيالسي أبو داود
    أخرجه أبو داود في المسند (368) ثنا محمد بن خازم عن الأعمش عن إبراهيم التيمي به

    4/ هناد بن السري
    أخرجه هناد في الزهد (410) ثنا أبو معاوية عن الأعمش عن إبراهيم التيمي به
    ومن طريق هناد أخرجه ابن حبان في الصحيح (2937)

    5/ عثمان بن أبي شيبة
    أخرجه ابن حبان في الصحيح (2937) نا عمران بن موسى بن مجاشع ثنا عثمان بن أبي شيبة ثنا أبو معاوية عن الأعمش عن ابراهيم التيمي به

    6/ محمد بن سعد
    أخرجه ابن سعد في الطبقات (2035) نا أبو معاوية الضرير نا الأعمش عن إبراهيم التيمي به

    7/ أحمد بن عبد الجبار
    أخرجه البيهقي في الشعب[1] (9316) نا أبو عبد الله الحافظ في آخرين
    وأخرجه البيهقي في الكبرى (6531) والثقفي في الأربعين (ص 206) قالا نا أبو سعيد بن أبي عمرو
    وأخرجه البيهقي في الكبرى (13416) نا أبو عبد الرحمن محمد بن عبد الرحمن الدهان قالوا ثنا أبو العباس محمد بن يعقوب ثنا أحمد بن عبد الجبار العطاردي ثنا أبو معاوية عن الأعمش عن إبراهيم التيمي به
    ومن طريق الثقفي أخرجه الحافظ في نتائج الأفكار (4/ 215)

    8/ إبراهيم بن يوسف

    ورواه عمرو بن علي الفلاس عن أبي معاوية عن الأعمش عن عمارة بن عمير عن الأسود عن عبد الله عن النبي صلى الله عليه وآله وسلم
    أخرجه البزار في المسند (1653) ثنا عمرو بن علي نا أبو معاوية عن الأعمش عن عمارة به
    عمرو بن علي هو الفلاس أحد الأيمة الكبار
    أقول : قول الجماعة أصح فإنه قول الحفاظ عن الأعمش كما تقدم وأحسب أبا معاوية الواهم فيه فهو وإن كان مقدما في الأعمش فإن له أوهاما عنه أيضا وقد استوفى الحافظ الدارقطني الخلاف في هذا الحديث ولم يذكر قول الفلاس هذا عن أبي معاوية ؟
    لا يتجِـهُ توهيم أبي معاوية الضّرير هنا
    ورواية الفلاّس الشّاذة عنه..
    لم يروها عن الفلاّس إلاّ الحافظ أبو بكر البزار، وقد خرّجها في "مسنده"
    والبزار حافظٌ كبيرٌ ثقة.. لكنه يخطئ؛ قال الدارقطني:
    "يخطئ في الإسناد والمتن ،حدّث بالمسند بمصر حفظاً،ينظر في كتب الناس ويحدث من حفظه ولم تكن معه كتب فأخطأ في أحاديث كثيرة يتكلمون فيه."
    فلعل مارواه هنا عن الفلاس عن أبي معاوية .. من أوهامه.
    لذلك لم يذكر هذا الوجه الدّارقطني -ويبعد أن يكون لم يقف عليها-
    واقتصر على الرواية المحفوظة عن أبي معاوية.

    وأنتَ صحّحت قول الجماعة عن أبي معاوية ..= يعني أنه هو المحفوظ عنه.وقول غيره عنه خطأ
    لكن كَبُر عليك توهيم الفلاّس وتخطئته ..
    فوهّمتَ أبا معاوية ؟؟..لتعودَ إلى تصحيح كلا القولين عنه ؟؟ ويصير أبا معاوية هو المضطرب في إسناده:
    يصيب تارة، فيرويه كما رواه أصحاب الأعمش عنه
    ويهم فيه تارة فيرويه كما رواه شريك (دون زيادة هماما في إسناده).
    والله أعلم.

    المشاركة الأصلية كتبت بواسطة عدلان الجزائري
    ورواه عمرو بن عبد الغفار عن الأعمش عن خيثمة بن عبد الرحمن عن عبد الله
    أخرجه ابن شاذان في الثامن من أجزائه (74) وابن البختري في حديثه (403) قالا ثنا يحيى بن جعفر أنبا عمرو بن عبد الغفار ثنا الأعمش عن خيثمة به
    قلت: هكذا نُسب هذا الجزء في "الشّاملة" لأبي عليّ ابن شاذان وأُراه خطأ ؛
    فالحسن بن أحمد بن إبراهيم بن الحسن بن محمّد بن شاذان، أبو علي البَزَّاز (339-426هـ)
    لا يمكنه أنْ يرويَ عن يحي بن جعفر أبي طالب بن عبد الله بن الزبرقان (182-275هـ)..
    هو يروي عنه بواسطة شيوخه مثل:
    أَحْمَدَ بْنِ سَلْمَانَ النَّجَّاد (253-348هـ) ،
    أو عُثْمَانَ بْنِ أَحْمَدَ الدَّقَّاقُ المعروف بابْنِ السَّمَّاكِ (ت344هـ) ،
    أو أبي سهل بن زياد القطّان ،
    أو عبد الله بن إسحاق الخراساني..
    فالجزء لا شكّ لأحد المحدّثين من شيوخ ابن شاذان، رواه عنه.. فنسبه ناسخ المخطوطة خطأً له.
    فلتُحرر نسبته ، والله أعلم.

    رد: الاختلاف على الأعمش في حديث ابن مسعود: "إني لأوعك كما يوعك رجلان منكم"

    بعد نشر صاحبنا عدلان حفظه الله لبحثه
    سأضعُ هنا تتمةَ ما كنت أنوي نشره من بحثي
    أنشره كما كتبته قبل قراءة بحث صاحبنا:

    (2) شريك بن عبد الله النخَعي الكُوفيّ، أبو عبد الله: صدوق يُخطئ كثيراً./خت م 4

    - البزّار في "المسند" 5/315 رقم (1933): حدّثنا أحمد بن سنان، ومحمّد بن عثمان بن مخلد، قالا: نا يزيد بن هارون، قال: أنا شريك.

    - الدّارقطني في "العلل" 5/155: حدّثنا عليّ بن محمّد السواق، قال: ثنا أحمد بن كثير أبو نافع، قال: ثنا يزيد بن هارون، أنبأ شريك، عن الأعمش، عن عمارة بن عمير، عن الأسود، وهمّام بن الحارث، عن عبد الله.

    قال البزّارُ: "وهذا الحديثُ قد رواهُ غيرُ شريك، عن الأعمش، عن عمارة، عن الأسود، عن عبد الله، وزاد شريكٌ عن الأعمش عن عمارة عن الأسود وهمّام." انتهى

    قلتُ:يشير إلى رواية أبي معاوية التي قدّم إخراجها في:
    - "مسنده" 5/81 رقم(1653):حدّثنا عمرو بن عليّ، قال: نا أبو معاوية، عن الأعمش، عن عمارة بن عمير، عن الأسود، عن عبد الله.

    هكذا قال البزّار عن عمرو بن علي عن أبي معاوية؛ لم يزد همّاما في الإسناد مجموعاً إلى الأسود..
    وهو وجهٌ آخر من الاختلاف عن الأعمش، لم يذكرْه الدّارقطني هنا.
    وقد رواهُ جماعة من الأئمة-كما سيأتي-:
    عن أبي معاوية عن الأعمش عن إبراهيم عن الحارث عن عبد الله..
    وهذا الوجه عن أبي معاوية هو المحفوظ عنه
    وهو الذي اقتصر عليه الدارقطني ..
    وهو الموافق لما رواه أصحاب الأعمش عنه
    وما رواه الفلاس تفرد بإخراجه عنه البزار في "مسنده" ، وأراه من أخطائه..
    <انقل كلام الدارقطني فيه و في مسنده>
    قلتُ:
    وقد رواه: إسرائيل بن يونس بن أبي إسحاق السبيعي مثل شريك ؛ قال ذلك يزيد بن هارون أيضاً:
    - ابن أبي الدّنيا في "المرض والكفّارات" رقم (225): حدّثني الفضل بن جعفر، حدّثنا يزيد بن هارون، أنبأنا إسرائيل، عن الأعمش، عن عمارة بن عمير، عن الأسود، وهمّام بن الحارث، عن عبد الله .
    الفضل بن جعفر بن عبد الله البغدادي أبو سهل بن أبي طالب أخو يحي بن أبي طالب واسطي الأصل ثقة مات سنة 252هـ وله 66 سنة روى عنه الترمذي.

    (3) عَمْرُو بن عبد الغفّار الفقيمي: يروي عن الأعمش وغيره ؛ قال أبو حاتم: متروك الحديث، وقال ابن عدي: متّهم إذا روى شيئاً في الفضائل، وقال ابن المديني:تركته لأجل الرفض، وقال العقيلي وغيره: منكر الحديث.

    - أبو عليّ ابن شاذان في "الثامن من أجزائه" رقم(74): حَدَّثَنَا يَحْيَى بْنُ جَعْفَرٍ , قَالَ : أنبا عَمْرُو بْنُ عَبْدِ الْغَفَّارِ , قَالَ : ثنا الأَعْمَشُ , عَنْ خَيْثَمَةَ بْنِ عَبْدِ الرَّحْمَنِ , قَالَ : قَالَ عَبْدُ اللَّهِ : دَخَلْتُ عَلَى رَسُولِ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ..فذكره .

    قلت: هكذا نُسب هذا الجزء في "الشّاملة" لأبي عليّ ابن شاذان وأراه خطأ ؟؟ فالحسن بن أحمد بن إبراهيم بن الحسن بن محمّد بن شاذان، أبو علي البَزَّاز (339-426هـ) لا يمكنه أنْ يرويَ عن يحي بن جعفر أبي طالب (182-275هـ).. هو يروي عنه بواسطة شيوخه مثل: أَحْمَدَ بْنِ سَلْمَانَ النَّجَّاد (253-348هـ) ، أو عُثْمَانَ بْنِ أَحْمَدَ الدَّقَّاقُ المعروف بابْنِ السَّمَّاكِ (ت344هـ) ، أو أبي سهل بن زياد القطّان ، أو عبد الله بن إسحاق الخراساني.

    فالجزء لا شكّ لأحد المحدّثين من شيوخ ابن شاذان، رواه عنه.. فنسبه ناسخ المخطوطة خطأ له.والله أعلم

    - أبو جعفر ابن البَخْتَري الرَّزَّاز في "الجزء الرابع من حديثه" رقم(159)-ضمن مجموع فيه مصنفات أبي جعفر ابن البختري ص319/رقم(403) قال: حَدَّثَنَا يَحْيَى بْنُ جَعْفَرٍ , أَخْبَرَنَا عَمْرُو بْنُ عَبْدِ الْغَفَّارِ , أَخْبَرَنَا الأَعْمَشُ ، عَنْ خَيْثَمَةَ بْنِ عَبْدِ الرَّحْمَنِ ، قَالَ : قَالَ عَبْدُ اللَّهِ : دَخَلْتُ عَلَى رَسُولِ اللَّهِ صَلَّى اللَّهُ عَلَيْهِ وَسَلَّمَ وَهُوَ يُوعَكُ وَعْكًا شَدِيدًا..فذكره.

    قلتُ: يحي بن أبي طالب جعفر بن عبد الله بن الزبرقان ، أبو بكر البغدادي (ت275هـ): الإمام المحدّث العالم، وثّقه الدّارقطني وغيره. ترجمته في الجرح والتعديل: 9/134 ، "تاريخ بغداد": 14/220 ، "تاريخ الإسلام": 6/638 ، "السير": 12/619 ، "ميزان الاعتدال":4/386 "لسان الميزان": 6/322
